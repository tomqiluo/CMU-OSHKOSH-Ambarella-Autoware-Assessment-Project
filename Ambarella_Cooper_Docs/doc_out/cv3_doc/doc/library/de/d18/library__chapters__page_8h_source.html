13984 * @subsection SmartFB_structure 2.5 FrameBuffer Common Structure
13985 *
13986 * FrameBuffer has the following commonly used structures: "fb_var_screeninfo",
13987 * "fb_fix_screeninfo", "fb_cmap", "ioctl". These structures are declared in
13988 * `ambarella/kernel/linux-4.14/include/uapi/linux/fb.h`.
13989 *
13990 * 1.<b>fb_var_screeninfo</b>
13991 *
13992 * This structure is used to record user-modifiable display attribute parameters,
13993 * including screen resolution, the number of bits per pixel, and so on.
13994 *
13995 * There are some frequently used information in this structure, such as:
13996 * offset (xoffset, yoffset), visible rows and columns of pixels (xres, yres), the
13997 * number of bits per pixel (bits_per_pixel), virtual resolution (xres_virtual, Yres_virtual).
13998 *
13999 * Ambarella can obtain these data through the ioctl() function, such as:
14000 * `ioctl(fbfd, FBIOGET_FSCREENINFO, &finfo)`. We can also re-assign and then set it into the
14001 * system through ioctl(), such as: `ioctl(fbfd, FBIOPUT_VSCREENINFO, &finfo)`.
14002 *
14003 * @image html SmartFB_fb_var_screeninfo.jpg "fb_var_screeninfo"
14004 *
14005 * @code
14006 * struct fb_var_screeninfo {
14007 * __u32 xres; /* visible resolution */
14008 * __u32 yres;
14009 * __u32 xres_virtual; /* virtual resolution */
14010 * __u32 yres_virtual;
14011 * __u32 xoffset; /* offset from virtual to visible */
14012 * __u32 yoffset; /* resolution */
14013 *
14014 * __u32 bits_per_pixel; /* guess what */
14015 * __u32 grayscale; /* 0 = color, 1 = grayscale, */
14016 * /* >1 = FOURCC */
14017 * struct fb_bitfield red; /* bitfield in fb mem if true color, */
14018 * struct fb_bitfield green; /* else only length is significant */
14019 * struct fb_bitfield blue;
14020 * struct fb_bitfield transp; /* transparency */
14021 *
14022 * __u32 nonstd; /* != 0 Non standard pixel format */
14023 *
14024 * __u32 activate; /* see FB_ACTIVATE_* */
14025 *
14026 * __u32 height; /* height of picture in mm */
14027 * __u32 width; /* width of picture in mm */
14028 *
14029 * __u32 accel_flags; /* (OBSOLETE) see fb_info.flags */
14030 *
14031 * /* Timing: All values in pixclocks, except pixclock (of course) */
14032 * __u32 pixclock; /* pixel clock in ps (pico seconds) */
14033 * __u32 left_margin; /* time from sync to picture */
14034 * __u32 right_margin; /* time from picture to sync */
14035 * __u32 upper_margin; /* time from sync to picture */
14036 * __u32 lower_margin;
14037 * __u32 hsync_len; /* length of horizontal sync */
14038 * __u32 vsync_len; /* length of vertical sync */
14039 * __u32 sync; /* see FB_SYNC_* */
14040 * __u32 vmode; /* see FB_VMODE_* */
14041 * __u32 rotate; /* angle we rotate counter clockwise */
14042 * __u32 colorspace; /* colorspace for FOURCC-based modes */
14043 * __u32 reserved[4]; /* Reserved for future compatibility */
14044 * };
14045 * @endcode
14046 *
14047 * 2.<b>fb_fix_screeninfo</b>
14048 *
14049 * This structure is mainly described with some fixed parameters.
14050 * These parameters are set during driver initialization and cannot be modified by the user.
14051 * For example: memory space size, memory start address, the number of bytes occupied by each line and so on.
14052 *
14053 * @code
14054 * struct fb_fix_screeninfo {
14055 * char id[16]; /* identification string eg "TT Builtin" */
14056 * unsigned long smem_start; /* Start of frame buffer mem */
14057 * /* (physical address) */
14058 * __u32 smem_len; /* Length of frame buffer mem */
14059 * __u32 type; /* see FB_TYPE_* */
14060 * __u32 type_aux; /* Interleave for interleaved Planes */
14061 * __u32 visual; /* see FB_VISUAL_* */
14062 * __u16 xpanstep; /* zero if no hardware panning */
14063 * __u16 ypanstep; /* zero if no hardware panning */
14064 * __u16 ywrapstep; /* zero if no hardware ywrap */
14065 * __u32 line_length; /* length of a line in bytes */
14066 * unsigned long mmio_start; /* Start of Memory Mapped I/O */
14067 * /* (physical address) */
14068 * __u32 mmio_len; /* Length of Memory Mapped I/O */
14069 * __u32 accel; /* Indicate to driver which */
14070 * /* specific chip/card we have */
14071 * __u16 capabilities; /* see FB_CAP_* */
14072 * __u16 reserved[2]; /* Reserved for future compatibility */
14073 * };
14074 * @endcode
14075 *
14076 * 3.<b>fb_cmap</b>
14077 *
14078 * This structure is mainly used to describe color mapping information.
14079 * The color mapping information can be set or get through the ioctl()
14080 * operation corresponding to FBIOGETCMAP and FBIOPUTCMAP.
14081 *
14082 * @code
14083 * struct fb_cmap {
14084 * __u32 start; /* First entry */
14085 * __u32 len; /* Number of entries */
14086 * __u16 *red; /* Red values */
14087 * __u16 *green;
14088 * __u16 *blue;
14089 * __u16 *transp; /* transparency, can be NULL */
14090 * };
14091 * @endcode
14092 *
14093 * 4.<b>parameters_of_ioctl()</b>
14094 *
14095 * The commonly used request parameters in ioctl().
14096 *
14097 * @code
14098 * - FBIOGET_VSCREENINFO /* indicates that the user obtains the variable parameters of the screen; */
14099 * - FBIOPUT_VSCREENINFO /* indicates that the user sets variable screen parameters; */
14100 * - FBIOGET_FSCREENINFO /* indicates that the user obtains fixed parameters of the screen; */
14101 * - FBIOBLANK /* clears the screen */
14102 * - FBIOPUTCMAP /* sets the color table of the screen */
14103 * - FBIOGETCMAP /* get color table */
14104 * @endcode
