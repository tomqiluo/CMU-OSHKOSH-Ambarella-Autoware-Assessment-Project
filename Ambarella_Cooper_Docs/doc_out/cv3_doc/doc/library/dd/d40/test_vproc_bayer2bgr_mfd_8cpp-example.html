void usage(void)
{
unsigned long i;
printf("test_vproc_cclb usage:\n");
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
}
int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, height, width;
input_df.sign = 0;
input_df.datasize = 0;
input_df.exp_offset = 0;
input_df.exp_bits = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index))
!= -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(vproc_bin_path, sizeof(vproc_bin_path), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(bayer_input_file_path, sizeof(bayer_input_file_path), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("filepath [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 'c':
out_pattern = atoi(optarg);
break;
case 's':
get_vect_resolution(optarg, &width, &height);
iw = width;
ih = height;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}

static int create_vect_mfd(vect_desc_mfd_t* vect,cv_mem_mfd_t* vm,
int h, int w, int d, int data_size, int with_cache)
{
memset((void*)vect,0,sizeof(vect_desc_mfd_t));
vect->shape.p = 1;
vect->shape.d = d;
vect->shape.h = h;
vect->shape.w = w;
vect->pitch = ALIGN_PITCH(w * data_size);
vect->color_space = CS_VECT;
vm->size = vect->pitch * vect->shape.h * vect->shape.d;
if (cavalry_mem_alloc_mfd(vm->size, &vm->mfd, &vm->virt, with_cache) < 0) {
perror("cavalry_mem_alloc_mfd err\n");
return -1;
}
vect->data_addr_fd = vm->mfd;
vect->data_addr_offset = 0;
return vm->size;
}
static int free_vmem_mfd(cv_mem_mfd_t* vm)
{
if (vm->virt != NULL) {
if (cavalry_mem_free_mfd(vm->size , vm->mfd, vm->virt) < 0) {
printf("cavalry_mem_free_mfd err\n");
return -1;
}
}
return 0;
}
static int load_bayer_ufix8_0_bin(char *bayer_file, uint8_t *dst_addr,
vect_desc_mfd_t *vect_desc, uint32_t *size)
{
uint32_t file_size = 0;
char c;
size_t t;
FILE *B=fopen(bayer_file,"rb");
while( !feof( B ) ){
t = fread( &c, 1, 1, B );
if ( t < 0 ){
fclose(B);
return 0;
}
file_size++;
}
file_size=file_size-1;
printf("bayer file size=%d\n",file_size);
fclose(B);
if (vect_desc->shape.h * vect_desc->shape.w != file_size){
printf("bayer file size mismatch\n");
exit(-1);
}
uint8_t *bayer = (uint8_t *)malloc(vect_desc->shape.h * vect_desc->shape.w * sizeof(uint8_t));
B=fopen(bayer_file, "rb");
t = fread( bayer, file_size, 1, B );
fclose(B);
if ( t < 0 ){
free(bayer);
bayer = NULL;
return 0;
}
for (uint32_t i = 0; i < vect_desc->shape.h; i++) {
for (uint32_t j = 0; j < vect_desc->shape.w; j++) {
dst_addr[i * vect_desc->pitch + j] = bayer[i * vect_desc->shape.w + j];
}
}
*size = vect_desc->pitch * vect_desc->shape.h;
free(bayer);
bayer = NULL;
return (*size);
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int dump_bgr_interleaved(char *file, vect_desc_mfd_t * vect_bgr, void *data_bgr)
{
uint8_t* bgr_interleaved_data = (uint8_t*)malloc(vect_bgr->shape.h * vect_bgr->shape.w * 3 *sizeof(uint8_t));
memset(bgr_interleaved_data, 0 , vect_bgr->shape.h * vect_bgr->shape.w * 3 );
uint32_t bgr_chanel_size = vect_bgr->shape.h * vect_bgr->pitch;
uint8_t *b = (uint8_t *)data_bgr;
uint8_t *g = (uint8_t *)data_bgr + bgr_chanel_size;
uint8_t *r = (uint8_t *)data_bgr + bgr_chanel_size * 2;
for (uint32_t i = 0; i < vect_bgr->shape.h; i++){
for (uint32_t j = 0; j < vect_bgr->shape.w; j++){
bgr_interleaved_data[i * vect_bgr->shape.w * 3 + j * 3] = b[i * vect_bgr->pitch + j];
bgr_interleaved_data[i * vect_bgr->shape.w * 3 + j * 3 + 1] = g[i * vect_bgr->pitch + j];
bgr_interleaved_data[i * vect_bgr->shape.w * 3 + j * 3 + 2] = r[i * vect_bgr->pitch + j];
}
}
store_bin(file, vect_bgr->shape.h * vect_bgr->shape.w * 3, bgr_interleaved_data);
free(bgr_interleaved_data);
bgr_interleaved_data = NULL;
return 0;
}
int main(int argc, char **argv)
{
struct timeval tv1, tv2;
struct vproc_version ver;
cv_mem_mfd_t lib_mem;
cv_mem_mfd_t bayer_in_mem;
vect_desc_mfd_t bayer_in_vect;
cv_mem_mfd_t bgr_out_mem;
vect_desc_mfd_t bgr_out_vect;
void* test_handler = NULL;
uint32_t size;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open("/dev/cavalry", O_RDWR, 0)) < 0) {
perror("/dev/cavalry");
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
memset(&lib_mem, 0, sizeof(lib_mem));
if (vproc_init(vproc_bin_path, &size) < 0) {
printf("vproc_init err\n");
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc_mfd(lib_mem.size, &lib_mem.mfd, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc_mfd err\n");
break;
}
vproc_load_mfd(fd_cavalry, (uint8_t*)lib_mem.virt, lib_mem.mfd, size);
bayer2bgr_cfg_t cfg;
cfg.height = ih;
cfg.width = iw;
cfg.out_pattern = out_pattern;
if ((test_handler = vproc_bayer2bgr_init(&cfg)) == NULL){
printf("vproc_bayer2bgr_init err\n");
break;
}
create_vect_mfd(&bayer_in_vect, &bayer_in_mem, ih, iw, 1, sizeof(uint8_t), 1);
create_vect_mfd(&bgr_out_vect, &bgr_out_mem, ih, iw, 3, sizeof(uint8_t), 1);
memset(bgr_out_mem.virt,0,bgr_out_vect.pitch * bgr_out_vect.shape.h * 3);
cavalry_mem_sync_cache_mfd(bgr_out_mem.size, 0, bgr_out_mem.mfd, 1, 0);
bayer_in_vect.data_format.datasize = 0;
bayer_in_vect.data_format.sign = 0;
bayer_in_vect.data_format.exp_offset = 0;
bayer_in_vect.data_format.exp_bits = 0;
bgr_out_vect.data_format.datasize = 0;
bgr_out_vect.data_format.sign = 0;
bgr_out_vect.data_format.exp_offset = 0;
bgr_out_vect.data_format.exp_bits = 0;
load_bayer_ufix8_0_bin(bayer_input_file_path, (uint8_t*)bayer_in_mem.virt, &bayer_in_vect, &size);
cavalry_mem_sync_cache_mfd(bayer_in_mem.size, 0, bayer_in_mem.mfd, 1, 0);
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
vproc_bayer2bgr_mfd(test_handler, &bayer_in_vect, &bgr_out_vect);
cavalry_mem_sync_cache_mfd(bgr_out_mem.size, 0, bgr_out_mem.mfd, 0, 1);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("vproc_bayer2bgr_mfd() takes %ld us\n",
(tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
if (output_flag) {
char fp[1500];
snprintf(fp, sizeof(fp),"%s/bgr_out.bin",output_name);
if (store_bin(fp, bgr_out_mem.size, bgr_out_mem.virt) < 0) {
printf("store bgr_out.bin fails!\n");
break;
}
snprintf(fp, sizeof(fp),"%s/bgr_interleaved_out.bin",output_name);
if (dump_bgr_interleaved(fp, &bgr_out_vect, bgr_out_mem.virt) < 0) {
printf("store bgr_interleaved_out.bin fails!\n");
break;
}
}
}while(0);
vproc_bayer2bgr_release(test_handler);
free_vmem_mfd(&bayer_in_mem);
free_vmem_mfd(&bgr_out_mem);
vproc_exit();
return 0;
}
