static void usage(void)
{
u32 i;
char itself[]="test_smartrc_v3";
printf("%s Usage:\n", itself);
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
printf("\nExamples:\n"
" 1. Enable uncovered area and static area module in stream A, "
"mark spots referred the uncovered area and static area:\n"
" # %s -A -m 4 \n\n", itself);
printf(" 2. Disable motion detection from cv output bbox in stream A :\n"
" # %s -A -r 15 --frame-control 1 --roi-control 1 --cv-enable 1 " \
"--cv-src 0 -t 0 -l 3 --disable-motion-from-cv 1\n\n", itself);
printf(" 3. Applying cv output in stream A with non frame sync mode:\n"
" # %s -A -r 15 --frame-control 1 --roi-control 1 --cv-enable 1 " \
"--cv-src 0 -t 0 -l 3 --disable-frame-sync-for-cv 1\n\n", itself);
printf(" 4. Enable applying ROI params for B frame when there is B frame in stream A :\n"
" # %s -A -r 15 --frame-control 1 --roi-control 1 -l 3 " \
"--enable-roi-for-B-frame 1\n\n", itself);
printf(" 5. Disable low light profile in stream A:\n"
" # %s -A -r 15 --frame-control 1 --roi-control 1 -l 3 " \
"--disable-low-light-profile 1\n\n", itself);
printf(" 6. Enable img scale downscale me0 in stream A:\n"
" # %s -A -r 15 --frame-control 1 --roi-control 1 -l 3 " \
"--enable-dsp-downscale-me0 1\n\n", itself);
}
static int init_param(int argc, char **argv)
{
int ch, tmp;
int option_index = 0;
smartrc_init_params_t *init_params = &smartrc_params.init_params;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'A':
cur_stream_id = 0;
init_params->process_stream_map |= (1U << cur_stream_id);
break;
case 'B':
cur_stream_id = 1;
init_params->process_stream_map |= (1U << cur_stream_id);
break;
case 'C':
cur_stream_id = 2;
init_params->process_stream_map |= (1U << cur_stream_id);
break;
case 'D':
cur_stream_id = 3;
init_params->process_stream_map |= (1U << cur_stream_id);
break;
case 'S':
tmp = atoi(optarg);
VERIFY_STREAM_ID(tmp);
cur_stream_id = tmp;
init_params->process_stream_map |= (1U << tmp);
break;
case 'l':
tmp = atoi(optarg);
VERIFY_LOG_LEVEL(tmp);
init_params->log_level = tmp;
break;
case SPECIFY_CV_ENABLE:
tmp = !!atoi(optarg);
init_params->cv_enable_stream_map |= tmp ?
(1U << cur_stream_id) : (0U << cur_stream_id);
break;
case SPECIFY_DISABLE_MOTION_FROM_CV:
tmp = !!atoi(optarg);
init_params->disable_motion_from_cv_stream_map |= tmp ?
(1U << cur_stream_id) : (0U << cur_stream_id);
break;
case SPECIFY_DISABLE_FRAME_SYNC_FOR_CV:
tmp = !!atoi(optarg);
smartrc_params.disable_frame_sync_for_cv_flags |= tmp ?
(1U << cur_stream_id) : (0U << cur_stream_id);
break;
case SPECIFY_ENABLE_ROI_FOR_B_FRAME:
tmp = !!atoi(optarg);
init_params->enable_roi_for_B_frame_stream_map |= tmp ?
(1U << cur_stream_id) : (0U << cur_stream_id);
break;
case SPECIFY_DISABLE_LOW_LIGHT_PROFILE:
tmp = !!atoi(optarg);
init_params->disable_low_light_profile_stream_map |= tmp ?
(1U << cur_stream_id) : (0U << cur_stream_id);
break;
case SPECIFY_ENABLE_DSP_DOWNSCALE_ME0:
tmp = !!atoi(optarg);
init_params->enable_dsp_scale_me0_stream_map |= tmp ?
(1U << cur_stream_id) : (0U << cur_stream_id);
break;
case SPECIFY_DEBUG_PROCESSING_TIME:
tmp = !!atoi(optarg);
init_params->debug_flag_map |= tmp ?
(1U << SMARTRC_DEBUG_TYPE_PROCESSING_TIME) : (0U << SMARTRC_DEBUG_TYPE_PROCESSING_TIME);
break;
case SPECIFY_CV_SRC:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
init_params->cv_src[cur_stream_id] = tmp;
break;
case SPECIFY_CV_OVERLAY_ENBALE:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
smartrc_params.bbox[cur_stream_id].enable = !!tmp;
smartrc_params.bbox_update_flags |= (1U << cur_stream_id);
break;
case SPECIFY_CV_OVERLAY_COLOR:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
if ((tmp < BBOX_COLOR_FIRST) || (tmp >= BBOX_COLOR_LAST)) {
printf("Invalid CV overlay color [%d], out of range [%d, %d).",
tmp, BBOX_COLOR_FIRST, BBOX_COLOR_LAST);
return -1;
}
smartrc_params.bbox[cur_stream_id].color = tmp;
smartrc_params.bbox_update_flags |= (1U << cur_stream_id);
break;
case SPECIFY_FRAME_CONTROL:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
smartrc_params.frame_ctrl_enable[cur_stream_id] = !!tmp;
smartrc_params.frame_ctrl_enable_flags |= (1U << cur_stream_id);
break;
case SPECIFY_ROI_CONTROL:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
smartrc_params.roi_ctrl_enable[cur_stream_id] = !!tmp;
smartrc_params.roi_ctrl_enable_flags |= (1U << cur_stream_id);
break;
case SPECIFY_DYNAMIC_GOP_ENABLE:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
smartrc_params.dynamic_gop_enable[cur_stream_id] = !!tmp;
smartrc_params.dynamic_gop_enable_flags |= (1U << cur_stream_id);
break;
case SPECIFY_FORCE_CTRL_ENABLE:
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
if (tmp > ((1 << CONTROL_NUM) - 1)) {
printf("Invalid force control map %d\n", tmp);
return -1;
}
smartrc_params.force_control_map[cur_stream_id] = tmp;
smartrc_params.force_control_map_flags |= (1U << cur_stream_id);
break;
case SPECIFY_LUA_FILE:
VERIFY_STREAM_ID(cur_stream_id);
snprintf(smartrc_params.lua_file[cur_stream_id].path,
SMARTRC_MAX_FILE_PATH_LENGTH, "%s", optarg);
smartrc_params.lua_file_flags |= (1U << cur_stream_id);
break;
case SPECIFY_LUA_DUMP:
VERIFY_STREAM_ID(cur_stream_id);
snprintf(smartrc_params.lua_dump[cur_stream_id].path,
SMARTRC_MAX_FILE_PATH_LENGTH, "%s", optarg);
smartrc_params.lua_dump_flags |= (1U << cur_stream_id);
break;
case SPECIFY_DEBUG_SKIP_VERISION_CHECK:
VERIFY_STREAM_ID(cur_stream_id);
debug_disable_ver_check = 1;
break;
case 'm':
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
VERIFY_MAP_TYPE(tmp);
cur_map_type = tmp;
smartrc_params.mask[cur_stream_id][tmp].enable = 1;
smartrc_params.mask[cur_stream_id][tmp].map_type = tmp;
smartrc_params.mask_type_flags |= (1U << tmp);
smartrc_params.mask_update_flags |= (1U << cur_stream_id);
break;
case 'c':
VERIFY_STREAM_ID(cur_stream_id);
VERIFY_MAP_TYPE(cur_map_type);
tmp = atoi(optarg);
if ((tmp < MASK_BLACK_COLOR) || (tmp > MASK_WHITE_COLOR)) {
printf("Invalid map color %d\n", tmp);
return -1;
}
smartrc_params.mask[cur_stream_id][cur_map_type].color = tmp;
smartrc_params.mask_update_flags |= (1U << cur_stream_id);
break;
case 'd':
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
if ((tmp < FILE_TYPE_FIRST) || (tmp > FILE_TYPE_LAST)) {
printf("Invalid dump file type %d\n", tmp);
return -1;
}
smartrc_params.file[cur_stream_id].type = tmp;
smartrc_params.file_update_flags |= (1U << cur_stream_id);
break;
case 'f':
VERIFY_STREAM_ID(cur_stream_id);
snprintf(smartrc_params.file[cur_stream_id].path,
SMARTRC_MAX_FILE_PATH_LENGTH, "%s", optarg);
smartrc_params.file_update_flags |= (1U << cur_stream_id);
break;
case 't':
tmp = atoi(optarg);
if ((tmp < AUTO_TEST_FIRST_TYPE) || (tmp >= AUTO_TEST_LAST_TYPE)) {
printf("Invalid auto test mode %d\n", tmp);
return -1;
}
smartrc_params.auto_test_flags |= (1U << tmp);
break;
case 'r':
VERIFY_STREAM_ID(cur_stream_id);
tmp = atoi(optarg);
if (tmp > ((1 << MAP_TYPE_NUM) - 1)) {
printf("Invalid control ref maps %d\n", tmp);
return -1;
}
smartrc_params.control_ref_map[cur_stream_id] = tmp;
smartrc_params.control_ref_map_flags |= (1U << cur_stream_id);
break;
default:
printf("Unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int smartrc_check_iav_version(void)
{
smartrc_iav_info_t iav_info = {0};
struct iav_queryinfo query_driver_info = {0};
struct iav_driver_version *driver_info = NULL;
u32 iav_api_version = 0;
int fd_iav = -1;
int rval = 0;
if ((fd_iav = open("/dev/iav", O_RDWR, 0)) < 0) {
perror("/dev/iav");
rval = SMARTRC_EFAULT;
goto EXIT_SMARTRC_CHECK_IAV_VERSION;
}
memset(&iav_info, 0, sizeof(smartrc_iav_info_t));
memset(&query_driver_info, 0, sizeof(query_driver_info));
query_driver_info.qid = IAV_INFO_DRIVER;
if (ioctl(fd_iav, IAV_IOC_QUERY_INFO, &query_driver_info) == 0) {
driver_info = &query_driver_info.arg.driver;
iav_api_version = driver_info->iav_api_version;
iav_info.iav_api_verison.major = (iav_api_version & 0xFF0000) >> 16;
iav_info.iav_api_verison.minor = (iav_api_version & 0xFF00) >> 8;
iav_info.iav_api_verison.patch = (iav_api_version & 0xFF);
iav_info.iav_api_verison.mod_time = driver_info->mod_time;
}
iav_info.sys_res_size = (u32)sizeof(struct iav_system_resource);
iav_info.stream_cfg_size = (u32)sizeof(struct iav_stream_cfg);
iav_info.stream_format_size = (u32)sizeof(struct iav_stream_format);
rval = smartrc_check_iav_info(&iav_info);
if (rval != SMARTRC_SUCCESS) {
printf("Smartrc: IAV info check failed, please update library.\n");
goto EXIT_SMARTRC_CHECK_IAV_VERSION;
}
EXIT_SMARTRC_CHECK_IAV_VERSION:
if (fd_iav >= 0) {
close(fd_iav);
fd_iav = -1;
}
return rval;
}
static int init_smartrc_lib(smartrc_params_t *smartrc_param)
{
smartrc_version_t version = {0};
smartrc_mask_info_t *mask = NULL;
smartrc_file_info_t *file = NULL;
smartrc_bbox_info_t *bbox = NULL;
smartrc_init_params_t *init_params = &smartrc_param->init_params;
smartrc_params_table_t param_table;
int i = 0, j = 0, rval = 0;
u8 map_valid_data[MAP_TYPE_NUM] = {0x01, 0xFF, 0x01, 0x01};
do {
rval = smartrc_get_version(&version);
if (rval != SMARTRC_SUCCESS) {
printf("Smartrc get version failed.\n");
break;
} else {
printf("\nSmart Rate Control Library Version: %s-%d.%d.%d (Last updated: %x)\n\n",
version.description, version.major, version.minor, version.patch, version.mod_time);
}
if (!debug_disable_ver_check) {
rval = smartrc_check_iav_version();
if (rval != SMARTRC_SUCCESS) {
printf("Smartrc check iav version failed.\n");
break;
}
}
rval = smartrc_init_lib(init_params);

static void *smartrc_auto_config_stream(void *arg)
{
smartrc_thread_t *test_thread = &auto_test_threads[AUTO_CONFIG_STREAM_RANDOM_TEST];
u32 force_control_map_flags = smartrc_params.force_control_map_flags;
u32 force_control_map = 0;
smartrc_params_table_t params_table = {0};
smartrc_frame_params_table_t *frame_table = NULL;
int sleep_seconds = 0, count = 0, index = 0;
int rval = 0, i = 0;
arg = arg; /* This is to fix compile warning */
while(!test_thread->stop_flag) {
for (i = 0; i < SMARTRC_MAX_STREAM_NUM; i++) {
if (force_control_map_flags & (1U << i)) {
rval = smartrc_get_params_table(i, &params_table);
if (rval == SMARTRC_EFAULT) {
printf("Get params table loop %d for stream %c failed.\n", count,
'A' + i);
}
force_control_map = smartrc_params.force_control_map[i];
if (force_control_map | (0x01 << CONTROL_GOP)) {
for (index = PROFILE_FIRST; index < PROFILE_NUM; index++) {
frame_table = &params_table.frame_table[index];
if (count % 2 == 0) {
frame_table->gop.N = frame_table->gop.N + 15;
} else {
frame_table->gop.N = frame_table->gop.N - 15;
}
}
}
if (force_control_map | (0x01 << CONTROL_BITRATE)) {
for (index = PROFILE_FIRST; index < PROFILE_NUM; index++) {
frame_table = &params_table.frame_table[index];
if (count % 2 == 0) {
frame_table->qp_params.bitrate += 100000;
} else {
frame_table->qp_params.bitrate -= 100000;
}
}
}
if (force_control_map | (0x01 << CONTROL_FORCE_IDR)) {
for (index = PROFILE_FIRST; index < PROFILE_NUM; index++) {
frame_table->force_idr = count % 2;
}
}
if (force_control_map | (0x01 << CONTROL_FORCE_PSKIP)) {
for (index = PROFILE_FIRST; index < PROFILE_NUM; index++) {
frame_table = &params_table.frame_table[index];
frame_table->skip_frame.repeat_num = count % 2 ? 15 : 0;
}
}
if (force_control_map | (0x01 << CONTROL_FORCE_DROP_FRAME)) {
for (index = PROFILE_FIRST; index < PROFILE_NUM; index++) {
frame_table = &params_table.frame_table[index];
frame_table->drop_frame.drop_num = count % 2 ? 1 : 0;
}
}
if (force_control_map | (0x01 << CONTROL_ZMV)) {
for (index = PROFILE_FIRST; index < PROFILE_NUM; index++) {
frame_table = &params_table.frame_table[index];
if (count % 2 == 0) {
frame_table->zmv_params.zmv_threshold_qp_offset += 5;
} else {
frame_table->zmv_params.zmv_threshold_qp_offset -= 5;
}
}
}
rval = smartrc_set_params_table(i, &params_table);
if (rval == SMARTRC_EFAULT) {
printf("Start process loop %d for stream %c failed.\n", count,
'A' + i);
}
count++;
sleep_seconds = 5 + (rand()%5);
usleep(sleep_seconds * 1000000);
printf("Sleep %d seconds done!\n\n\n", sleep_seconds);
}
}
}
return NULL;
}
static void *smartrc_auto_socket_test(void *arg)
{
smartrc_thread_t *test_thread = &auto_test_threads[AUTO_SOCKET_TEST];
vcap_frame_desc_t vcap_frame;
cv_outputs_desc_t outputs_desc;
smartrc_init_params_t *init_params = &smartrc_params.init_params;

u8 canvas_id = init_params->cv_src[cur_stream_id];
u8 disable_frame_sync_map = smartrc_params.disable_frame_sync_for_cv_flags;
arg = arg; /* This is to fix compile warning */
rval = smartrc_client_init(&client_id);

printf("Smartrc client init failed.\n");

} else {
printf("client %d: dsp_pts %d, y_addr_offset 0x%x, uv_addr_offset 0x%x\n",
client_id, vcap_frame.dsp_pts, vcap_frame.y_addr_offset,
vcap_frame.uv_addr_offset);
}
memset(&outputs_desc, 0, sizeof(cv_outputs_desc_t));
outputs_desc.type = OUTPUT_BBOX;
outputs_desc.num = 1;
outputs_desc.apply_stream_map = 1U << cur_stream_id;
outputs_desc.disable_frame_sync = (disable_frame_sync_map & (1U << cur_stream_id)) ?
1 : 0;
outputs_desc.dsp_pts = outputs_desc.disable_frame_sync ? 0 : vcap_frame.dsp_pts;
outputs_desc.width = 120;
outputs_desc.height = 68;
outputs_desc.pitch = 120;
memset(&bbox, 0, sizeof(bbox));
for(i = 0; i < MAX_BBOX_NUM; i++) {
bbox[0].roi_idx = 0;
bbox[0].x_min = 0;
bbox[0].x_max = 0.5;
bbox[0].y_min = 0;
bbox[0].y_max = 0.5;
}
rval = smartrc_client_update_bbox(client_id, &outputs_desc, &bbox[0]);
if (rval == SMARTRC_CLIENT_EINVAL) {
printf("Update CV outputs failed, invalid content for client %d.\n", client_id);
} else if (rval == SMARTRC_CLIENT_ECOMM) {
printf("Update CV outputs failed, communication err for client %d.\n", client_id);
} else if (rval == SMARTRC_CLIENT_ESYS) {
printf("Update CV outputs failed, memory err for client %d.\n", client_id);
break;
}
clock_gettime(CLOCK_REALTIME, &current_time);
printf("Time (us): [%05ld]\n",
(current_time.tv_sec - start_time.tv_sec) * 1000000
+ (current_time.tv_nsec - start_time.tv_nsec) / 1000);
usleep(50000);
}
}
rval = smartrc_client_deinit(client_id);
if (rval == SMARTRC_CLIENT_EINVAL) {
printf("Client deinit failed, invalid client %d.\n", client_id);
}
printf("Exit auto socket test.\n");
return NULL;
}
static int smartrc_auto_run_test(smartrc_params_t *smartrc_param)
{
smartrc_thread_t *test_thread = &auto_test_threads[0];

switch(i) {

rval = pthread_create(&test_thread[i].tid, NULL,
smartrc_auto_socket_test, NULL);

break;

break;

smartrc_auto_config_stream, NULL);
if(rval < 0) {
printf("Create auto config stream process failed.\n");
rval = -1;

break;

printf("Unsupport test mode %d.\n", i);
rval = -1;
break;
}
}
if (rval < 0) {
break;
}
}
return rval;

{
