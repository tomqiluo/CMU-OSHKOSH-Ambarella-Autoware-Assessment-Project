static int lua_is_item_exist_str(lua_State* L,const char* item)
{
int is_exist=1;
lua_pushstring(L,item);
lua_gettable(L,-2);
if(lua_isnil(L,-1)){
is_exist=0;
}
lua_pop(L,1);
return is_exist;
}
static int lua_get_int_item_str(lua_State *L, const char *item, int* item_value)
{
int rtn_value=0;
if(lua_is_item_exist_str(L,item)){
lua_pushstring(L,item);
lua_gettable(L,-2);
*item_value=lua_tointeger(L,-1);
lua_pop(L,1);
}else{
rtn_value=-1;
}
return rtn_value;
}
static int lua_get_str_item_str(lua_State *L, const char *item, char *item_value){
int rtn_value=0;
const char* tmp_str=NULL;
if(lua_is_item_exist_str(L,item)){
lua_pushstring(L,item);
lua_gettable(L,-2);
tmp_str=lua_tostring(L,-1);
memcpy(item_value,tmp_str,strlen(tmp_str)+1);
lua_pop(L,1);
}
else{
rtn_value=-1;
}
return rtn_value;
}
static int product_get_chan_cfg(lua_State* L, product_info_t* lua_product_config, int chan_id)
{
int rtn=0;
int item_value;
char item[128];
// for ir virtual type
sprintf(item,"img_virtual_type");
if(!lua_get_int_item_str(L,item,&item_value)){
lua_product_config->product_cfg[chan_id].img_virtual_type=item_value;
}
return rtn;
}
static int parse_product_config(const char *lua_file, product_info_t* product_info,int chan_num)
{
int rtn=0;
char item_value_str[128];
int item_value;
char item[128];
int i;
lua_State* L=NULL;
if((lua_file==NULL) || (product_info==NULL)){
printf("some params is NULL\n");
return -1;
}
// init lua
L=luaL_newstate();
if(!L){
printf("fail to alloc lua state\n");
return -1;
}
if(luaL_loadfile(L,lua_file) || lua_pcall(L,0,0,0)){
printf("lua error msg is %s\n",lua_tostring(L,-1));
return -1;
}
// global information
lua_getglobal(L,"global");
if(lua_isnil(L,-1)){
printf("not exist global field in the config lua\n");
return -1;
}
sprintf(item,"version");
if(!lua_get_int_item_str(L,item,&item_value)){
product_info->version = item_value;
}else {
printf("not exist version field in the config lua\n");
return -1;
}
sprintf(item,"mod_time");
if(!lua_get_int_item_str(L,item,&item_value)){
product_info->mod_time = item_value;
}else {
printf("not exist mod_time field in the config lua\n");
return -1;
}
sprintf(item,"aaa_iq_config_lua_path");
if(!lua_get_str_item_str(L,item,item_value_str)){
memcpy( product_info->aaa_iq_config_lua_path,item_value_str,strlen(item_value_str)+1);
}else {
printf("not exist aaa_iq_config_lua_path field in the config lua\n");
return -1;
}
// parse parameters for channels
for(i=0;i<chan_num;i++){
sprintf(item_value_str,"chan_%d",i);
lua_getglobal(L,item_value_str);
if(lua_isnil(L,-1)){
product_info->product_cfg[i].img_virtual_type = 0;
continue;
}
if(product_get_chan_cfg(L,product_info,i)){
printf("get chan config error\n");
rtn=-1;
goto EXIT;
}
}
EXIT:
lua_close(L);
return rtn;
}
int get_product_lua_config(product_info_t* p_product_info,char *p_product_file,int chan_num)
{
int reval=0;
if (access(p_product_file, 0) < 0) {
printf("Error: Can't find product config lua file:%s\n",p_product_file);
reval =-1;
}else{
reval=parse_product_config(p_product_file,p_product_info,chan_num);
if(reval<0){
printf("parser_product_config failed\n");
}
}
return reval;
}
