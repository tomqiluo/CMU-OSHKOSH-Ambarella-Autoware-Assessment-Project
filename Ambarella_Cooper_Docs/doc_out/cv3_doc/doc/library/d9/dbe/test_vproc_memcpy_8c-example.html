#define FILENAME_LENGTH (256)
#define NO_ARG 0
#define HAS_ARG 1
#define ALIGN_PITCH(x) (((x) + (CAVALRY_PORT_PITCH_ALIGN) - 1) & ~((CAVALRY_PORT_PITCH_ALIGN) - 1))
struct cv_mem {
void *virt;
unsigned long phys;
unsigned long size;
};
static int fd_cavalry;
static int measure_time_flag = 0;
static int loop_forever_flag = 0;
static int enable_cache_flag = 0;
static long long int loop_cnt = 0;
static char bin_name[FILENAME_LENGTH] = "/usr/share/ambarella/vproc/vproc.bin";
static char input_name[FILENAME_LENGTH] = "src.bin";
static char output_name[FILENAME_LENGTH] = "dst.bin";
static struct option long_options[] = {
{"input", HAS_ARG, 0, 'i'},
{"output", HAS_ARG, 0, 'o'},
{"loop", HAS_ARG, 0, 'l'},
{"time", NO_ARG, 0, 't'},
{"cache", NO_ARG, 0, 'c'},
{0, 0, 0, 0},
};
static const char *short_options = "i:o:l:tc";
struct hint_s {
const char *arg;
const char *str;
};
static const struct hint_s hint[] = {
{"", "\t\tPath to input binary."},
{"", "\t\tPath to output binary."},
{"", "\t\tLoop cnt, -1 forever"},
{"", "\t\tMeasure run time."},
{"", "\t\tEnable cache"},
};
void usage(void)
{
unsigned long i;
printf("test_vproc_memcpy usage:\n");
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
}
int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index))
!= -1) {
switch (ch) {
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n",
optarg,
value,
FILENAME_LENGTH);
return -1;
}
snprintf(input_name, sizeof(input_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n",
optarg,
value,
FILENAME_LENGTH);
return -1;
}
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 'l':
loop_cnt = atoi(optarg);
if (loop_cnt < 0) {
loop_forever_flag = 1;
loop_cnt = 0;
} else if (loop_cnt > 0) {
loop_cnt--;
}
break;
case 't':
measure_time_flag = 1;
break;
case 'c':
enable_cache_flag = 1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
long int get_file_size(const char *file_name)
{
FILE *in = NULL;
long int filesize = 0;
do {
if ((in = fopen(file_name, "rb")) == NULL) {
perror(file_name);
break;
}
if (fseek(in, 0, SEEK_END) < 0) {
perror("fseek input file");
break;
}
filesize = ftell(in);
} while(0);
if (in) {
fclose(in);
in = NULL;
}
return filesize;
}
int load_file(const char *file_name, uint8_t *virt_addr, uint32_t size)
{
FILE *in = NULL;
uint32_t file_size = 0;
int rval = 0;
do {
if ((in = fopen(file_name, "rb")) == NULL) {
perror(file_name);
rval = -1;
break;
}
if (fseek(in, 0, SEEK_END) < 0) {
perror("fseek input file");
rval = -1;
break;
}
file_size = ftell(in);
if (file_size != size) {
printf("current file size [%u] doesn't match input file size [%u]\n",
size,
file_size);
rval = -1;
break;
}
if (fseek(in, 0, SEEK_SET) < 0) {
perror("fseek input file");
rval = -1;
break;
}
if (fread(virt_addr, sizeof(uint8_t), size, in) != size) {
perror("fread input file");
rval = -1;
break;
}
} while (0);
if (in) {
fclose(in);
in = NULL;
}
return rval;
}
int store_file(const char *file_name, uint8_t *virt_addr, uint32_t size)
{
FILE *out = NULL;
int rval = 0;
do {
if ((out = fopen(file_name, "w+b")) == NULL) {
perror(file_name);
rval = -1;
break;
}
if (fwrite(virt_addr, sizeof(uint8_t), size, out) != size) {
perror(file_name);
rval = -1;
break;
}
} while (0);
if (out) {
fclose(out);
out = NULL;
}
return rval;
}
static void sigstop()
{
loop_forever_flag = 0;
}
int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
struct cv_mem lib_mem;
struct cv_mem src;
struct cv_mem dst;
long file_size;
uint32_t size = 0;
int rval = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return -1;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open(CAVALRY_DEV_NODE, O_RDWR, 0)) < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
file_size = get_file_size(input_name);
if (file_size == -1) {
printf("Can not find file: %s\n", input_name);
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description,
ver.major,
ver.minor,
ver.patch,
ver.mod_time);
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&src, 0, sizeof(src));
memset(&dst, 0, sizeof(dst));
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc(&lib_mem.size, &lib_mem.phys, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
vproc_load(fd_cavalry, lib_mem.virt, lib_mem.phys, size);
src.size = file_size;
dst.size = file_size;
if (cavalry_mem_alloc(&src.size, &src.phys, &src.virt, enable_cache_flag) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
if (cavalry_mem_alloc(&dst.size, &dst.phys, &dst.virt, enable_cache_flag) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
// Must clean cache after load data to memory if cache enabled, otherwise VP may
// access the wrong data
if (load_file(input_name, (uint8_t *)src.virt, file_size)) {
printf("Failed to load file, exit!\n");
rval = -1;
break;
}
// ===================
// ARM <--> cache <--> | physical memory |
// VP_MEMCPY <-------> | physical memory |
// ===================
// clean cache to make sure the file load to physical memory, not cached in ARM
// cache memory
if (enable_cache_flag) {
if (cavalry_mem_sync_cache(src.size, src.phys, 1, 0) < 0) {
printf("cavalry_mem_sync_cache err\n");
rval = -1;
break;
}
}
if (loop_forever_flag) {
printf("Start memcpy:\n");
} else {
printf("Start memcpy for %lld loops:\n", loop_cnt + 1);
}
do {
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
vproc_memcpy(dst.phys, src.phys, file_size);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("memcpy takes %ld us\n",
(tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
} while (loop_forever_flag == 1 ? loop_forever_flag : loop_cnt--);
// Invalid cache to make sure the cache can access the new data
if (enable_cache_flag) {
if (cavalry_mem_sync_cache(dst.size, dst.phys, 0, 1) < 0) {
printf("cavalry_mem_sync_cache err\n");
rval = -1;
break;
}
}
// Must invalid cache before ARM access the physical memory if cache enabled,
// because the data may changed by VP in memory, cached data out of date
if (store_file(output_name, (uint8_t *)dst.virt, file_size) < 0) {
printf("Failed to store file, exit!\n");
rval = -1;
break;
}
} while (0);
printf("Stop memcpy\n");
if (src.virt && src.size) {
if (cavalry_mem_free(src.size, src.phys, src.virt) < 0) {
printf("cavalry_mem_free src err\n");
rval = -1;
}
src.virt = NULL;
src.phys = 0;
src.size = 0;
}
if (dst.virt && dst.size) {
if (cavalry_mem_free(dst.size, dst.phys, dst.virt) < 0) {
printf("cavalry_mem_free dst err\n");
rval = -1;
}
dst.virt = NULL;
dst.phys = 0;
dst.size = 0;
}
vproc_exit();
if (lib_mem.virt != NULL) {
if (cavalry_mem_free(lib_mem.size, lib_mem.phys, lib_mem.virt) < 0) {
printf("cavalry_mem_free err\n");
rval = -1;
}
}
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
