int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, sign, datasize, qval, ebits, depth, height, width;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(bin_name, sizeof(bin_name), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(data_name, sizeof(data_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 's':
get_vect_resolution(optarg, &width, &height, &depth);
ic = depth;
ih = height;
iw = width;
break;
case 'p':
ip = atoi(optarg);
if (CHECK_PITCH_ALIGN(ip)) {
printf("Pitch [%d] is not %u-byte aligned.\n",
ip, CAVALRY_PORT_PITCH_ALIGN);
return -1;
}
break;
case 'P':
op = atoi(optarg);
if (CHECK_PITCH_ALIGN(op)) {
printf("Pitch [%d] is not %u-byte aligned.\n",
op, CAVALRY_PORT_PITCH_ALIGN);
return -1;
}
break;
case 'I':
get_vect_dataformat(optarg, &sign, &datasize, &qval, &ebits);
input_df.sign = sign;
input_df.datasize = datasize;
input_df.exp_offset = qval;
input_df.exp_bits = ebits;
break;
case 'O':
get_vect_dataformat(optarg, &sign, &datasize, &qval, &ebits);
output_df.sign = sign;
output_df.datasize = datasize;
output_df.exp_offset = qval;
output_df.exp_bits = ebits;
break;
case 'r':
priority = atoi(optarg);
if (priority < CAVALRY_PRIORITY_LOWEST
|| priority > CAVALRY_PRIORITY_HIGHEST) {
printf("Priority [%d] is not in range [%d, %d].\n",
priority,
CAVALRY_PRIORITY_LOWEST,
CAVALRY_PRIORITY_HIGHEST);
return -1;
}
break;
case 't':
measure_time_flag = 1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int load_vect_bin(char *file, void *dst_addr,
vect_desc_mfd_t *vect_desc, uint32_t *size)
{
FILE *fp = NULL;
int rval = 0;
uint32_t i, data_sz;
uint32_t ch, ht, wd, pch, llen;
unsigned long file_length;
do {
if ((fp = fopen(file, "rb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fseek(fp, 0, SEEK_END) < 0) {
perror("SEEK_END");
rval = -1;
break;
}
file_length = ftell(fp);
if (fseek(fp, 0, SEEK_SET) < 0) {
perror("SEEK_SET");
rval = -1;
break;
}
ch = vect_desc->shape.d;
ht = vect_desc->shape.h;
wd = vect_desc->shape.w;
pch = vect_desc->pitch;
data_sz = (1<<vect_desc->data_format.datasize);
if (vect_desc->color_space >= CS_YUV_FIRST) {
switch(vect_desc->color_space) {
case CS_NV12:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
default:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
}
} else {
*size = ch * ht * pch;
llen = ch * ht;
}
if (file_length == *size){
if (fread(dst_addr, 1, file_length, fp) != file_length) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
} else {
for (i = 0; i < llen; i++) {
if (fread(dst_addr + i * pch,
data_sz, wd, fp) != wd) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
memset(dst_addr + (i * pch + wd * data_sz),
0, pch - wd * data_sz);
}
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}

int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
struct cv_mem lib_mem;
struct cv_mem in_mem;
struct cv_mem out_mem;
vect_desc_mfd_t in, ot;
uint32_t size = 0;
int rval = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
memset(&input_df, 0, sizeof(input_df));
memset(&output_df, 0, sizeof(output_df));
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&in_mem, 0, sizeof(in_mem));
memset(&out_mem, 0, sizeof(out_mem));
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open(CAVALRY_DEV_NODE, O_RDWR, 0)) < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
memset(&in, 0, sizeof(in));
memset(&ot, 0, sizeof(ot));
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc_mfd(lib_mem.size, &lib_mem.mfd, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
vproc_load_mfd(fd_cavalry, lib_mem.virt, lib_mem.mfd, size);
if (!ip) {
ip = ALIGN_PITCH(iw * (1<<input_df.datasize));
}
if (ip < iw * (1<<input_df.datasize)) {
printf("pitch is smaller than width!\n");
rval = -1;
break;
}
if (!op) {
op = ALIGN_PITCH(iw * (1<<output_df.datasize));
}
if (op < iw * (1<<output_df.datasize)) {
printf("pitch is smaller than width!\n");
rval = -1;
break;
}
in.shape.d = ic;
in.shape.h = ih;
in.shape.w = iw;
in.pitch = ip;
in.data_format.sign = input_df.sign;
in.data_format.datasize = input_df.datasize;
in.data_format.exp_offset = input_df.exp_offset;
in.data_format.exp_bits = input_df.exp_bits;
in_mem.size = in.shape.d * in.shape.h * in.pitch;
if (cavalry_mem_alloc_mfd(in_mem.size, &in_mem.mfd, &in_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
in.data_addr_fd = in_mem.mfd;
in.data_addr_offset = 0;
if (load_vect_bin(data_name, in_mem.virt, &in, &size) < 0) {
printf("load data input fails!\n");
rval = -1;
break;
}
ot.shape.d = ic;
ot.shape.h = ih;
ot.shape.w = iw;
ot.pitch = op;
ot.data_format.sign = output_df.sign;
ot.data_format.datasize = output_df.datasize;
ot.data_format.exp_offset = output_df.exp_offset;
ot.data_format.exp_bits = output_df.exp_bits;
out_mem.size = ot.shape.d * ot.shape.h * ot.pitch;
if (cavalry_mem_alloc_mfd(out_mem.size, &out_mem.mfd, &out_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(out_mem.virt, 0, out_mem.size);
ot.data_addr_fd = out_mem.mfd;
ot.data_addr_offset = 0;
set_vproc_priority(vproc_get_current_tid(), priority);
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
rval = vproc_dtcvt_mfd(&in, &ot);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("dtcvt takes %ld us\n", (tv2.tv_sec - tv1.tv_sec) * 1000000 +
(tv2.tv_usec - tv1.tv_usec));
}
if (rval < 0) {
printf("dtcvt failure!\n");
break;
}
if (output_flag) {
if (store_bin(output_name, out_mem.size, out_mem.virt) < 0) {
printf("store output fails!\n");
rval = -1;
break;
}
}
} while (0);
vproc_exit();
if (out_mem.virt != NULL) {
if (cavalry_mem_free_mfd(out_mem.size , out_mem.mfd, out_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (in_mem.virt != NULL) {
if (cavalry_mem_free_mfd(in_mem.size , in_mem.mfd, in_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (lib_mem.virt != NULL) {
if (cavalry_mem_free_mfd(lib_mem.size , lib_mem.mfd, lib_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
