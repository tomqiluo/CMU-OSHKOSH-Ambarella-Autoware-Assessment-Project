void usage(void)
{
unsigned long i;
printf("test_vproc_harris_live usage:\n");
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
}
int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index))
!= -1) {
switch (ch) {
case 't':
param_harris_threshold = atof(optarg);
if (param_harris_threshold <= 0) {
printf("Invalid harris_threshold %f.\n", param_harris_threshold);
printf("allowed harris_threshold higher than zero.\n");
return -1;
}
break;
case 'n':
param_max_harris_num = atoi(optarg);
if (param_max_harris_num > MAX_MAX_HARRIS_NUM) {
printf("maximal detected harris points is %d.\n", MAX_MAX_HARRIS_NUM);
return -1;
}
break;
case 'w':
nms_win_size= atoi(optarg);
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
#define DRAW_SQUARE_HALF_SIZE 6
#define DRAW_SQUARE_THICKNESS 3
static int extract_harris_desc_with_draw(uint8_t* harris_bitmap,
int32_t im_h, int32_t im_w, int32_t max_harris_det_num)
{
static int frame_cnt = 0;
char draw_string[128];
int Harris_cnt = 0;
int32_t i, j, n;
uint8_t *check_row_ptr;
uint8_t check_bits_mask = 1;
int check_step_bytes = sizeof(uint8_t);
int check_step_bits = check_step_bytes * 8;
int32_t bitmap_pitch = ALIGN_PITCH((ALIGN_8_BITS(im_w) / 8));
const ea_osd_area_buffer_t *ol_info = ea_osd_area_current_buffer(display, 0, NULL);
ea_osd_refresh_stream_params_t stream_param;
stream_param.dsp_pts = 0;
uint8_t* ol_buffer = ea_osd_area_current_buffer(display, 0, NULL)->buf;
memset(ol_buffer,0,ol_info->height * ol_info->pitch);
Mat mframe_live = Mat(ol_info->height,ol_info->width, CV_8UC1, ol_buffer, ol_info->pitch);
float fb_rate_x = (float)ol_info->width / im_w;
float fb_rate_y = (float)ol_info->height / im_h;
for (i = 0; i < im_h; i++) {
check_row_ptr = (uint8_t*)(harris_bitmap + i * bitmap_pitch);
for (j = 0; j < bitmap_pitch; j += check_step_bytes){
if ((check_row_ptr[j]) != 0){
for (n = 0; n < check_step_bits; n++){
if (((check_row_ptr[j] >> n) & check_bits_mask)
&& ((j * 8 + n) < im_w)){
int c_x=j * 8 + n;
int c_y =i;
rectangle(mframe_live,
Rect(fb_rate_x * c_x - DRAW_SQUARE_HALF_SIZE,
fb_rate_y * c_y - DRAW_SQUARE_HALF_SIZE,
DRAW_SQUARE_HALF_SIZE * 2 + 1,
DRAW_SQUARE_HALF_SIZE * 2 + 1), Scalar(254,254,254), 2);
Harris_cnt++;
if (Harris_cnt >= max_harris_det_num) {
return Harris_cnt;
}
break;
}
}
}
}
}
snprintf(draw_string, sizeof(draw_string), "%d", frame_cnt++);
putText(mframe_live, draw_string, Point(50, 30), FONT_HERSHEY_PLAIN , 2, Scalar(254,254,254), 2, 8, false);
if (ea_osd_refresh(display, &stream_param) < 0) {
exit(-1);
}
return Harris_cnt;
}

static int create_vect(vect_desc_t* vect,cv_mem_t* vm,
int h, int w, int d, int data_size, int with_cache)
{
vect->shape.p = 1;
vect->shape.d = d;
vect->shape.h = h;
vect->shape.w = w;
vect->pitch = ALIGN_PITCH(w * data_size);
vect->color_space = CS_VECT;
vm->size = vect->pitch * vect->shape.h * vect->shape.d;
if (cavalry_mem_alloc(&vm->size, &vm->phys, &vm->virt, with_cache) < 0) {
perror("cavalry_mem_alloc err\n");
return -1;
}
vect->data_addr = vm->phys;
return vm->size;
}
static int create_bitvect(vect_desc_t* vect,cv_mem_t* vm,
int h, int w, int d, int with_cache)
{
vect->shape.d = d;
vect->shape.h = h;
vect->shape.w = w;
vect->pitch = ALIGN_PITCH((ALIGN_8_BITS(w) / 8));
vect->color_space = CS_VECT;
vm->size = vect->pitch * vect->shape.h * vect->shape.d;
if (cavalry_mem_alloc(&vm->size, &vm->phys, &vm->virt, with_cache) < 0) {
perror("cavalry_mem_alloc err\n");
return -1;
}
vect->data_addr = vm->phys;
memset(vm->virt,0,vm->size);
return vm->size;
}
static int free_vmem(cv_mem_t* vm)
{
if (vm->virt != NULL) {
if (cavalry_mem_free(vm->size , vm->phys, vm->virt) < 0) {
printf("cavalry_mem_alloc err\n");
return -1;
}
}
return 0;
}
static int cv_env_init(int inbuf_id, int outstream_id)
{
int rval = 0;
int env_ret = -1;
int features = 0;
do {
features = EA_ENV_ENABLE_IAV
| EA_ENV_ENABLE_VPROC
| EA_ENV_ENABLE_OSD_STREAM;
env_ret = ea_env_open(features);
if (env_ret < 0) {
printf("ea_env_open failed !\n");
rval = -1;
break;
}
img_resource = ea_img_resource_new(EA_CANVAS, (void *)(unsigned long)inbuf_id);
RVAL_ASSERT(img_resource != NULL);
display = ea_osd_new(EA_OSD_STREAM, outstream_id, NULL);
RVAL_ASSERT(display != NULL);
} while(0);
if (rval < 0) {
if (display) {
ea_osd_free(display);
display = NULL;
}
if (img_resource) {
ea_img_resource_free(img_resource);
img_resource = NULL;
}
if (env_ret >= 0) {
ea_env_close();
}
}
return rval;
}
static void cv_env_deinit()
{
if (display) {
ea_osd_free(display);
display = NULL;
}
if (img_resource) {
ea_img_resource_free(img_resource);
img_resource = NULL;
}
ea_env_close();
}
int main(int argc, char **argv)
{
struct timeval tv1, tv2;
ea_img_resource_data_t yuv_info;
ea_osd_color_rgba_t colors_table[256];
int rval = -1;
vect_desc_t image_idsp_in_vect;
cv_mem_t harris_out_detbit_mem;
vect_desc_t harris_out_detbit_vect;
cv_mem_t maxresp_mem;
vect_desc_t maxresp_vect;
int harris_det_num;
uint16_t *resp_ptr;
float peak_Harris_resp;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if (cv_env_init(param_buf_id, output_stream_id) < 0) {
printf("cv_env_init() failed\n");
return -1;
} else {
for (int i = 0; i < 256; i++) {
colors_table[i].r = i;
colors_table[i].g = i;
colors_table[i].b = i;
colors_table[i].alpha = 255;
}
colors_table[0].alpha = 0;
if (ea_osd_area_set_color_table_rgba(display, 0, colors_table, 256) < 0) {
printf("set overlay color table error!\n");
return -1;
}
}
do {
harris_cfg_t harris_param;
harris_param.nms_size = nms_win_size;
harris_param.threshold = param_harris_threshold;
RVAL_OK(ea_img_resource_hold_data(img_resource, &yuv_info));
ea_tensor_vect_desc(yuv_info.tensor_group[0], &image_idsp_in_vect);
ea_img_resource_drop_data(img_resource, &yuv_info);
if ( !((image_idsp_in_vect.shape.h == 1080) &&
(image_idsp_in_vect.shape.w == 1920) &&
(image_idsp_in_vect.pitch == 1920)) &&
!((image_idsp_in_vect.shape.h == 720) &&
(image_idsp_in_vect.shape.w == 1280) &&
(image_idsp_in_vect.pitch == 1280))){
printf("the IDSP input size must be 1920px1080 or 1280px720\n");
break;
}
create_bitvect(&harris_out_detbit_vect, &harris_out_detbit_mem, image_idsp_in_vect.shape.h,
image_idsp_in_vect.shape.w, 1, 1);
create_vect(&maxresp_vect, &maxresp_mem, 1, 1, 1, sizeof(uint16_t), 0);
while (live_running_flag){
RVAL_OK(ea_img_resource_hold_data(img_resource, &yuv_info));
ea_tensor_vect_desc(yuv_info.tensor_group[0], &image_idsp_in_vect);
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
if (vproc_harris(&image_idsp_in_vect, &harris_param,
&harris_out_detbit_vect, &maxresp_vect) < 0) {
break;
}
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("vproc_harris() takes %ld us per loop\n",
(tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
cavalry_mem_sync_cache(harris_out_detbit_mem.size, harris_out_detbit_mem.phys, 0, 1);
resp_ptr = (uint16_t *)(maxresp_mem.virt);
peak_Harris_resp = (float)(*resp_ptr) / HARRIS_RESPONSE_Q_SCALE;
harris_det_num = extract_harris_desc_with_draw(
(uint8_t*)harris_out_detbit_mem.virt, image_idsp_in_vect.shape.h,
image_idsp_in_vect.shape.w, param_max_harris_num);
printf("vproc_harris() det %d points, peak value = %f\n", harris_det_num, peak_Harris_resp);
ea_img_resource_drop_data(img_resource, &yuv_info);
}
} while (0);
free_vmem(&harris_out_detbit_mem);
free_vmem(&maxresp_mem);
cv_env_deinit();
return 0;
}
