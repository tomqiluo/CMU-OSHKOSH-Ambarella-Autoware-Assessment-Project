int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, width, height, depth;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(bin_name, sizeof(bin_name), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
input_flag = 1;
snprintf(input_name, sizeof(input_name), "%s", optarg);
break;
case 's':
get_vect_resolution(optarg, &width, &height, &depth);
ic = depth;
ih = height;
iw = width;
break;
case 'p':
ip = atoi(optarg);
if (CHECK_PITCH_ALIGN(ip)) {
printf("Pitch [%d] is not %u-byte aligned.\n",
ip, CAVALRY_PORT_PITCH_ALIGN);
return -1;
}
break;
case 'S':
get_vect_resolution(optarg, &width, &height, &depth);
oc = depth;
oh = height;
ow = width;
break;
case 'P':
op = atoi(optarg);
if (CHECK_PITCH_ALIGN(op)) {
printf("Pitch [%d] is not %u-byte aligned.\n",
op, CAVALRY_PORT_PITCH_ALIGN);
return -1;
}
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 'c':
cs = atoi(optarg);
if (cs < CS_VECT || cs >= CS_YUV_LAST
|| (cs >= CS_RGB_LAST && cs < CS_YUV_FIRST)) {
printf("Color space [%d] is not supported.\n", cs);
return -1;
}
break;
case 'f':
value = strtoul(optarg, NULL, 0);
if (value & (~0x0f)) {
printf("Invalid rotate-flip bitmap value: 0x%x, [0x0 ~ 0x0f]\n", value);
return -1;
}
rotate_flip_bitmap = value;
break;
case 't':
measure_time_flag = 1;
break;
case 'v':
verbose_level = atoi(optarg);
if (verbose_level < VPROC_LOG_ERR || verbose_level > VPROC_LOG_VERBOSE) {
printf("log level [%d] is not supported.\n", verbose_level);
return -1;
}
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int load_vect_bin(char *file, void *dst_addr,
vect_desc_t *vect_desc, uint32_t *size)
{
FILE *fp = NULL;
int rval = 0;
uint32_t i, data_sz;
uint32_t ch, ht, wd, pch, llen, lwd;
unsigned long file_length;
do {
if ((fp = fopen(file, "rb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fseek(fp, 0, SEEK_END) < 0) {
perror("SEEK_END");
rval = -1;
break;
}
file_length = ftell(fp);
if (fseek(fp, 0, SEEK_SET) < 0) {
perror("SEEK_SET");
rval = -1;
break;
}
ch = vect_desc->shape.d;
ht = vect_desc->shape.h;
wd = vect_desc->shape.w;
pch = vect_desc->pitch;
data_sz = (1<<vect_desc->data_format.datasize);
if (vect_desc->color_space >= CS_YUV_FIRST) {
switch(vect_desc->color_space) {
case CS_NV12:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
lwd = wd;
break;
case CS_Y:
*size = ht * pch;
llen = ht;
lwd = wd;
break;
case CS_ITL:
*size = ht * pch;
llen = ht;
lwd = 2 * wd;
break;
default:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
lwd = wd;
break;
}
} else {
*size = ch * ht * pch;
llen = ch * ht;
lwd = wd;
}
if (file_length == *size){
if (fread(dst_addr, 1, file_length, fp) != file_length) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
} else {
for (i = 0; i < llen; i++) {
if (fread(dst_addr + i * pch,
data_sz, lwd, fp) != lwd) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
memset(dst_addr + (i * pch + lwd * data_sz),
0, pch - lwd * data_sz);
}
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}

int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
struct cv_mem lib_mem;
struct cv_mem in_mem;
struct cv_mem out_mem;
vect_desc_t it, ot;
uint32_t size = 0;
int rval = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open(CAVALRY_DEV_NODE, O_RDWR, 0)) < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
if (!ip) {
if (cs == CS_ITL) {
ip = ALIGN_PITCH(2 * iw);
} else {
ip = ALIGN_PITCH(iw);
}
}
if (!op) {
if (cs == CS_ITL) {
op = ALIGN_PITCH(2 * ow);
} else {
op = ALIGN_PITCH(ow);
}
}
memset(&it, 0, sizeof(it));
memset(&ot, 0, sizeof(ot));
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&in_mem, 0, sizeof(in_mem));
memset(&out_mem, 0, sizeof(out_mem));
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc(&lib_mem.size, &lib_mem.phys, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
vproc_load(fd_cavalry, lib_mem.virt, lib_mem.phys, size);
vproc_set_log_level(verbose_level);
if ((cs == CS_NV12) && ((iw & 0x1) || (ih & 0x1))) {
printf("[Warning]: input NV12 image [width, height] should be even, "
"uv size round up to [%d, %d]\n", (iw + 1)/2, (ih + 1)/2);
}
if ((cs == CS_NV12) && ((ow & 0x1) || (oh & 0x1))) {
printf("[Warning]: output NV12 image [width, height] should be even, "
"uv size round up to [%d, %d]\n", (iw + 1)/2, (ih + 1)/2);
}
it.shape.w = iw;
it.shape.h = ih;
it.shape.d = ic;
it.pitch = ip;
it.color_space = cs;
if (input_flag) {
if (it.color_space == CS_NV12) {
in_mem.size = it.pitch * it.shape.h;
in_mem.size += it.pitch * ((it.shape.h + 1 ) >> 1);
} else if (it.color_space == CS_ITL) {
in_mem.size = it.shape.h * it.pitch;
} else {
in_mem.size = it.shape.d * it.shape.h * it.pitch;
}
if (cavalry_mem_alloc(&in_mem.size, &in_mem.phys, &in_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
it.data_addr = in_mem.phys;
if (load_vect_bin(input_name, in_mem.virt, &it, &size) < 0) {
printf("load input fails!\n");
rval = -1;
break;
}
}
ot.shape.w = ow;
ot.shape.h = oh;
ot.shape.d = oc;
ot.pitch = op;
ot.color_space = cs;
printf("src image size: %dx%d\n", it.shape.w, it.shape.h);
printf("dst image size: %dx%d\n", ot.shape.w, ot.shape.h);
if (ot.color_space == CS_NV12) {
out_mem.size = ot.shape.h * ot.pitch;
out_mem.size += ot.pitch * ((ot.shape.h + 1) >> 1);
} else if (it.color_space == CS_ITL) {
out_mem.size = ot.shape.h * ot.pitch;
} else {
out_mem.size = ot.shape.d * ot.shape.h * ot.pitch;
}
if (cavalry_mem_alloc(&out_mem.size, &out_mem.phys, &out_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(out_mem.virt, 0, out_mem.size);
ot.data_addr = out_mem.phys;
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
rval = vproc_rotate(&it, &ot, rotate_flip_bitmap);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("rotate takes %ld us\n", (tv2.tv_sec - tv1.tv_sec) * 1000000 +
(tv2.tv_usec - tv1.tv_usec));
}
if (rval < 0) {
printf("rotate failure!\n");
break;
}
if (output_flag) {
if (store_bin(output_name, out_mem.size, out_mem.virt) < 0) {
printf("store output fails!\n");
rval = -1;
break;
}
}
} while (0);
vproc_exit();
if (out_mem.virt != NULL) {
if (cavalry_mem_free(out_mem.size , out_mem.phys, out_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (in_mem.virt != NULL) {
if (cavalry_mem_free(in_mem.size , in_mem.phys, in_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (lib_mem.virt != NULL) {
if (cavalry_mem_free(lib_mem.size , lib_mem.phys, lib_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
