static int convert_df_cv(const char* s_df, char* cv_df, size_t size)
{
int i = 0;
int num = sizeof(s_df_cv_pair) / sizeof(s_df_cv_pair[0]);
if (!s_df || !cv_df) {
printf("s_df or cv_df is null\n");
return -1;
}
for (i = 0; i < num; ++i) {
if(strcmp(s_df, s_df_cv_pair[i].key) == 0){
memset(cv_df, 0, size);
snprintf(cv_df, size, "%s", s_df_cv_pair[i].val);
return 0;
}
}
printf(" unsupported data format %s \n", s_df);
return -1;
}
static int get_cv_df(const char* s_data, data_format_t *cvdf)
{
char c_data[64];
int ret = convert_df_cv(s_data, c_data, sizeof(c_data));
if (ret < 0){
return -1;
}
string ss_data=c_data;
char *data_i;
data_i=strtok((char*)ss_data.c_str(),",");
cvdf->sign = atoi(data_i);
data_i=strtok(NULL,",");
cvdf->datasize = atoi(data_i);
data_i=strtok(NULL,",");
cvdf->exp_offset = atoi(data_i);
data_i=strtok(NULL,",");
cvdf->exp_bits = atoi(data_i);
return 0;
}
int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, channel, height, width;
memset(&left_in_vect, 0, sizeof(left_in_vect));
memset(&left_in_mem, 0, sizeof(left_in_mem));
memset(&right_in_vect, 0, sizeof(right_in_vect));
memset(&right_in_mem, 0, sizeof(right_in_mem));
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index))
!= -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(vproc_bin_path, sizeof(vproc_bin_path), "%s", optarg);
break;
case 'l':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(left_bin_input_path, sizeof(left_bin_input_path), "%s", optarg);
break;
case 'r':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(right_bin_input_path, sizeof(right_bin_input_path), "%s", optarg);
break;
case 'L':
get_cv_df(optarg, &left_in_vect.data_format);
break;
case 'R':
get_cv_df(optarg, &right_in_vect.data_format);
break;
case 's':
get_vect_resolution(optarg, &channel, &width, &height);
ic = channel;
iw = width;
ih = height;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}

static int create_vect_mfd(vect_desc_mfd_t* vect,cv_mem_mfd_t* vm,
int h, int w, int d, int data_size, int with_cache)
{
vect->shape.p = 1;
vect->shape.d = d;
vect->shape.h = h;
vect->shape.w = w;
vect->pitch = ALIGN_PITCH(w * data_size);
vect->color_space = CS_VECT;
vm->size = vect->pitch * vect->shape.h * vect->shape.d;
if (cavalry_mem_alloc_mfd(vm->size, &vm->mfd, &vm->virt, with_cache) < 0) {
perror("cavalry_mem_alloc_mfd err\n");
return -1;
}
vect->data_addr_fd = vm->mfd;
vect->data_addr_offset = 0;
return vm->size;
}
static int free_vmem_mfd(cv_mem_mfd_t* vm)
{
if (vm->virt != NULL) {
if (cavalry_mem_free_mfd(vm->size , vm->mfd, vm->virt) < 0) {
printf("cavalry_mem_free_mfd err\n");
return -1;
}
}
return 0;
}
static int load_bin_file(const char *file_name, uint8_t *virt_addr, uint32_t size)
{
FILE *in = NULL;
uint32_t file_size = 0;
int rval = 0;
do {
if ((in = fopen(file_name, "rb")) == NULL) {
perror(file_name);
rval = -1;
break;
}
if (fseek(in, 0, SEEK_END) < 0) {
perror("fseek input file");
rval = -1;
break;
}
file_size = ftell(in);
if (fseek(in, 0, SEEK_SET) < 0) {
perror("fseek input file");
rval = -1;
break;
}
if (file_size >= size) {
if (fread(virt_addr, 1, size, in) != size) {
perror("fread input file");
rval = -1;
break;
}
}
} while (0);
if (in) {
fclose(in);
in = NULL;
}
return rval;
}
int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
cv_mem_mfd_t lib_mem;
uint32_t size = 0;
int rval = -1;
float cdist_val;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open("/dev/cavalry", O_RDWR, 0)) < 0) {
perror("/dev/cavalry");
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
memset(&lib_mem, 0, sizeof(lib_mem));
if (vproc_init(vproc_bin_path, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc_mfd(lib_mem.size, &lib_mem.mfd, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc_mfd err\n");
rval = -1;
break;
}
vproc_load_mfd(fd_cavalry, (uint8_t*)lib_mem.virt, lib_mem.mfd, size);
if (vproc_cdist_init() < 0){
rval = -1;
break;
}
create_vect_mfd(&left_in_vect, &left_in_mem, ih, iw, ic, int(pow(2, left_in_vect.data_format.datasize)), 1);
create_vect_mfd(&right_in_vect, &right_in_mem, ih, iw, ic, int(pow(2, right_in_vect.data_format.datasize)), 1);
if (load_bin_file(left_bin_input_path, (uint8_t*)left_in_mem.virt , left_in_mem.size) < 0){
rval = -1;
break;
}
if (load_bin_file(right_bin_input_path, (uint8_t*)right_in_mem.virt , right_in_mem.size) < 0){
rval = -1;
break;
}
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
cavalry_mem_sync_cache_mfd(left_in_mem.size, 0, left_in_mem.mfd, 1, 0);
cavalry_mem_sync_cache_mfd(right_in_mem.size, 0, right_in_mem.mfd, 1, 0);
if (vproc_cdist_mfd(&left_in_vect, &right_in_vect, &cdist_val) < 0) {
printf("Fail at vproc_cdist_mfd()\n");
break;
}
printf("the cosine similarity value is %f\n",cdist_val);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("Whole vproc_cdist_mfd() takes %ld us\n",
(tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
} while (0);
vproc_exit();
vproc_cdist_release();
free_vmem_mfd(&left_in_mem);
free_vmem_mfd(&right_in_mem);
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
