int cv_env_init(test_eazyai_params_t *params)
{
int rval = EA_SUCCESS;
int features;
ea_env_arch_t arch;
const char *chip_name = NULL;
do {
features = EA_ENV_ENABLE_CAVALRY
| EA_ENV_ENABLE_VPROC
| EA_ENV_ENABLE_NNCTRL;
if (params->mode == RUN_MODE_LIVE) {
features |= EA_ENV_ENABLE_IAV;
if (params->out_type_map & NN_OUT_DST_VOUT) {
features |= EA_ENV_ENABLE_OSD_VOUT;
} else {
features |= EA_ENV_ENABLE_OSD_STREAM;
}
}
RVAL_OK(ea_env_open(features));
arch = ea_env_arch();
if (arch == EA_ENV_ARCH_INVALID) {
printf("chip is not supported, please ask Ambarella support for detail.\n");
print_supported_chips();
rval = EA_FAIL;
break;
}
chip_name = get_chip_name(arch);
RVAL_ASSERT(chip_name);
EA_LOG_NOTICE("current chip: %s\n", chip_name);
} while (0);
return rval;
}
void cv_env_deinit()
{
ea_env_close();
}
int in_file_init(test_eazyai_ctx_t *ctx, test_eazyai_params_t *params, nn_in_file_params_t *in_params)
{
int rval = EA_SUCCESS;
char str[MAX_STR_LEN + 1];
char *ptr = NULL;
char *p_str = NULL;
int i, len;
struct stat s_buf;
do {
memset(in_params, 0, sizeof(*in_params));
for (i = 0; i < params->multi_in_num; i++) {
p_str = &(params->multi_in_params[i][0]);
while ((ptr = strchr(p_str, '|')) != NULL) {
memset(str, 0, MAX_STR_LEN + 1);
len = ptr - p_str;
RVAL_ASSERT(len > 0);
strncpy(str, p_str, len);
RVAL_OK(parse_input_info(in_params->port_name[i], in_params->in_dir[i], str));
RVAL_OK(parse_file_type((int *)&(in_params->file_type[i]), str));
p_str = ptr + 1;
}
RVAL_ASSERT(strlen(p_str) > 0);
strncpy(str, p_str, MAX_STR_LEN);
RVAL_OK(parse_input_info(in_params->port_name[i], in_params->in_dir[i], str));
RVAL_OK(parse_file_type((int *)&(in_params->file_type[i]), str));
stat(in_params->in_dir[i], &s_buf);
if (S_ISDIR(s_buf.st_mode) != 1) {
ctx->in_single_raw_file = 1;
}
}
in_params->multi_in_num = params->multi_in_num;
ctx->in_ctx = nn_in_file_init(in_params);
RVAL_ASSERT(ctx->in_ctx != NULL);
} while (0);
return rval;
}

int preprocess_file_init(test_eazyai_ctx_t *ctx, test_eazyai_params_t *params,
nn_prep_params_t *preproc_params, nn_in_file_params_t *in_params)
{
int rval = EA_SUCCESS;
char str[MAX_STR_LEN + 1];
char *ptr = NULL;
char *p_str = NULL;
int i, len;
int *x, *y, *h, *w, *color;
do {
memset(preproc_params, 0, sizeof(*preproc_params));
memcpy(preproc_params->port_name, in_params->port_name, sizeof(preproc_params->port_name));
preproc_params->multi_in_num = params->multi_in_num;
preproc_params->log_level = params->log_level;
for (i = 0; i < params->multi_in_num; i++) {
preproc_params->device[i] = EA_CPU;
p_str = &(params->multi_in_params[i][0]);
while ((ptr = strchr(p_str, '|')) != NULL) {
memset(str, 0, MAX_STR_LEN + 1);
len = ptr - p_str;
RVAL_ASSERT(len > 0);
strncpy(str, p_str, len);
x = &(preproc_params->roi[i].x);
y = &(preproc_params->roi[i].y);
h = &(preproc_params->roi[i].h);
w = &(preproc_params->roi[i].w);
RVAL_OK(parse_roi_info(x, y, h, w, str));
color = (int *)&(preproc_params->out_color_format[i]);
RVAL_OK(parse_color_type(color, str));
RVAL_OK(parse_device_info((int *)&preproc_params->device[i], str));
p_str = ptr + 1;
}
RVAL_ASSERT(strlen(p_str) > 0);
strncpy(str, p_str, MAX_STR_LEN);
x = &(preproc_params->roi[i].x);
y = &(preproc_params->roi[i].y);
h = &(preproc_params->roi[i].h);
w = &(preproc_params->roi[i].w);
RVAL_OK(parse_roi_info(x, y, h, w, str));
color = (int *)&(preproc_params->out_color_format[i]);
RVAL_OK(parse_color_type(color, str));
RVAL_OK(parse_device_info((int *)&preproc_params->device[i], str));
if (preproc_params->device[i] == EA_CPU) {
EA_LOG_NOTICE("preprocessing of input %s use CPU.\n", preproc_params->port_name[i]);
} else {
EA_LOG_NOTICE("preprocessing of input %s use CVflow.\n", preproc_params->port_name[i]);
}
}
ctx->preproc_ctx = nn_prep_init(preproc_params);
RVAL_ASSERT(ctx->preproc_ctx != NULL);
} while (0);
return rval;
}

int postprocess_init(test_eazyai_ctx_t *ctx,
test_eazyai_params_t *params,
nn_postp_params_t *postproc_params,
nn_prep_params_t *preproc_params)
{
int rval = EA_SUCCESS;
int label_num = 0;
do {
memset(postproc_params, 0, sizeof(*postproc_params));
if (params->label_path) {
postproc_params->labels = (nn_label_t *)malloc(sizeof(nn_label_t) * MAX_LABEL_COUNT);
RVAL_ASSERT(postproc_params->labels != NULL);
memset(postproc_params->labels, 0, sizeof(nn_label_t) * MAX_LABEL_COUNT);
label_num = label_init(params->label_path, (char*)postproc_params->labels);
if (label_num > 0) {
postproc_params->class_num = label_num;
} else {
EA_LOG_ERROR("label number: %d <= 0 !\n", label_num);
rval = EA_FAIL;
break;
}
}
postproc_params->arm_nn_name = params->arm_nn_name;
postproc_params->log_level = params->log_level;
postproc_params->mode = params->mode;
if (postproc_params->arm_nn_name &&
strncmp(postproc_params->arm_nn_name, TO_FILE_POSTPROCESS_NAME, strlen(TO_FILE_POSTPROCESS_NAME)) == 0) {
if (strlen(params->output_dir) == 0) {
rval = -1;
EA_LOG_ERROR("The path of output folder for postprocess to_file isn't set.\n");
break;
}
}
postproc_params->output_dir = params->output_dir;
postproc_params->extra_input = params->extra_input;
if ((params->in_type_map & NN_IN_SRC_PYRAMID_BUF) == NN_IN_SRC_PYRAMID_BUF) {
postproc_params->enable_pyramid_flag = 1;
postproc_params->img_src_id = params->pyramid[1];
}
postproc_params->enable_fsync_flag = params->enable_fsync_flag;
postproc_params->input = nn_inference_input(&ctx->nn_inf, 0);
postproc_params->vp_output = ctx->vp_output;
postproc_params->f_result = ctx->f_result;
postproc_params->lua_file_path = params->lua_file_path;
postproc_params->roi = preproc_params->roi[0];
postproc_params->display = ctx->display;
ctx->postproc_ctx = nn_postp_init(postproc_params);
RVAL_ASSERT(ctx->postproc_ctx != NULL);
} while (0);
if (rval < 0) {
if (postproc_params->labels) {
free(postproc_params->labels);
postproc_params->labels = NULL;
}
}
return rval;
}

#define SLIDE_WINDOWS_LEN 100
static int run_dummy_loop(test_eazyai_ctx_t *ctx, test_eazyai_params_t *params)
{
int rval = EA_SUCCESS;
float fps;
int i;
ea_tensor_t *tmp = NULL;
double total_vp_time = 0;
double interval_vp_time = 0;
double vp_time = 0;
unsigned long loop_count = 0;
unsigned long interval_count = 0;
int sleep_time = 0;
int sleep_windows[SLIDE_WINDOWS_LEN];
int windows_len = 0;
int sleep_count = 0;
do {
RVAL_ASSERT(ctx != NULL);
for (i = 0; i < ctx->nn_inf.in_num; i++) {
tmp = nn_inference_input(&ctx->nn_inf, i);
RVAL_OK(dummy_input_generate_data(tmp));
}
do {
RVAL_OK(nn_inference_process(&ctx->nn_inf));
vp_time = (double)ea_net_params(ctx->nn_inf.net)->vp_time_us;
if (params->cavalry_load_percent != 1) {
if (windows_len == SLIDE_WINDOWS_LEN) {
sleep_time = sleep_time - sleep_windows[sleep_count] + vp_time;
sleep_windows[sleep_count] = vp_time;
} else {
sleep_time = sleep_time + vp_time;
sleep_windows[sleep_count] = vp_time;
windows_len++;
}
usleep((useconds_t)(sleep_time / windows_len * (1 - params->cavalry_load_percent) / params->cavalry_load_percent));
sleep_count++;
sleep_count = sleep_count >= SLIDE_WINDOWS_LEN ? 0 : sleep_count;
}
total_vp_time += vp_time / 1000;
interval_vp_time += vp_time / 1000;
loop_count++;
interval_count++;
if (interval_count == params->interval_loop_count) {
if (interval_count != params->total_loop_count) {
printf("Network forward interval average time %f us\n", (float)interval_vp_time / params->interval_loop_count * 1000);
}
fps = 1000.0 * params->interval_loop_count / interval_vp_time;
if (fps > 0) {
EA_LOG_NOTICE("FPS %.1f\n", fps);
}
interval_count = 0;
interval_vp_time = 0;
}
if (loop_count == params->total_loop_count) {
printf("Network forward average time %f us\n", (float)total_vp_time / params->total_loop_count * 1000);
total_vp_time = 0;
break;
}
} while (ctx->sig_flag == 0);
} while (0);
return rval;
}
static int run_file_loop(test_eazyai_ctx_t *ctx, test_eazyai_params_t *params)
{
int rval = EA_SUCCESS;
int i = 0;
ea_calc_fps_ctx_t calc_fps_ctx;
memset(&calc_fps_ctx, 0, sizeof(ea_calc_fps_ctx_t));
calc_fps_ctx.count_period = params->total_loop_count;
ea_img_resource_data_t img[NN_MAX_PORT_NUM];
int img_src_id = 0;
const char *name = NULL;
ea_tensor_t *bgr = NULL;
memset(img, 0, sizeof(img));
nn_prep_type_t prep_type;
nn_prep_params_t *prep_params = NULL;
int j;
uint32_t k;
nn_in_file_params_t *in_params;
ea_tensor_t **tmp = NULL;
ea_tensor_t *in_tensor = NULL;
unsigned long time_start = 0;
unsigned long time_end = 0;
double fp16tofp32_time = 0.0;
double total_fp16tofp32_time = 0.0;
double interval_fp16tofp32_time = 0.0;
int fp16tofp32_flag = 0;
uint32_t interval_count = 0;
double vp_time = 0.0;
double total_vp_time = 0.0;
double interval_vp_time = 0.0;
ea_display_obj_params_t *display_obj_params = NULL;
double post_time = 0.0;
double interval_post_time = 0.0;
double total_loop_time = 0.0;
double interval_loop_time = 0.0;
unsigned long loop_time_start = 0;
unsigned long loop_time_end = 0;
do {
if (ctx->sig_flag) {
break;
}
//get network input data.
prep_params = &ctx->preproc_ctx->params;
for (i = 0; i < ctx->nn_inf.in_num; i++) {
RVAL_OK(nn_in_file_query(ctx->in_ctx,
&img[i], ea_net_input_name(ctx->nn_inf.net, i)));
if (img[i].tensor_group == NULL) {
EA_LOG_NOTICE("All files are handled\n");
ctx->sig_flag = 1;
break;
}
name = ea_net_input_name(ctx->nn_inf.net, i);
j = get_index_by_name(prep_params, name);
RVAL_ASSERT(j >= 0 && j < ctx->nn_inf.in_num);
in_params = &ctx->in_ctx->file_params;
in_tensor = nn_inference_input(&ctx->nn_inf, i);
// Raw file doesn't need to do preprocess, update net input to resource image tensor directly.
if (in_params->file_type[j] == FILE_RAW) {
RVAL_OK(ea_net_update_input(ctx->nn_inf.net,
name, img[i].tensor_group[0]));
}
}
RVAL_BREAK();
if (ctx->sig_flag) {
break;
}
for (k = 0; k < params->total_loop_count; k++) {
if (ctx->sig_flag) {
break;
}
// network preprocess process
for (i = 0; i < ctx->nn_inf.in_num; i++) {
name = ea_net_input_name(ctx->nn_inf.net, i);
j = get_index_by_name(prep_params, name);
RVAL_ASSERT(j >= 0 && j < ctx->nn_inf.in_num);
in_params = &ctx->in_ctx->file_params;
in_tensor = nn_inference_input(&ctx->nn_inf, i);
if (in_params->file_type[j] != FILE_RAW) {
RVAL_OK(get_file_prep_type(img[i].tensor_group[0], ea_tensor_shape(in_tensor),
prep_params, j, &prep_type));
tmp = &in_tensor;
RVAL_OK(nn_prep_process(ctx->preproc_ctx,
img[i].tensor_group[0], tmp, prep_type, j));
}
}
RVAL_BREAK();
time_start = ea_gettime_us();
loop_time_start = time_start;
RVAL_OK(nn_inference_process(&ctx->nn_inf));
fp16tofp32_time = (double)ea_net_params(ctx->nn_inf.net)->fp16tofp32_time_us / 1000;
fp16tofp32_flag = ea_net_params(ctx->nn_inf.net)->fp16tofp32_time_us == 0 ? 0 : 1;
interval_fp16tofp32_time += fp16tofp32_time;
total_fp16tofp32_time += fp16tofp32_time;
vp_time = (double)ea_net_params(ctx->nn_inf.net)->vp_time_us / 1000;
total_vp_time += vp_time;
interval_vp_time += vp_time;
// network postprocessing process.
time_start = ea_gettime_us();
RVAL_OK(nn_postp_process(ctx->postproc_ctx, ctx->vp_output, &img[img_src_id], bgr));
time_end = ea_gettime_us();
post_time += (double)(time_end - time_start) / 1000.0;
interval_post_time += (double)(time_end - time_start) / 1000.0;
loop_time_end = time_end;
total_loop_time += (double)(loop_time_end - loop_time_start) / 1000.0;
interval_loop_time += (double)(loop_time_end - loop_time_start) / 1000.0;
interval_count++;
if (interval_count % params->interval_loop_count == 0) {
if (interval_count != params->total_loop_count) {
printf("CVflow Average Execute Time: %f us\n", interval_vp_time / params->interval_loop_count * 1000);
if (fp16tofp32_flag == 1) {
printf("CPU Average FP16 To FP32 Time: %f us\n", interval_fp16tofp32_time / params->interval_loop_count * 1000);
}
if ((params->arm_nn_name != NULL) &&
(strncmp(params->arm_nn_name, TO_FILE_POSTPROCESS_NAME, strlen(TO_FILE_POSTPROCESS_NAME)) != 0)) {
printf("CPU Postprocess Average Time: %f us\n", interval_post_time / params->interval_loop_count * 1000);
}
printf("Total Loop Average Time: %f us\n", interval_loop_time / params->interval_loop_count * 1000);
}
interval_vp_time = 0.0;
interval_fp16tofp32_time = 0.0;
interval_post_time = 0.0;
interval_loop_time = 0.0;
}
}
RVAL_BREAK();
if (ctx->sig_flag) {
break;
}
// network shows results.
if (ctx->display != NULL) {
display_obj_params = ea_display_obj_params(ctx->display);
display_obj_params->src_img_path = img[0].file_path;
}
nn_show_results_process(&ctx->postproc_ctx->nn_arm_ctx, &img[0], img_src_id);
printf("CVflow Average Execute Time: %f us\n", total_vp_time / params->total_loop_count * 1000);
if (fp16tofp32_flag == 1) {
printf("CPU Average FP16 To FP32 Time: %f us\n", total_fp16tofp32_time / params->total_loop_count * 1000);
}
if ((params->arm_nn_name != NULL) &&
(strncmp(params->arm_nn_name, TO_FILE_POSTPROCESS_NAME, strlen(TO_FILE_POSTPROCESS_NAME)) != 0)) {
printf("CPU Postprocess Average Time: %f us\n", post_time / params->total_loop_count * 1000);
}
printf("Total Loop Average Time: %f us\n", total_loop_time / params->total_loop_count * 1000);
total_vp_time = 0.0;
interval_vp_time = 0.0;
interval_fp16tofp32_time = 0.0;

static int run_live_loop(test_eazyai_ctx_t *ctx, test_eazyai_params_t *params)
{
int rval = EA_SUCCESS;
ea_calc_fps_ctx_t calc_fps_ctx;
float fps;
memset(&calc_fps_ctx, 0, sizeof(ea_calc_fps_ctx_t));
calc_fps_ctx.count_period = params->total_loop_count;
int fps_notice_flag = 0;
ea_img_resource_data_t img[NN_MAX_PORT_NUM];
int img_src_id = 0;
ea_tensor_t *bgr = NULL;
memset(img, 0, sizeof(img));
nn_prep_type_t prep_type;
nn_prep_params_t *prep_params = NULL;
ea_tensor_t **tmp = NULL;
ea_tensor_t *in_tensor = NULL;
unsigned long time_start = 0;
unsigned long time_end = 0;
double fp16tofp32_time = 0.0;
double total_fp16tofp32_time = 0.0;
int fp16tofp32_flag = 0;
uint32_t loop_count = 0;
double vp_time = 0.0;
double total_vp_time = 0.0;
double post_time = 0.0;
do {
if (ctx->sig_flag) {
break;
}
//get network input data.
img_src_id = ((params->in_type_map & NN_IN_SRC_PYRAMID_BUF) == NN_IN_SRC_PYRAMID_BUF) ? params->pyramid[1] : 0;
RVAL_OK(nn_in_live_query(ctx->in_ctx, &img[0]));
// network preprocess process
prep_params = &ctx->preproc_ctx->params;
if (params->nn_in_cf == YUV_NV12 && ctx->nn_inf.in_num == 2) {
tmp = &(ctx->yuv_tensor);
RVAL_OK(get_live_prep_type(img[0].tensor_group[img_src_id],
NULL, prep_params, 0, &prep_type));
} else {
in_tensor = nn_inference_input(&ctx->nn_inf, 0);
tmp = &in_tensor;
RVAL_OK(get_live_prep_type(img[0].tensor_group[img_src_id],
ea_tensor_shape(in_tensor), prep_params, 0, &prep_type));
}
RVAL_OK(nn_prep_process(ctx->preproc_ctx,
img[0].tensor_group[img_src_id], tmp, prep_type, 0));
// network inference.
RVAL_OK(nn_inference_process(&ctx->nn_inf));
fp16tofp32_time = (double)ea_net_params(ctx->nn_inf.net)->fp16tofp32_time_us / 1000;
fp16tofp32_flag = ea_net_params(ctx->nn_inf.net)->fp16tofp32_time_us == 0 ? 0 : 1;
total_fp16tofp32_time += fp16tofp32_time;
vp_time = (double)ea_net_params(ctx->nn_inf.net)->vp_time_us / 1000;
total_vp_time += vp_time;
fps = ea_calc_fps(&calc_fps_ctx);
if (fps > 0) {
if (fps_notice_flag == 0) {
EA_LOG_NOTICE("!!! FPS based on frame query, preprocess, inference, postprocess and showing results.\n");
fps_notice_flag = 1;
}
EA_LOG_NOTICE("FPS %.1f\n", fps);
}
// network postprocessing process.
if (params->enable_hold_img_flag == 1) {
RVAL_OK(convert_yuv2bgr_data(params->in_type_map,
&img[0], &bgr, params->pyramid[1]));
}
time_start = ea_gettime_us();
RVAL_OK(nn_postp_process(ctx->postproc_ctx, ctx->vp_output, &img[img_src_id], bgr));
time_end = ea_gettime_us();
post_time += (double)(time_end - time_start) / 1000.0;
// network shows results.
nn_show_results_process(&ctx->postproc_ctx->nn_arm_ctx, &img[0], img_src_id);
loop_count++;
if (loop_count == params->total_loop_count) {
EA_LOG_NOTICE("CVflow Average Execute Time: %f us\n", total_vp_time / params->total_loop_count * 1000);
if (fp16tofp32_flag == 1) {
EA_LOG_NOTICE("CPU Average FP16 To FP32 Time: %f us\n", total_fp16tofp32_time / params->total_loop_count * 1000);
}
EA_LOG_NOTICE("CPU Postprocess Average Time: %f us\n", (double)post_time / params->total_loop_count * 1000);
loop_count = 0;
total_vp_time = 0.0;
total_fp16tofp32_time = 0.0;
post_time = 0.0;
}
} while (ctx->sig_flag == 0);
if (bgr != NULL) {
ea_tensor_free(bgr);
bgr = NULL;
}
return rval;
}
