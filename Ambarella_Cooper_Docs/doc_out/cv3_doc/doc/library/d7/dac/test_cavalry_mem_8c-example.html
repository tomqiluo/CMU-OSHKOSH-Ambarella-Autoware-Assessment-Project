static int init_param(int argc, char **argv, struct cv_mem_ctx *pctx)
{
int ch = 0;
int option_index = 0;
int mem_id = 0;
unsigned long long_value = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 's':
long_value = strtoul(optarg, NULL, 0);
if (long_value > MAX_MEM_SIZE) {
printf("Memory size [%lu] is too large [%lu].\n", long_value, MAX_MEM_SIZE);
return -1;
}
mem_id = pctx->num;
pctx->size[mem_id] = long_value;
pctx->num++;
pctx->alloc_flag = 1;
break;
case 'd':
long_value = strtoul(optarg, NULL, 0);
if (long_value > MAX_MEM_SIZE) {
printf("Memory size [%lu] is too large [%lu].\n", long_value, MAX_MEM_SIZE);
return -1;
}
mem_id = pctx->num;
pctx->addr[mem_id] = long_value;
pctx->num++;
pctx->free_flag = 1;
break;
case 'm':
long_value = strtoul(optarg, NULL, 0);
if (long_value > MAX_MEM_SIZE) {
printf("Memory size [%lu] is too large [%lu].\n", long_value, MAX_MEM_SIZE);
return -1;
}
pctx->memcpy_size = long_value;
pctx->memcpy_flag = 1;
break;
case PERF_DMACOPY:
long_value = strtoul(optarg, NULL, 0);
if (long_value > MAX_MEM_SIZE) {
printf("Memory size [%lu] is too large [%lu].\n", long_value, MAX_MEM_SIZE);
return -1;
}
pctx->memcpy_size = long_value;
pctx->dmacpy_flag = 1;
break;
case PERF_DMACOPY_START_SIZE:
pctx->perf_dma_copy_start_size = strtoul(optarg, NULL, 0);
break;
case PERF_DMACOPY_INC_SIZE:
pctx->perf_dma_copy_inc_size = strtoul(optarg, NULL, 0);
break;
case PERF_DMACOPY_INC_ADDR:
pctx->perf_dma_copy_inc_addr = strtoul(optarg, NULL, 0);
break;
case LOOP_CAVALRY_MEM:
pctx->loop_cvmem_size = strtoul(optarg, NULL, 0);
pctx->stress_loop_cvmem_flag = 1;
printf("Stress cavalry memory size: 0x%lx (%lu MB)\n",
pctx->loop_cvmem_size, (pctx->loop_cvmem_size >> 20));
break;
case LOOP_OS_MEM:
pctx->loop_osmem_size = strtoul(optarg, NULL, 0);
pctx->stress_loop_osmem_flag = 1;
printf("Stress OS user memory size: 0x%lx (%lu MB)\n",
pctx->loop_osmem_size, (pctx->loop_osmem_size >> 20));
break;
case 't':
pctx->iteration_num = atoi(optarg);
printf("Iteration num: %u\n", pctx->iteration_num);
break;
case 'c':
pctx->cache_en_flag = 1;
break;
case 'y':
pctx->auto_recycle_flag = 1;
break;
case 'p':
pctx->share_to_dsp_flag = 1;
break;
case 'g':
pctx->usage_mem_flag = 1;
break;
case 'f':
pctx->mem_fd_flag = 1;
break;
case QUERY_BUF_PHYS_ADDR:
long_value = strtoul(optarg, NULL, 0);
if ((long_value > CAVALRY_MEM_LAST)) {
printf("Invalid buf_id: %lu, range [0, %u]\n",
long_value, CAVALRY_MEM_LAST);
return -1;
}
pctx->query_buf_id = long_value;
pctx->query_buf_phys_flag = 1;
break;
case SHOW_ALL_CAVALRY_BUF:
pctx->show_all_buf_phys_flag = 1;
break;
case 'v':
pctx->verbose_flag = 1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int mmap_write(struct cv_mem_ctx *pctx, struct cavalry_mem *mem)
{
uint8_t *virt = NULL;
int rval = 0;
virt = mmap(NULL, mem->length, PROT_WRITE, MAP_SHARED,
pctx->fd_cav, mem->offset);
if (virt == MAP_FAILED) {
perror("mmap\n");
return -1;
}
memset(virt, 0xde, mem->length);
if (munmap(virt, mem->length) < 0) {
perror("munmap");
rval = -1;
}
virt = NULL;
return rval;
}
static int mmap_write_memfd(struct cavalry_mfd_alloc *mem)
{
uint8_t *virt = NULL;
int rval = 0;
virt = mmap(NULL, mem->length, PROT_WRITE, MAP_SHARED, mem->fd, 0);
if (virt == MAP_FAILED) {
perror("mmap\n");
return -1;
}
memset(virt, 0xdf, mem->length);
if (munmap(virt, mem->length) < 0) {
perror("munmap");
rval = -1;
}
virt = NULL;
return rval;
}
static int alloc_mem(struct cv_mem_ctx *pctx)
{
struct cavalry_mem mem;
int i = 0, rval = 0;
for (i = 0; i < pctx->num; i++) {
memset(&mem, 0, sizeof(mem));
mem.length = pctx->size[i];
mem.cache_en = pctx->cache_en_flag;
mem.auto_recycle = pctx->auto_recycle_flag;
mem.share_to_dsp = pctx->share_to_dsp_flag;
if (ioctl(pctx->fd_cav, CAVALRY_ALLOC_MEM, &mem) < 0) {
perror("CAVALRY_ALLOC_MEM");
rval = -1;
break;
}
printf("Alloc cv_mem addr: 0x%lx, size: 0x%lx, cache: %d, auto_recycle: %d\n",
mem.offset, mem.length, mem.cache_en, mem.auto_recycle);
if (mmap_write(pctx, &mem) < 0) {
printf("mmap_write err\n");
rval = -1;
break;
}
}
return rval;
}
static int alloc_memfd(struct cv_mem_ctx *pctx)
{
struct cavalry_mfd_alloc mem;
int i = 0, rval = 0;
for (i = 0; i < pctx->num; i++) {
memset(&mem, 0, sizeof(mem));
mem.length = pctx->size[i];
mem.cache_en = pctx->cache_en_flag;
mem.share_to_dsp = pctx->share_to_dsp_flag;
if (ioctl(pctx->fd_cav, CAVALRY_ALLOC_MEMFD, &mem) < 0) {

rval = -1;
break;

}

}

static int perf_memcpy(struct cv_mem_ctx *pctx)
{
void *virt = NULL, *user = NULL;
unsigned long phys = 0;
unsigned long size = 0;
struct cavalry_mem_attr attr = {0};
int rval = 0;
MEASURE_TIME_VAR();
attr.cache_en = pctx->cache_en_flag;
attr.no_recycle = pctx->auto_recycle_flag;
attr.share_to_dsp = pctx->share_to_dsp_flag;
size = pctx->memcpy_size;
do {
/* cavalry_mem */
if (cavalry_mem_alloc_with_attr(size, &phys, &virt, &attr) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(virt, 0, size);
/* user_mem */
user = malloc(size);
if (user == NULL) {
printf("malloc err\n");
rval = -1;
break;
}
memset(user, MAGIC_DATA, size);
printf("[%s], memory size: 0x%lx, cma pointer: %p, malloc pointer: %p\n",
pctx->cache_en_flag ? "cache" : "non-cache", size, virt, user);
MEASURE_TIME_START();
/* arm write: user_mem -> cavalry_mem */
memcpy(virt, user, size);
if (pctx->cache_en_flag) {
cavalry_mem_sync_cache(size, phys, 1, 0);
}
MEASURE_TIME_END("ARM Write");
/* arm read: cavalry_mem -> user_mem */
if (pctx->cache_en_flag) {
cavalry_mem_sync_cache(size, phys, 0, 1);
}
memcpy(user, virt, size);
MEASURE_TIME_END("ARM Read ");
} while (0);
if (virt) {
cavalry_mem_free(size, phys, virt);
virt = NULL;
}
if (user) {
free(user);
user = NULL;
}
return rval;
}
static int perf_memcpy_mfd(struct cv_mem_ctx *pctx)
{
void *virt = NULL, *user = NULL;
int fd = 0;
unsigned long size = 0;
int rval = 0;
MEASURE_TIME_VAR();
size = pctx->memcpy_size;
do {
/* cavalry_mem */
if (cavalry_mem_alloc_mfd(size, &fd, &virt, pctx->cache_en_flag) < 0) {
printf("cavalry_mem_alloc_mfd err\n");
rval = -1;
break;
}
memset(virt, 0, size);
/* user_mem */
user = malloc(size);
if (user == NULL) {
printf("malloc err\n");
rval = -1;
break;
}
memset(user, MAGIC_DATA, size);
printf("[%s], memory size: 0x%lx, fd: %d, dmabuf pointer: %p, malloc pointer: %p\n",
pctx->cache_en_flag ? "cache" : "non-cache", size, fd, virt, user);
MEASURE_TIME_START();
/* arm write: user_mem -> cavalry_mem */
memcpy(virt, user, size);
if (pctx->cache_en_flag) {
cavalry_mem_sync_cache_mfd(size, 0, fd, 1, 0);
}
MEASURE_TIME_END("ARM Write");
/* arm read: cavalry_mem -> user_mem */
if (pctx->cache_en_flag) {
cavalry_mem_sync_cache_mfd(size, 0, fd, 0, 1);
}
memcpy(user, virt, size);
MEASURE_TIME_END("ARM Read ");
} while (0);
if (virt) {
cavalry_mem_free_mfd(size, fd, virt);
virt = NULL;
}
if (user) {
free(user);
user = NULL;
}
return rval;
}
static int check_buffer_data(char *buffer, char data, uint32_t size)
{
uint32_t i;
int rval = 0;
if (!size) {
return 0;
}
for (i = 0; i < size; i++) {
if (data != buffer[i]) {
printf("Index [%u] mismatch, real: %u, expect: %u\n", i, buffer[i], data);
rval = -1;
break;
}
}
return rval;
}
static int perf_dma_copy(struct cv_mem_ctx *pctx)
{
struct cavalry_dma_copy gdma = {0};
void *src_virt = NULL, *dst_virt = NULL;
unsigned long src_phys = 0, dst_phys = 0;
unsigned long total_size = 0;
unsigned long do_size, do_addr_offset;
uint32_t i = 0;
int rval = 0;
char write_d, other_d;
total_size = pctx->memcpy_size;
do {
/* cavalry_mem src */
if (cavalry_mem_alloc_with_attr(total_size, &src_phys, &src_virt, NULL) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(src_virt, 0, total_size);
/* cavalry_mem dst */
if (cavalry_mem_alloc_with_attr(total_size, &dst_phys, &dst_virt, NULL) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(src_virt, 0, total_size);
printf("------------------------------------------\n");
printf("dst phys: 0x%lx, src phys: 0x%lx, total_size: 0x%lx\n", dst_phys, src_phys, total_size);
printf("perf dma copy start size: 0x%x, inc_size: 0x%x\n",
pctx->perf_dma_copy_start_size, pctx->perf_dma_copy_inc_size);
printf("------------------------------------------\n");
if (pctx->iteration_num == 0) {
pctx->iteration_num = 0xfffffff0;
}
if (pctx->perf_dma_copy_inc_size == 0) {
printf("Invaild inc_size: %u, please specify it by --inc-size option\n",
pctx->perf_dma_copy_inc_size);
return -1;
}
do_size = pctx->perf_dma_copy_start_size;
do_addr_offset = 0;
for (i = 0; i < pctx->iteration_num; i++) {
if (!G_running) {
break;
}
printf("do_addr_offset: 0x%lx, do_size: 0x%lx\n", do_addr_offset, do_size);
write_d = i + 1;
other_d = i;
memset(dst_virt, other_d, total_size);
memset(src_virt, write_d, do_size);
/* dma copy */
#if 1
gdma.src_phys = src_phys;
gdma.dst_phys = dst_phys + do_addr_offset;
gdma.size = do_size;
if (ioctl(pctx->fd_cav, CAVALRY_DMA_COPY, &gdma) < 0) {
perror("CAVALRY_DMA_COPY");
rval = -1;
break;
}
#else
memcpy(dst_virt + do_addr_offset, src_virt, do_size); /* always pass */
#endif
/* check write area if equal to write_d */
rval = memcmp(dst_virt + do_addr_offset, src_virt, do_size);
if (rval) {
printf("ERROR: %d copy mismatch at do_size: 0x%lx\n", rval, do_size);
break;
}
/* check other area if keep as other_d */
rval = check_buffer_data(dst_virt, other_d, do_addr_offset); // head
if (rval) {
printf("ERROR: check other data in head err, size: 0x%lx\n", do_addr_offset);
break;
}
rval = check_buffer_data(dst_virt + do_addr_offset + do_size, other_d,
total_size - do_addr_offset - do_size); // tail
if (rval) {
printf("ERROR: check other data in tail err, offset: 0x%lx, size: 0x%lx\n",
do_addr_offset + do_size, total_size - do_addr_offset - do_size);
break;
}
do_size += pctx->perf_dma_copy_inc_size;
do_addr_offset += pctx->perf_dma_copy_inc_addr;
if (do_size >= total_size) {
do_size = pctx->perf_dma_copy_start_size;
}
if ((do_addr_offset + do_size) >= total_size) {
do_addr_offset = 0;
}
}
} while (0);
if (src_virt) {
cavalry_mem_free(total_size, src_phys, src_virt);
src_virt = NULL;
}
if (dst_virt) {
cavalry_mem_free(total_size, dst_phys, dst_virt);
dst_virt = NULL;
}
return rval;
}
static int loop_alloc_free_cavalry_mem(struct cv_mem_ctx *pctx)
{
void *virt = NULL;
unsigned long phys = 0;
unsigned long size = 0;
unsigned long i = 0;
int rval = 0;
size = pctx->loop_cvmem_size;
