static int create_vect_mfd(vect_desc_mfd_t* vect,cv_mem_mfd_t* vm,
int h, int w, int d, int data_size, int with_cache)
{
vect->shape.p = 1;
vect->shape.d = d;
vect->shape.h = h;
vect->shape.w = w;
vect->pitch = ALIGN_PITCH(w * data_size);
vect->color_space = CS_VECT;
vm->size = vect->pitch * vect->shape.h * vect->shape.d;
if (cavalry_mem_alloc_mfd(vm->size, &vm->mfd, &vm->virt, with_cache) < 0) {
perror("cavalry_mem_alloc_mfd err\n");
return -1;
}
vect->data_addr_fd = vm->mfd;
return vm->size;
}
static int free_vmem_mfd(cv_mem_mfd_t* vm)
{
if (vm->virt != NULL) {
if (cavalry_mem_free_mfd(vm->size , vm->mfd, vm->virt) < 0) {
printf("cavalry_mem_free_mfd err\n");
return -1;
}
}
return 0;
}

static int get_bracket_data_float(const char* s_data, float* v_data)
{
int bracket_pos = -1;
char c_data[1024];
snprintf(c_data, sizeof(c_data), "%s", s_data);
string ss_data = c_data;
bracket_pos = ss_data.find("[");
if (bracket_pos >= 0) {
ss_data.replace(bracket_pos, 1, "");
}
bracket_pos = ss_data.find("]");
if (bracket_pos >= 0) {
ss_data.replace(bracket_pos, 1, "");
}
char *data_i;
data_i = strtok((char*)ss_data.c_str(), ",");
int data_num = 0;
while (data_i) {
v_data[data_num] = atof(data_i);
data_i=strtok(NULL, ",");
data_num++;
}
return data_num;
}
int init_param(int argc, char **argv)
{
int ch;
int ker_size=0;
int option_index = 0;
int value, height, width;
input_df.sign = 0;
input_df.datasize = 0;
input_df.exp_offset = 0;
input_df.exp_bits = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("ERR: Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(bin_name, sizeof(bin_name), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("ERR: Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(input_name, sizeof(input_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("ERR: filepath [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 'm':
morph_method = atoi(optarg);
break;
case 't':
bw_threshold = atoi(optarg);
if(bw_threshold<=0 || bw_threshold>=255){
printf("the allowed bw threshold is >0 and <255\n");
return -1;
}
break;
case 'r':
morph_border_replicated= atoi(optarg);
if(morph_border_replicated !=0 && morph_border_replicated != 1){
printf("the allowed border relicated mode is 0 or 1\n");
return -1;
}
break;
case 's':
get_vect_resolution(optarg,&width, &height);
iw = width;
ih = height;
ic=1;
printf("input dimensions [%d,%d,%d]\n",ic,iw,ih);
break;
case 'f':
get_vect_resolution(optarg,&morph_w, &morph_h);
printf("morph size[width,height]=[%d,%d]\n",morph_w,morph_h);
break;
case 'k':
ker_size=get_bracket_data_float(optarg, ker_data);
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
if (ker_size > 0) {
printf("filter kernel data =>\n");
int ker_rows = (int)(ker_size / morph_w);
int last_cols = (int)(ker_size % morph_w);
float sum_val = 0;
for (int i = 0; i < ker_rows; i++) {
for (int j = 0; j<morph_w; j++) {
printf("%f," ,ker_data[i * morph_w + j]);
sum_val = sum_val + ker_data[i * morph_w + j];
}
printf("\n");
}
for (int i = 0;i < last_cols; i++) {
printf("%f,", ker_data[ker_rows * morph_w + i]);
sum_val = sum_val + ker_data[ker_rows * morph_w + i];
}
printf("\nmorph kernel data sum=%f\n", sum_val);
if (morph_w * morph_h != ker_size) {
printf("ERR: morph kernel size mismatch the data size,[%d*%d vs %d]\n",
morph_h, morph_w, ker_size);
return -1;
}
}
return 0;
}
static int load_bw_img_ufix8_0_bin(char *img_file, uint8_t *dst_addr,
vect_desc_mfd_t *vect_desc, uint32_t *size, int bw_thresh)
{
Mat rgb, gray;
uint8_t* gray_data;
rgb = imread(img_file);
resize(rgb, rgb, Size(vect_desc->shape.w, vect_desc->shape.h), INTER_AREA);
if (rgb.empty()) {
return -1;
}
cvtColor(rgb, gray, COLOR_BGR2GRAY);
gray_data = (uint8_t*)gray.data;
for (uint32_t i =0; i < vect_desc->shape.h; i++) {
for (uint32_t j = 0; j < vect_desc->shape.w; j++) {
if (gray_data[i * vect_desc->shape.w + j] < bw_thresh) {
dst_addr[i * vect_desc->pitch + j] = 0;
gray_data[i * vect_desc->shape.w + j] = 0;
} else {
dst_addr[i * vect_desc->pitch + j] = 1;
gray_data[i * vect_desc->shape.w + j] = 255;
}
}
}
*size = vect_desc->pitch * vect_desc->shape.h;
return (*size);
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int store_bw_img_ufix8_0(char *file, int im_h, int im_pitch, void *addr)
{
Mat img;
uint8_t* data_ptr = (uint8_t*)addr;
img.create(im_h, im_pitch, CV_8UC1);
uint8_t* Mat_data = (uint8_t*)img.data;
for (int i = 0;i<im_h;i++) {
for (int j=0; j < im_pitch; j++) {
Mat_data[i * im_pitch + j] = data_ptr[i * im_pitch + j] * 255;
}
}
imwrite(file, img);
return 0;
}

int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
cv_mem_mfd_t lib_mem;
cv_mem_mfd_t in_mem;
cv_mem_mfd_t out_mem;
vect_desc_mfd_t im_vect, out_vect;
cv_mem_mfd_t kernel_mem;
vect_desc_mfd_t kernel_vect;
cvker_mfd_t ker_cfg;
uint32_t size = 0;
int rval = -1;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open("/dev/cavalry", O_RDWR, 0)) < 0) {
perror("/dev/cavalry");
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
memset(&im_vect, 0, sizeof(im_vect));
memset(&out_vect, 0, sizeof(out_vect));
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&in_mem, 0, sizeof(in_mem));
memset(&out_mem, 0, sizeof(out_mem));
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc_mfd(lib_mem.size, &lib_mem.mfd, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc_mfd err\n");
rval = -1;
break;
}
vproc_load_mfd(fd_cavalry, (uint8_t*)lib_mem.virt, lib_mem.mfd, size);
if (!ip || !op) {
ip = ALIGN_PITCH(iw * in_byte_width);
op = ALIGN_PITCH(iw * out_byte_width);
}
create_vect_mfd(&im_vect, &in_mem, ih, iw, 1, sizeof(uint8_t), 1);
create_vect_mfd(&out_vect, &out_mem, ih, iw, 1, sizeof(uint8_t), 1);
create_vect_mfd(&kernel_vect, &kernel_mem, 1, BMORPH_KERNEL_MEM_SIZE, 1, sizeof(uint8_t), 1);
ker_cfg.ker_content = ker_data;
ker_cfg.ker_h = morph_h;
ker_cfg.ker_w = morph_w;
ker_cfg.mem_mfd = kernel_mem.mfd;
ker_cfg.mem_size = kernel_mem.size;
ker_cfg.mem_virt = kernel_mem.virt;
ker_cfg.mem_addr_offset = kernel_vect.data_addr_offset;
if (load_bw_img_ufix8_0_bin(input_name, (uint8_t*)in_mem.virt, &im_vect,
&size, bw_threshold) < 0) {
printf("load bw image fails!\n");
rval = -1;
break;
}
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
cavalry_mem_sync_cache_mfd(in_mem.size, 0, in_mem.mfd, 1, 0); //clean cache,write to DRAM
if (vproc_morph_mfd(&im_vect, &out_vect,&ker_cfg, (morph_methods_t)morph_method) < 0) {
rval = -1;
break;
}
cavalry_mem_sync_cache_mfd(out_mem.size, 0, out_mem.mfd, 0, 1);//invalid cache, write to cache
cavalry_mem_sync_cache_mfd(in_mem.size, 0, in_mem.mfd, 0, 1); //invalid cache, write to cache
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("Whole vproc_morph_mfd_ext takes %ld us\n",
(tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
if (output_flag) {
char fp[1500];
snprintf(fp, sizeof(fp),"%s/morph_result_out.bin",output_name);
if (store_bin(fp, out_mem.size, out_mem.virt) < 0) {
printf("store output morph_result_out.bin fails!\n");
rval = -1;
break;
}
snprintf(fp, sizeof(fp),"%s/morph_result.jpg",output_name);
if(store_bw_img_ufix8_0(fp, ih,op,out_mem.virt)<0){
printf("store output morph_result.jpg fails!\n");
rval = -1;
break;
}
}
} while (0);
vproc_exit();
free_vmem_mfd(&out_mem);
free_vmem_mfd(&in_mem);
free_vmem_mfd(&lib_mem);
free_vmem_mfd(&kernel_mem);
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
