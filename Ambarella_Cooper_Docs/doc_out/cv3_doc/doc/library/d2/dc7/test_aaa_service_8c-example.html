static int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'a':
break;
case 'i':
interactive_enble = 1;
break;
case 's'://only for debug
display_enable = 1;
break;
case 'd':
debug_enble = 1;
break;
case 'c':
calibration_enable = 1;
calibration_mode = atoi(optarg);
break;
case 'p':
load_product_info_enable = 1;
snprintf(product_file, IF_FILE_NAME_SIZE, "%s", optarg);
product_file[IF_FILE_NAME_SIZE - 1] = '\0';
break;
case 'N':
channel_id = atoi(optarg);
break;
default:
printf("unknown option %c\n", ch);
return -1;
}
}
return 0;
}

static int global_setting(int *aaa_running)
{
int i, exit_flag, error_opt;
char ch;
show_global_setting_menu();
ch = getchar();
while(ch){
exit_flag = 0;
error_opt = 0;
switch (ch) {
case 'a':
if(display_aaa_lib_version() < 0){
printf("display_aaa_lib_version error!");
return -1;
}
break;
case 'v':
if(display_img_flow_lib_version() < 0){
printf("display_img_flow_lib_version error!");
return -1;
}
break;
case 'j':
if(display_adj_parameter_version(current_chan_id) < 0){
printf("display_adj_parameter_version error!");
return -1;
}
break;
case 'e':
if(display_aeb_parameter_version(current_chan_id) < 0){
printf("display_aeb_parameter_version error!");
return -1;
}
break;
case 'l':
printf("\nInput log option: (ae=0x1, awb=0x2, af=0x4, parser=0x8, main=0x10, hdr=0x20, debug=0x40)\n> ");
SCANF_INT(&i);
if(imgflow_set_log_level(i) < 0){ //ae=0x1, awb=0x2, af=0x4, parser=0x8, main=0x10, hdr=0x20, debug=0x40
printf("imgflow_set_log_option error!");
return -1;
}
break;
case 'c':
if(select_chan_id() < 0){
printf("select_chan_id error!");
return -1;
}
break;
case 's':// control aaa flow start or stop for all channels
printf("\nAAA flow is %srunning.\n", *aaa_running ? "" : "not ");
printf("0 - Stop AAA flow, 1 - Start AAA flow\n> ");
SCANF_INT(&i);
if(i == 0){
if(*aaa_running == 1){
destroy_display_thread();
if(destroy_img_flow() < 0){
printf("destroy_img_flow error!\n");
return -1;
}
*aaa_running = 0;
}
}else if (i == 1){
if(*aaa_running == 0){
create_img_flow(fd_iav);
*aaa_running = 1;
}
if(display_enable){
display_loop_flag = 1;
if((pthread_create(&thread_display_on_screen, NULL, (void*)display_thread_proc,NULL) != 0)){
printf("create a new thread fail!\n");
}
}
}else{
printf("Invalid input for this option!\n");
}
break;
case 'i':
if(imgflow_dump_algo_config(NULL, 0) < 0){
printf("imgflow_dump_algo_config error!\n");
return -1;
}
break;
case 'S':
if(imgflow_save_lua_file(LUA_FILE_PATH) < 0){
printf("imgflow_save_lua_file error!\n");
return -1;
}
printf("successfully saved iq_aaa_config.lua, the path is %s\n", LUA_FILE_PATH);
break;
case 'q':
exit_flag = 1;
break;
default:
error_opt = 1;
break;
}
if(exit_flag)
break;
if(error_opt == 0){
show_global_setting_menu();
}
ch = getchar();
}
return 0;
}

static int run_interactive_mode()
{
char ch, error_opt;
int quit_flag;
int aaa_running_flag = 1;

int awb_running_flag = 1;
int adj_running_flag = 1;
show_menu();
ch = getchar();
while(ch){
quit_flag = 0;
error_opt = 0;
switch(ch){
case 'g':
global_setting(&aaa_running_flag);
break;
case 'e':
exposure_setting(&ae_running_flag);
break;
case 'w':
white_balance_setting(&awb_running_flag);
break;
case 'a':
adjustment_setting(&adj_running_flag);
break;
case 'x':
break;

quit_flag = 1;

default:
error_opt = 1;
break;
}
if(quit_flag){
break;
}
if(error_opt == 0){
show_menu();
}
ch = getchar();
}
return 0;
}

void *dds_lib = NULL;
ea_dds_create_publisher ea_dds_create_publisher_f = NULL;
ea_dds_destroy_publisher ea_dds_destroy_publiser_f = NULL;
ea_dds_publisher_write ea_dds_publisher_write_f = NULL;
void *aaa_publisher = NULL;
if (NULL == dds_lib) {
dds_lib = dlopen(EAZY_DDS_LIB_PATH, RTLD_NOW);
if (!dds_lib) {
printf("Failed to load eazy_dds library.\n");
goto DISPLAY_THREAD_EXIT;
}
ea_dds_create_publisher_f = (ea_dds_create_publisher) dlsym(dds_lib, "ea_dds_create_publisher_impl");
ea_dds_destroy_publiser_f = (ea_dds_destroy_publisher) dlsym(dds_lib, "ea_dds_destroy_publisher");
ea_dds_publisher_write_f = (ea_dds_publisher_write) dlsym(dds_lib, "ea_dds_publisher_write");
if (!ea_dds_create_publisher_f || !ea_dds_destroy_publiser_f || !ea_dds_publisher_write_f) {
printf("Failed to dynamic load APIs!\n");
goto DISPLAY_THREAD_EXIT;
}
}
aaa_publisher = ea_dds_create_publisher_f("amba_aaa_info", NULL, sizeof(aaa_debug_info), 0);
if (aaa_publisher == NULL) {
printf("Failed to create aaa publisher for easy dds!\n");
goto DISPLAY_THREAD_EXIT;
}
while(display_loop_flag){
wait_irq_count(1);
if (imgflow_get_video_pts(&aaa_debug_info.pts, current_chan_id) < 0) {
printf("imgflow_get_video_pts error\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_aaa_output(&aaa_output, current_chan_id)<0){
printf("imgflow_get_aaa_output error\n");
goto DISPLAY_THREAD_EXIT;
}

printf("imgflow_get_hdr_sensor_shutter_group error!\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_sensor_agc(sensor_gain_db, current_chan_id) < 0){
printf("imgflow_get_sensor_agc error!\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_ae_target(&ae_target,current_chan_id)<0){
printf("imgflow_get_ae_target error\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_histo_cntl_ratio(&histo_cntl_ratio,current_chan_id)<0){
printf("imgflow_get_histo_cntl_ratio error\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_ae_stable(&ae_stable,current_chan_id)<0){
printf("imgflow_get_ae_stable error\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_ae_hist_stable(&ae_hist_stable,current_chan_id)<0){
printf("imgflow_get_ae_hist_stable error\n");
goto DISPLAY_THREAD_EXIT;
}

printf("imgflow_get_cfa_luma error!\n");
goto DISPLAY_THREAD_EXIT;
}
if(imgflow_get_rgb_luma(&ae_rgb_luma,current_chan_id)){
printf("imgflow_get_rgb_luma error!\n");
goto DISPLAY_THREAD_EXIT;
}
if(get_img_config(img_cfg)<0){
printf("get_img_config error!\n");
goto DISPLAY_THREAD_EXIT;
}
aaa_debug_info.channel_id = current_chan_id;
expo_num=img_cfg[current_chan_id].expo_num;
switch(expo_num) {
case 1:
snprintf(aaa_debug_info.debug_info[3], sizeof(aaa_debug_info.debug_info[3]),"shutter=[%d,0,0],agc=[%d,0,0],extra_dgain=%d,ae_target=%d,ae_luma=%d",
aaa_output.ae_output[0].shutter_index, aaa_output.ae_output[0].agc_index,
aaa_output.extra_gain,ae_target,ae_cfa_luma);
snprintf(aaa_debug_info.debug_info[2], sizeof(aaa_debug_info.debug_info[2]),"shutter=[1/%d,0,0]s,agc=[%d,0,0]db,hist_ratio=%d",
DIV_ROUND(512000000, shutter_time[0]),sensor_gain_db[0]>>24,
histo_cntl_ratio);
break;
case 2:
snprintf(aaa_debug_info.debug_info[3], sizeof(aaa_debug_info.debug_info[3]),"shutter=[%d,%d,0],agc=[%d,%d,0],extra_dgain=%d,ae_target=%d,ae_luma=%d",
aaa_output.ae_output[0].shutter_index, aaa_output.ae_output[1].shutter_index,
aaa_output.ae_output[0].agc_index, aaa_output.ae_output[1].agc_index,
aaa_output.extra_gain,ae_target,ae_cfa_luma);

DIV_ROUND(512000000, shutter_time[0]), DIV_ROUND(512000000, shutter_time[1]),
sensor_gain_db[0]>>24,sensor_gain_db[1]>>24,
histo_cntl_ratio, (float)aaa_output.ae_output[0].shutter_index / aaa_output.ae_output[1].shutter_index);
break;
case 3:
snprintf(aaa_debug_info.debug_info[3], sizeof(aaa_debug_info.debug_info[3]),"shutter=[%d,%d,%d],agc=[%d,%d,%d],extra_dgain=%d,ae_target=%d,ae_luma=%d",
aaa_output.ae_output[0].shutter_index, aaa_output.ae_output[1].shutter_index, aaa_output.ae_output[2].shutter_index,
aaa_output.ae_output[0].agc_index, aaa_output.ae_output[1].agc_index, aaa_output.ae_output[2].agc_index,
aaa_output.extra_gain,ae_target,ae_cfa_luma);
snprintf(aaa_debug_info.debug_info[2], sizeof(aaa_debug_info.debug_info[2]),"shutter=[1/%d,1/%d,1/%d]s,agc=[%d,%d,%d]db",
DIV_ROUND(512000000, shutter_time[0]), DIV_ROUND(512000000, shutter_time[1]), DIV_ROUND(512000000, shutter_time[2]),
sensor_gain_db[0]>>24,sensor_gain_db[1]>>24,sensor_gain_db[2]>>24);
break;
default:
printf("please confirm expo number of channel[%d]\n", current_chan_id);
goto DISPLAY_THREAD_EXIT;
}
snprintf(aaa_debug_info.debug_info[1], sizeof(aaa_debug_info.debug_info[1]),"Rgain=%d,Ggain=%d,Bgain=%d",
aaa_output.awb_output.video_wb_gain.r_gain,aaa_output.awb_output.video_wb_gain.g_gain,aaa_output.awb_output.video_wb_gain.b_gain);

expo_num,ae_stable, ae_hist_stable);
//printf("%s\n",debug_info_str);
if (ea_dds_publisher_write_f(aaa_publisher, &aaa_debug_info) < 0) {
printf("Failed to write data to easy dds!\n");
goto DISPLAY_THREAD_EXIT;
}
}
DISPLAY_THREAD_EXIT:
if (aaa_publisher != NULL && ea_dds_destroy_publiser_f(aaa_publisher) < 0) {
printf("Failder to destory aaa publisher!\n");
}
if (dds_lib) {
dlclose(dds_lib);
dds_lib = NULL;
}
}

{

}
}
if(create_img_flow(fd_iav)<0){
return -1;
}
if(calibration_enable){
usleep(100*1000);

printf("run_calibration_mode error\n");
return -1;
}
printf("calibration calc done!\n");
return 0;
}
if(display_enable){
if((pthread_create(&thread_display_on_screen,NULL,(void*)display_thread_proc,NULL)!=0)){
printf("create a new thread fail!\n");
}
}
if(debug_enble){
}
if(interactive_enble){
usleep(33*1000);
if (run_interactive_mode() < 0) {
printf("run_interactive_mode error\n");
return -1;
}
destroy_display_thread();
if (destroy_img_flow() < 0) {
printf("destroy_img_flow error!\n");
return -1;
}
close(fd_iav);
exit(1);
}
while (1) {
usleep(33*1000);
}
return 0;
