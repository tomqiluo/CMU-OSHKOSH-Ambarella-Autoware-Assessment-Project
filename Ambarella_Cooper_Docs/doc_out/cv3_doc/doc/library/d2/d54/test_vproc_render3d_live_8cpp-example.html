static int create_vect(vect_desc_t *vect, cv_mem_t *vm,
int h, int w, int d, int data_size, int with_cache)
{
vect->shape.p = 1;
vect->shape.d = d;
vect->shape.h = h;
vect->shape.w = w;
vect->pitch = ALIGN_PITCH(w * data_size);
vect->color_space = CS_VECT;
vm->size = vect->pitch * vect->shape.h * vect->shape.d;
if (cavalry_mem_alloc(&vm->size, &vm->phys, &vm->virt, with_cache) < 0) {
perror("cavalry_mem_alloc err\n");
return -1;
}
vect->data_addr = vm->phys;
memset(vm->virt, 0, vm->size);
if (with_cache) {
cavalry_mem_sync_cache(vm->size, vm->phys, 1, 0);
}
return vm->size;
}
static int free_vmem(cv_mem_t* vm)
{
if (vm->virt != NULL) {
if (cavalry_mem_free(vm->size , vm->phys, vm->virt) < 0) {
printf("cavalry_mem_alloc err\n");
return -1;
}
}
return 0;
}
int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value;
uint32_t height, width;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("ERR: Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
strncpy(vproc_bin_path, optarg, sizeof(vproc_bin_path) - 1);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("ERR: Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
strncpy(depth_file, optarg, sizeof(depth_file) - 1);
break;
case 's':
get_vect_resolution(optarg,&width, &height);
iw = width;
ih = height;
printf("input depth map dimensions [%d,%d]\n",iw,ih);
break;
case 'S':
get_vect_resolution(optarg,&width, &height);
prj_w = width;
prj_h = height;
printf("projection dimensions [%d,%d]\n",prj_w,prj_h);
break;
case 'f':
value = atoi(optarg);
dm_focal_length= value;
printf("input depth map focal length %d\n", dm_focal_length);
break;
case 'F':
value = atoi(optarg);
prj_focal_length= value;
printf("projector focal length %d\n", prj_focal_length);
break;
case 'r':
get_rt_input(optarg, &pitch_angle, &roll_angle, &yaw_angle);
printf("Rotations [Pitch, Roll, Yaw] [%d,%d, %d]\n",pitch_angle, roll_angle, yaw_angle);
break;
case 'c':
get_rt_input(optarg, &r3d_cshift[0], &r3d_cshift[1], &r3d_cshift[2]);
printf("Center shift [X, Y, Z] [%d,%d, %d]\n",r3d_cshift[0], r3d_cshift[1], r3d_cshift[2]);
break;
case 't':
get_rt_input(optarg, &r3d_tshift[0], &r3d_tshift[1], &r3d_tshift[2]);
printf("Transfer shift [X, Y, Z] [%d,%d, %d]\n",r3d_tshift[0], r3d_tshift[1], r3d_tshift[2]);
break;
case 'q':
get_pc_depth_range_input(optarg, &pc_depth_range_near, &pc_depth_range_far);
printf("point clounds display range [Z-near, Z-far] [%d,%d]\n",
pc_depth_range_near, pc_depth_range_far);
break;
case 'd':
value = atoi(optarg);
pc_stride = value;
break;
case 'Q':
depth_scale = atof(optarg);
break;
case 'm':
value = atoi(optarg);
if (value >= 0){
run_freq = value;
}else{
run_freq = 0;
}
break;
case 'v':
set_log_level_flag = 1;
libvproc_log_level = atoi(optarg);
break;
case 'D':
dump_dir = optarg;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}

static int load_dm_from_file(const char *file_name, int16_t *buffer, uint32_t size)
{
FILE *in = NULL;
uint32_t file_size = 0;
int rval = 0;
do {
if ((in = fopen(file_name, "rb")) == NULL) {
perror(file_name);
rval = -1;
break;
}
if (fseek(in, 0, SEEK_END) < 0) {
perror("fseek input file");
rval = -1;
break;
}
file_size = ftell(in);
if (fseek(in, 0, SEEK_SET) < 0) {
perror("fseek input file");
rval = -1;
break;
}
if (file_size == size*sizeof(int16_t)) {
if (fread(buffer, sizeof(int16_t), size, in) != size) {
perror("fread input file");
rval = -1;
break;
}
}
} while (0);
if (in) {
fclose(in);
in = NULL;
}
return rval;
}

int16_t *pw_y_buffer = pw_out + dm_h * dm_p;
int16_t *pw_z_buffer = pw_out + 2 * dm_h * dm_p;
for (int y = 0; y < dm_h; y++) {
for (int x = 0; x < dm_w; x++) {
depth_val = dm_buffer[y * dm_w + x] * depth_scale;
if (depth_val >= pc_near_depth_th && depth_val <= pc_far_depth_th) {
pw_x_buffer[y * dm_p + x] = (int16_t)((x - u0) * depth_val / focal_length);
pw_y_buffer[y * dm_p + x] = (int16_t)((y - v0) * depth_val / focal_length);
pw_z_buffer[y * dm_p + x] = depth_val;
avg_depth += depth_val;
avg_cnt++;
}
}
}
free(dm_buffer);
dm_buffer=NULL;
if (avg_cnt) {
avg_depth = avg_depth / avg_cnt;
}
printf("avg cnt is %d, avg depth is %f\n", avg_cnt, avg_depth);
return 0;
}
static uint32_t get_dim_width_size(uint32_t width, uint32_t data_fmt_size, uint32_t bitsize)
{
uint32_t width_line_size = 0;
if (bitsize) {
switch (bitsize) {
case 1:
width_line_size = (width + 7) >> 3;
break;
case 4:
width_line_size = (width + 3) >> 1;
break;
case 8:
width_line_size = width;
break;
case 16:
width_line_size = width << 1;
break;
case 32:
width_line_size = width << 2;
break;
default:
width_line_size = 1;
break;
}
} else {
switch (data_fmt_size) {
case 0:
width_line_size = width; /* 8bit */
break;
case 1:
width_line_size = width << 1; /* 16bit */
break;
case 2:
width_line_size = width << 2; /* 32bit */
break;
default:
width_line_size = 1;
break;
}
}
return width_line_size;
}
static int store_vector_no_padding_file(const char *file_name, void *virt_addr,
vect_shape_t *dim, data_format_t *data_fmt, uint32_t final_pitch)
{
FILE *out = NULL;
uint8_t *virt = NULL;
uint32_t one_line = 0, column = 0;
uint32_t c = 0;
int rval = 0;
virt = (uint8_t *)virt_addr;
one_line = get_dim_width_size(dim->w, data_fmt->datasize, 0);
column = dim->p * dim->d * dim->h;
out = fopen(file_name, "w+b");
if (out == NULL) {
perror(file_name);
printf("fopen error\n");
return -1;
}
for (c = 0; c < column; c++) {
if (fwrite(virt, 1, one_line, out) != one_line) {
perror(file_name);
printf("fwrite error\n");
rval = -1;
break;
}
virt += final_pitch;
}
if (out) {
fclose(out);
out = NULL;
}
return rval;
}
