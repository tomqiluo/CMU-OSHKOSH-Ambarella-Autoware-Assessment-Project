void usage(void)
{
unsigned long i;
printf("test_vproc_yuv2rgb usage:\n");
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");

int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, width, height;
memset(&input_roi, 0, sizeof(input_roi));
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(bin_name, sizeof(bin_name), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(input_name, sizeof(input_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 's':
get_vect_resolution(optarg, &width, &height);
iw = width;
ih = height;
break;
case 'p':
ip = atoi(optarg);
if (CHECK_PITCH_ALIGN(ip)) {
printf("Pitch [%d] is not %u-byte aligned.\n",
ip, CAVALRY_PORT_PITCH_ALIGN);
return -1;
}
break;
case 'R':
get_roi_area(optarg, &input_roi);
break;
case 'y':
cs_yuv = atoi(optarg);
if (cs_yuv < CS_YUV_FIRST || cs_yuv >= CS_YUV_LAST) {
printf("Invalid yuv format %d.\n", cs_yuv);
return -1;
}
break;
case 'r':
cs_rgb = atoi(optarg);
if (cs_rgb == 0) {
cs_rgb = CS_RGB;
} else if (cs_rgb == 1) {
cs_rgb = CS_BGR;
}
if (cs_rgb < CS_RGB_FIRST || cs_rgb >= CS_RGB_LAST) {
printf("Invalid RGB format %d.\n", cs_rgb);
return -1;
}
break;
case 'f':
matrix_format = atoi(optarg);
if (matrix_format > 1) {
printf("Invalid matrix format: %d\n", matrix_format);
return -1;
}
break;
case 'm':
feed_mem_flag = 1;
break;
case 'H':
high_accuracy_mode = 1;
break;
case 'N':
usr_nvp_affinity = strtoul(optarg, NULL, 0);
break;
case 'd':
set_log_level_flag = 1;
libvproc_log_level = atoi(optarg);
break;
case 't':
measure_time_flag = 1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int load_vect_bin(char *file, void *dst_addr,
vect_desc_t *vect_desc, uint32_t *size)
{
FILE *fp = NULL;
int rval = 0;
uint32_t i, data_sz;
uint32_t ch, ht, wd, pch, llen;
unsigned long file_length;
do {
if ((fp = fopen(file, "rb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fseek(fp, 0, SEEK_END) < 0) {
perror("SEEK_END");
rval = -1;
break;
}
file_length = ftell(fp);
if (fseek(fp, 0, SEEK_SET) < 0) {
perror("SEEK_SET");
rval = -1;
break;
}
ch = vect_desc->shape.d;
ht = vect_desc->shape.h;
wd = vect_desc->shape.w;
pch = vect_desc->pitch;
data_sz = (1<<vect_desc->data_format.datasize);
if (vect_desc->color_space >= CS_YUV_FIRST) {
switch(vect_desc->color_space) {
case CS_NV12:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
default:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
}
} else {
*size = ch * ht * pch;
llen = ch * ht;
}
if (file_length == *size){
if (fread(dst_addr, 1, file_length, fp) != file_length) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
} else {
for (i = 0; i < llen; i++) {
if (fread(dst_addr + i * pch,
data_sz, wd, fp) != wd) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
memset(dst_addr + (i * pch + wd * data_sz),
0, pch - wd * data_sz);
}
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static uint32_t get_file_size(const char *path)
{
struct stat statbuff;
if (stat(path, &statbuff) < 0) {
perror(path);
return 0;
} else {
return statbuff.st_size;
}
}
static uint8_t *alloc_mem_for_file(const char *file_name, uint32_t *size)
{
uint8_t *virt = NULL;
FILE *in = NULL;
uint32_t file_size = 0;
do {
file_size = get_file_size(file_name);
if (!file_size) {
break;
}
virt = (uint8_t *)malloc(file_size);
if (!virt) {
printf("Can not alloc memory size: %u\n", file_size);
break;
}
in = fopen(file_name, "rb");
if (in == NULL) {
perror(file_name);
break;
}
if (fread(virt, file_size, 1, in) != 1) {
perror(file_name);
if (virt) {
free(virt);
virt = NULL;
}
break;
}
} while (0);
if (in) {
fclose(in);
in = NULL;
}
if (size && virt) {
*size = file_size;
}
return virt;
}

int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
struct cv_mem lib_mem;
struct cv_mem in_mem;
struct cv_mem out_mem;
yuv2rgb_mat_t *yuv2rgb_mat;
vect_desc_t y, uv, rgb;
vproc_tid_attr_t pid_attr;
cv_daddr_t y_addr = 0, uv_addr = 0, rgb_addr = 0;
unsigned long out_dram_size;
uint32_t size = 0;
int curr_tid = 0;
int rval = -1;
void *p = NULL;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open(CAVALRY_DEV_NODE, O_RDWR, 0)) < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
memset(&y, 0, sizeof(y));
memset(&uv, 0, sizeof(uv));
memset(&rgb, 0, sizeof(rgb));
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&in_mem, 0, sizeof(in_mem));
memset(&out_mem, 0, sizeof(out_mem));
if (!feed_mem_flag) {
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
} else {
p = alloc_mem_for_file(bin_name, NULL);
if (!p) {
printf("alloc mem for file: %s err\n", bin_name);
rval = -1;
break;
}
if (vproc_init_from_mem(p, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
}
lib_mem.size = size;
if (cavalry_mem_alloc(&lib_mem.size, &lib_mem.phys, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
vproc_load(fd_cavalry, lib_mem.virt, lib_mem.phys, size);
if (set_log_level_flag) {
vproc_set_log_level(libvproc_log_level);
}
if (usr_nvp_affinity) {
memset(&pid_attr, 0, sizeof(pid_attr));
pid_attr.nvp_affinity = usr_nvp_affinity;
curr_tid = vproc_get_current_tid();
printf("Set tid: %d, nvp_affinity: 0x%x\n", curr_tid, pid_attr.nvp_affinity);
vproc_set_tid_attr(curr_tid, &pid_attr);
}
if (p) {
free(p);
p = NULL;
}
if (!ip) {
ip = ALIGN_PITCH(iw);
}
if (iw % 2 == 1) {
printf("image width should be even!\n");
rval = -1;
break;
}
y.shape.w = iw;
y.shape.h = ih;
y.shape.d = 1;
y.pitch = ip;
y.color_space = cs_yuv;
memcpy(&y.roi, &input_roi, sizeof(y.roi));
uv.shape.w = iw / 2;
uv.shape.h = ih / 2;
uv.shape.d = 2;
uv.pitch = ip;
uv.color_space = cs_yuv;
uv.roi.xoffset = (input_roi.xoffset + 1) / 2;
uv.roi.yoffset = (input_roi.yoffset + 1) / 2;
uv.roi.width = (input_roi.width + 1) / 2;
uv.roi.height = (input_roi.height + 1) / 2;
in_mem.size = y.shape.h * y.pitch + uv.shape.h * uv.pitch;
if (cavalry_mem_alloc(&in_mem.size, &in_mem.phys, &in_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
y_addr = in_mem.phys;
y.data_addr = y_addr;
uv_addr = y_addr + y.shape.h * y.pitch;
uv.data_addr = uv_addr;
if (load_vect_bin(input_name, in_mem.virt, &y, &size) < 0) {
printf("load input fails!\n");
rval = -1;
break;
}
if (input_roi.width > 0 && input_roi.height > 0) {
rgb.shape.w = input_roi.width;
rgb.shape.h = input_roi.height;
} else {
rgb.shape.w = iw;
rgb.shape.h = ih;
}
rgb.shape.d = 3;
rgb.pitch = ALIGN_PITCH(rgb.shape.w);
rgb.color_space = cs_rgb;
if (!matrix_format) {
yuv2rgb_mat = &yuv2rgb_mat_bt601_full;
} else {
yuv2rgb_mat = &yuv2rgb_mat_bt601_tv;
}
yuv2rgb_mat->high_accuracy_mode = high_accuracy_mode;
out_dram_size = rgb.shape.d * rgb.shape.h * rgb.pitch;
out_mem.size = out_dram_size;
if (cavalry_mem_alloc(&out_mem.size, &out_mem.phys, &out_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(out_mem.virt, 0, out_mem.size);
rgb_addr = out_mem.phys;
rgb.data_addr = rgb_addr;
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
rval = vproc_yuv2rgb_420(&y, &uv, &rgb, yuv2rgb_mat);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("yuv2rgb takes %ld us\n", (tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
if (rval < 0) {
printf("yuv2rgb failure!\n");
break;
}
if (output_flag) {
if (store_bin(output_name, out_dram_size, out_mem.virt) < 0) {
printf("store output fails!\n");
rval = -1;
break;
}
}
} while (0);
vproc_exit();
if (out_mem.virt != NULL) {
if (cavalry_mem_free(out_mem.size , out_mem.phys, out_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (in_mem.virt != NULL) {
if (cavalry_mem_free(in_mem.size , in_mem.phys, in_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (lib_mem.virt != NULL) {
if (cavalry_mem_free(lib_mem.size , lib_mem.phys, lib_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
