void usage(void)
{
unsigned long i;
printf("test_vproc_rgb2rgb_mfd usage:\n");
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);

int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, width, height;
memset(&input_roi, 0, sizeof(input_roi));
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(bin_name, sizeof(bin_name), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(input_name, sizeof(input_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 's':
get_vect_resolution(optarg, &width, &height);
iw = width;
ih = height;
break;
case 'p':
ip = atoi(optarg);
if (CHECK_PITCH_ALIGN(ip)) {
printf("Pitch [%d] is not %u-byte aligned.\n",
ip, CAVALRY_PORT_PITCH_ALIGN);
return -1;
}
break;
case 'R':
get_roi_area(optarg, &input_roi);
break;
case 'r':
cs_rgb = atoi(optarg);
if (cs_rgb == 0) {
cs_rgb = CS_RGB;
} else if (cs_rgb == 1) {
cs_rgb = CS_BGR;
}
if (cs_rgb < CS_RGB_FIRST || cs_rgb >= CS_RGB_LAST) {
printf("Invalid RGB format %d.\n", cs_rgb);
return -1;
}
break;
case 'y':
cs_yuv = atoi(optarg);
if (cs_yuv < CS_YUV_FIRST || cs_yuv >= CS_YUV_LAST) {
printf("Invalid yuv format %d.\n", cs_yuv);
return -1;
}
break;
case 't':
measure_time_flag = 1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int load_vect_bin(char *file, void *dst_addr,
vect_desc_mfd_t *vect_desc, uint32_t *size)
{
FILE *fp = NULL;
int rval = 0;
uint32_t i, data_sz;
uint32_t ch, ht, wd, pch, llen;
unsigned long file_length;
do {
if ((fp = fopen(file, "rb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fseek(fp, 0, SEEK_END) < 0) {
perror("SEEK_END");
rval = -1;
break;
}
file_length = ftell(fp);
if (fseek(fp, 0, SEEK_SET) < 0) {
perror("SEEK_SET");
rval = -1;
break;
}
ch = vect_desc->shape.d;
ht = vect_desc->shape.h;
wd = vect_desc->shape.w;
pch = vect_desc->pitch;
data_sz = (1<<vect_desc->data_format.datasize);
if (vect_desc->color_space >= CS_YUV_FIRST) {
switch(vect_desc->color_space) {
case CS_NV12:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
default:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
}
} else {
*size = ch * ht * pch;
llen = ch * ht;
}
if (file_length == *size){
if (fread(dst_addr, 1, file_length, fp) != file_length) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
} else {
for (i = 0; i < llen; i++) {
if (fread(dst_addr + i * pch,
data_sz, wd, fp) != wd) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
memset(dst_addr + (i * pch + wd * data_sz),
0, pch - wd * data_sz);
}
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}

int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
struct cv_mem lib_mem;
struct cv_mem in_mem;
struct cv_mem out_mem;
rgb2yuv_mat_t rgb2yuv_mat;
vect_desc_mfd_t y, uv, rgb;
unsigned long out_dram_size;
uint32_t size = 0;
int rval = 0;
void *p = NULL;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open(CAVALRY_DEV_NODE, O_RDWR, 0)) < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
memset(&y, 0, sizeof(y));
memset(&uv, 0, sizeof(uv));
memset(&rgb, 0, sizeof(rgb));
memset(&rgb2yuv_mat, 0, sizeof(rgb2yuv_mat));
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&in_mem, 0, sizeof(in_mem));
memset(&out_mem, 0, sizeof(out_mem));
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc_mfd(lib_mem.size, &lib_mem.mfd, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
vproc_load_mfd(fd_cavalry, lib_mem.virt, lib_mem.mfd, size);
if (p) {
free(p);
p = NULL;
}
if (!ip) {
ip = ALIGN_PITCH(iw);
}
rgb.shape.w = iw;
rgb.shape.h = ih;
rgb.shape.d = 3;
rgb.pitch = ip;
rgb.color_space = cs_rgb;
memcpy(&rgb.roi, &input_roi, sizeof(rgb.roi));
in_mem.size = rgb.shape.d * rgb.shape.h * rgb.pitch;
if (cavalry_mem_alloc_mfd(in_mem.size, &in_mem.mfd, &in_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
rgb.data_addr_fd = in_mem.mfd;
rgb.data_addr_offset = 0;
if (load_vect_bin(input_name, in_mem.virt, &rgb, &size) < 0) {
printf("load input fails!\n");
rval = -1;
break;
}
if (input_roi.width > 0 && input_roi.height > 0) {
y.shape.w = input_roi.width;
y.shape.h = input_roi.height;
} else {
y.shape.w = iw;
y.shape.h = ih;
}
if (y.shape.w % 2 == 1 || y.shape.h % 2 == 1) {
printf("image width/height should be even!\n");
rval = -1;
break;
}
uv.shape.w = y.shape.w >> 1;
uv.shape.h = y.shape.h >> 1;
y.shape.d = 1;
y.pitch = ALIGN_PITCH(y.shape.w);
uv.shape.d = 2;
uv.pitch = ALIGN_PITCH(y.shape.w);
switch (cs_yuv) {
case CS_NV12:
y.color_space = CS_Y;
uv.color_space = CS_ITL;
break;
default:
printf("Unsupported yuv format %d.\n", cs_yuv);
break;
}
out_dram_size = y.shape.h * y.pitch + uv.shape.h * uv.pitch;
out_mem.size = out_dram_size;
if (cavalry_mem_alloc_mfd(out_mem.size, &out_mem.mfd, &out_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
memset(out_mem.virt, 0, out_mem.size);
y.data_addr_fd = out_mem.mfd;
y.data_addr_offset = 0;
uv.data_addr_fd = out_mem.mfd;
uv.data_addr_offset = y.shape.h * y.pitch;
rgb2yuv_mat.matrix[0] = 0.25683594;
rgb2yuv_mat.matrix[1] = 0.50390625;
rgb2yuv_mat.matrix[2] = 0.09765625;
rgb2yuv_mat.matrix[3] = -0.14794922;
rgb2yuv_mat.matrix[4] = -0.29052734;
rgb2yuv_mat.matrix[5] = 0.43896484;
rgb2yuv_mat.matrix[6] = 0.43896484;
rgb2yuv_mat.matrix[7] = -0.36767578;
rgb2yuv_mat.matrix[8] = -0.07080078;
rgb2yuv_mat.bias[0] = 16;
rgb2yuv_mat.bias[1] = 128;
rgb2yuv_mat.bias[2] = 128;
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
rval = vproc_rgb2yuv_420_mfd(&rgb, &y, &uv, &rgb2yuv_mat);
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("rgb2yuv takes %ld us\n", (tv2.tv_sec - tv1.tv_sec) * 1000000 + (tv2.tv_usec - tv1.tv_usec));
}
if (rval < 0) {
printf("rgb2yuv failure!\n");
break;
}
if (output_flag) {
if (store_bin(output_name, out_dram_size, out_mem.virt) < 0) {
printf("store output fails!\n");
rval = -1;
break;
}
}
} while (0);
vproc_exit();
if (out_mem.virt != NULL) {
if (cavalry_mem_free_mfd(out_mem.size , out_mem.mfd, out_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (in_mem.virt != NULL) {
if (cavalry_mem_free_mfd(in_mem.size , in_mem.mfd, in_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (lib_mem.virt != NULL) {
if (cavalry_mem_free_mfd(lib_mem.size , lib_mem.mfd, lib_mem.virt) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
}
}
if (fd_cavalry >= 0) {
close(fd_cavalry);
fd_cavalry = -1;
}
return rval;
}
