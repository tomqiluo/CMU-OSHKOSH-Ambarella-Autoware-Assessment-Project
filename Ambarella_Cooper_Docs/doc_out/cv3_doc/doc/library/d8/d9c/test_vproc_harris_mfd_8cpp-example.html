int init_param(int argc, char **argv)
{
int ch;
int option_index = 0;
int value, height, width;
input_df.sign = 0;
input_df.datasize = 0;
input_df.exp_offset = 8;
input_df.exp_bits = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index))
!= -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
dag_flag = 1;
snprintf(bin_name, sizeof(bin_name), "%s", optarg);
break;
case 'l':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(left_img_input_name, sizeof(left_img_input_name), "%s", optarg);
break;
case 'r':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
snprintf(right_img_input_name, sizeof(right_img_input_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("filepath [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
output_flag = 1;
snprintf(output_name, sizeof(output_name), "%s", optarg);
break;
case 's':
get_vect_resolution(optarg, &width, &height);
iw = width;
ih = height;
break;
case 't':
harris_threshold = atof(optarg);
if (harris_threshold <= 0) {
printf("Invalid harris_threshold %f.\n", harris_threshold);
printf("allowed harris_threshold higher than zero.\n");
return -1;
}
break;
case 'n':
max_harris_num = atoi(optarg);
if (max_harris_num > MAX_MAX_HARRIS_NUM) {
printf("maximal detected harris points is %d.\n", MAX_MAX_HARRIS_NUM);
return -1;
}
break;
case 'w':
nms_win_size= atoi(optarg);
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static int load_gray_img_ufix8_8_bin(char *img_file, uint8_t *dst_addr,
vect_desc_mfd_t *vect_desc, uint32_t *size)
{
Mat rgb, gray;
uint8_t* gray_data;
rgb = imread(img_file);
resize(rgb, rgb, Size(vect_desc->shape.w, vect_desc->shape.h), INTER_AREA);
if (rgb.empty()) {
return -1;
}
cvtColor(rgb, gray, COLOR_BGR2GRAY);
gray_data = (uint8_t*)gray.data;
for (uint32_t i = 0; i < vect_desc->shape.h; i++){
for (uint32_t j = 0; j < vect_desc->shape.w; j++){
dst_addr[i * vect_desc->pitch + j] = gray_data[i * vect_desc->shape.w + j];
}
}
*size = vect_desc->pitch * vect_desc->shape.h;
return (*size);
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}

static int generate_brief_desc(uint8_t* img_data,uint8_t* harris_bitmap,int32_t im_h,
int32_t im_w,int32_t im_pitch, uint32_t* out_harris_pos,uint8_t* out_Brief_bitbytes,
int32_t max_harris_det_num)
{
int32_t i, j;
int32_t m, n, k, x;
uint8x16_t vect_data_A;
uint8x16_t vect_data_B;
uint8x16_t vect_AB_cmp_bits;
check_type *check_row_ptr;
check_type check_bits_mask = 1;
int check_step_bytes = sizeof(check_type);
int check_step_bits = check_step_bytes * 8;
int32_t harris_index;
int Harris_cnt = 0;
int32_t bitmap_pitch = ALIGN_PITCH((ALIGN_8_BITS(im_w) / 8));
int32_t bitmap_brief_pitch;
bitmap_brief_pitch = ALIGN_PITCH((ALIGN_8_BITS(im_w - BRIEF_RADIUS) / 8));
uint8_t pA[BRIEF_SIZE];
uint8_t pB[BRIEF_SIZE];
uint8_t local_AB_cmp_bits[16];
uint8_t* brief_out_byte_ptr;
uint16_t* brief_out_2byte_ptr;
uint32_t* pos_out_ptr;
uint8_t* PA_tr;
uint8_t* PB_tr;
int32_t brief_bits_radius = ceil(BRIEF_RADIUS / 8.0);
if (BRIEF_SIZE % 16 != 0) {
printf("brief size should be N times of 16\n");
return -1;
}
int32_t brief_neon_load_times = BRIEF_SIZE / 16;
int32_t brief_desc_bytes_num = BRIEF_SIZE / 8;
for (i = BRIEF_RADIUS; i < (im_h - BRIEF_RADIUS); i++) {
check_row_ptr = (check_type*)(harris_bitmap + i * bitmap_pitch);
for (j = brief_bits_radius; j < bitmap_brief_pitch; j += check_step_bytes){
if ((check_row_ptr[j]) != 0){
for (n = 0; n < check_step_bits; n++){
if (((check_row_ptr[j] >> n) & check_bits_mask)
&& ((j * 8 + n) < im_w)){
harris_index = i * im_pitch + j * 8 + n;
pos_out_ptr = out_harris_pos + Harris_cnt * 2;
pos_out_ptr[0] = i;
pos_out_ptr[1] = j * 8 + n;
brief_out_byte_ptr = out_Brief_bitbytes +
Harris_cnt * brief_desc_bytes_num;
Harris_cnt++;
for (m = 0; m < BRIEF_SIZE; m++) {
pA[m] = img_data[harris_index + brief_256A_index_offsets[m]];
pB[m] = img_data[harris_index + brief_256B_index_offsets[m]];
}
PA_tr = pA;
PB_tr = pB;
//do bit compare here
for (k = 0; k < brief_neon_load_times; k++) {
vect_data_A = vld1q_u8(PA_tr);
vect_data_B = vld1q_u8(PB_tr);
PA_tr += 16;
PB_tr += 16;
vect_AB_cmp_bits = vcltq_u8(vect_data_A, vect_data_B);
vst1q_u8(local_AB_cmp_bits, vect_AB_cmp_bits);
brief_out_2byte_ptr = (uint16_t*)(brief_out_byte_ptr + k * 2);
*brief_out_2byte_ptr = 0;
for (x = 0; x < 16; x++) {
if (local_AB_cmp_bits[x] > 0) {
*brief_out_2byte_ptr = (*brief_out_2byte_ptr) | (0x0001<<x);
}
}
}
if (Harris_cnt >= max_harris_det_num) {
printf("A:total detect %d Harris points, [%d, %d]\n", Harris_cnt, i, j);
return Harris_cnt;
}
}
}
}
}
}
return Harris_cnt;
}
