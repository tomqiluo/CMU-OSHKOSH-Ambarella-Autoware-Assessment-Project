static int load_vect_bin(char *file, void *dst_addr,
vect_desc_t *vect_desc, uint32_t *size)
{
FILE *fp = NULL;
int rval = 0;
uint32_t i, data_sz;
uint32_t ch, ht, wd, pch, llen;
unsigned long file_length;
do {
if ((fp = fopen(file, "rb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fseek(fp, 0, SEEK_END) < 0) {
perror("SEEK_END");
rval = -1;
break;
}
file_length = ftell(fp);
if (fseek(fp, 0, SEEK_SET) < 0) {
perror("SEEK_SET");
rval = -1;
break;
}
ch = vect_desc->shape.d;
ht = vect_desc->shape.h;
wd = vect_desc->shape.w;
pch = vect_desc->pitch;
data_sz = (1<<vect_desc->data_format.datasize);
if (vect_desc->color_space >= CS_YUV_FIRST) {
switch(vect_desc->color_space) {
case CS_NV12:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
case CS_Y:
*size = ht * pch;
llen = ht;
break;
default:
*size = 3 * ht / 2 * pch;
llen = 3 * ht / 2;
break;
}
} else {
*size = ch * ht * pch;
llen = ch * ht;
}
if (file_length == *size){
if (fread(dst_addr, 1, file_length, fp) != file_length) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
} else {
for (i = 0; i < llen; i++) {
if (fread(dst_addr + i * pch,
data_sz, wd, fp) != wd) {
printf("file mismatch expected size!\n");
rval = -1;
break;
}
memset(dst_addr + (i * pch + wd * data_sz),
0, pch - wd * data_sz);
}
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int store_bin(char *file, unsigned long size, void *addr)
{
FILE *fp = NULL;
int rval = 0;
do {
if ((fp = fopen(file, "wb")) == NULL) {
perror(file);
rval = -1;
break;
}
if (fwrite(addr, 1, size, fp) != size) {
perror("fwrite");
rval = -1;
break;
}
fclose(fp);
fp = NULL;
} while(0);
if (fp) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int image_deformation(vect_desc_t *src, vect_desc_t *dst)
{
int rval = 0;
vect_desc_t rgb;
struct cv_mem rgb_mem;
int is_yuv2rgb = 0, is_resize = 0, is_crop = 0;
uint32_t sw, sh;
vect_desc_t y, uv;
resize_cfg_t resize_cfg;
struct timeval tv1, tv2;
if (src->roi.width && src->roi.height) {
sw = src->roi.width;
sh = src->roi.height;
} else {
sw = src->shape.w;
sh = src->shape.h;
}
if (src->color_space == CS_NV12 && dst->color_space >= CS_RGB_FIRST
&& dst->color_space <= CS_RGB_LAST) {
is_yuv2rgb = 1;
if (sw != dst->shape.w || sh != dst->shape.h) {
is_resize = 1;
}
}
if (src->color_space == dst->color_space) {
if (sw != dst->shape.w || sh != dst->shape.h) {
is_resize = 1;
} else if (sw == dst->shape.w && sh == dst->shape.h) {
if (src->shape.w != dst->shape.w
|| src->shape.h != dst->shape.h) {
is_crop = 1;
}
}
}
memset(&rgb_mem, 0, sizeof(rgb_mem));
if (is_yuv2rgb && is_resize) {
rgb_mem.size = 3 * sh * ALIGN_PITCH(sw);
if (cavalry_mem_alloc(&rgb_mem.size, &rgb_mem.phys, &rgb_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
return -1;
}
memset(rgb_mem.virt, 0, rgb_mem.size);
}
if (measure_time_flag) {
gettimeofday(&tv1, NULL);
}
if (is_crop) {
rval = vproc_crop(src, dst);
if (rval < 0) {
printf("vproc_crop failed!\n");
}
} else if (is_yuv2rgb) {
memset(&y, 0, sizeof(y));
memset(&uv, 0, sizeof(uv));
y.shape.w = src->shape.w;
y.shape.h = src->shape.h;
y.shape.d = 1;
y.pitch = src->pitch;
y.color_space = CS_NV12;
memcpy(&y.roi, &src->roi, sizeof(y.roi));
uv.shape.w = src->shape.w / 2;
uv.shape.h = src->shape.h / 2;
uv.shape.d = 2;
uv.pitch = src->pitch;
uv.color_space = CS_NV12;
uv.roi.width = (src->roi.width + 1) / 2;
uv.roi.height = (src->roi.height + 1) / 2;
uv.roi.xoffset = (src->roi.xoffset + 1) / 2;
uv.roi.yoffset = (src->roi.yoffset + 1) / 2;
y.data_addr = src->data_addr;
uv.data_addr = src->data_addr + src->shape.h * src->pitch;
if (!is_resize) {
rval = vproc_yuv2rgb_420(&y, &uv, dst, NULL);
if (rval < 0) {
printf("vproc_yuv2rgb_420 failed!\n");
}
} else {
do {
memset(&rgb, 0, sizeof(rgb));
rgb.shape.w = sw;
rgb.shape.h = sh;
rgb.shape.d = 3;
rgb.pitch = ALIGN_PITCH(rgb.shape.w);
rgb.color_space = dst->color_space;
rgb.data_addr = rgb_mem.phys;
rval = vproc_yuv2rgb_420(&y, &uv, &rgb, NULL);
if (rval < 0) {
printf("vproc_yuv2rgb_420 failed!\n");
break;
}
memset(&resize_cfg, 0, sizeof(resize_cfg));
resize_cfg.batch_num = 1;
resize_cfg.disable_chpromo = 0;
resize_cfg.method = RESIZE_SIN_STEP;
rval = vproc_resize_ext(&rgb, dst, &resize_cfg);
if (rval < 0) {
printf("vproc_resize_ext failed!\n");
}
} while (0);
}
} else if (is_resize) {
memset(&resize_cfg, 0, sizeof(resize_cfg));
resize_cfg.batch_num = 1;
resize_cfg.disable_chpromo = 0;
resize_cfg.method = RESIZE_SIN_STEP;
rval = vproc_resize_ext(src, dst, &resize_cfg);
if (rval < 0) {
printf("vproc_resize_ext failed!\n");
}
} else {
printf("error: unknown operation\n");
rval = -1;
}
if (measure_time_flag) {
gettimeofday(&tv2, NULL);
printf("deformation takes %ld us\n", (tv2.tv_sec - tv1.tv_sec) * 1000000 +
(tv2.tv_usec - tv1.tv_usec));
}
if (rgb_mem.virt != NULL) {
if (cavalry_mem_free(rgb_mem.size , rgb_mem.phys, rgb_mem.virt) < 0) {
printf("cavalry_mem_free err\n");
rval = -1;
}
}
return rval;
}
static void sigstop()
{;}
int main(int argc, char **argv)
{
struct vproc_version ver;
struct timeval tv1, tv2;
struct cv_mem lib_mem;
struct cv_mem in_mem;
struct cv_mem mean_mem;
struct cv_mem out_mem;
polish_extra_t polish_extra;
uint32_t size = 0;
int rval = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
memset(&lib_mem, 0, sizeof(lib_mem));
memset(&in_mem, 0, sizeof(in_mem));
memset(&out_mem, 0, sizeof(out_mem));
memset(&mean_mem, 0, sizeof(mean_mem));
memset(&polish_extra, 0, sizeof(polish_extra));
if (argc < 2) {
usage();
return 0;
}
if (init_param(argc, argv) < 0) {
return -1;
}
if ((fd_cavalry = open(CAVALRY_DEV_NODE, O_RDWR, 0)) < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
do {
if (cavalry_mem_init(fd_cavalry, 0) < 0) {
printf("cavalry_mem_init err\n");
rval = -1;
break;
}
if (vproc_get_version(&ver) < 0) {
printf("vproc get version err");
rval = -1;
break;
}
printf("%s: %u.%u.%u, mod-time: 0x%x\n",
ver.description, ver.major, ver.minor, ver.patch, ver.mod_time);
if (vproc_init(bin_name, &size) < 0) {
printf("vproc_init err\n");
rval = -1;
break;
}
lib_mem.size = size;
if (cavalry_mem_alloc(&lib_mem.size, &lib_mem.phys, &lib_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
vproc_load(fd_cavalry, lib_mem.virt, lib_mem.phys, size);
if (!in.pitch) {
in.pitch = ALIGN_PITCH(in.shape.w * (1<<in.data_format.datasize));
}
if (in.pitch < in.shape.w * (1<<in.data_format.datasize)) {
printf("pitch is smaller than width!\n");
rval = -1;
break;
}
if (in.color_space == CS_NV12) {
in_mem.size = in.shape.h * in.pitch + in.shape.h / 2 * in.pitch;
} else {
in_mem.size = in.shape.d * in.shape.h * in.pitch;
}
if (cavalry_mem_alloc(&in_mem.size, &in_mem.phys, &in_mem.virt, 0) < 0) {
printf("cavalry_mem_alloc err\n");
rval = -1;
break;
}
in.data_addr = in_mem.phys;
if (load_vect_bin(data_name, in_mem.virt, &in, &size) < 0) {
printf("load data input fails!\n");
rval = -1;
break;
}
if (!mean.pitch) {
mean.pitch = ALIGN_PITCH(mean.shape.w * (1<<mean.data_format.datasize));
}
if (mean.pitch < mean.shape.w * (1<<mean.data_format.datasize)) {
printf("pitch is smaller thatn width!\n");
rval = -1;
break;
}
mean_mem.size = mean.shape.d * mean.shape.h * mean.pitch;
if (mean_mem.size) {
