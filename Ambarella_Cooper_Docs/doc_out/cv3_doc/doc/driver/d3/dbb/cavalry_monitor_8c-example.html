#define NO_ARG 0
#define HAS_ARG 1
static int running_loop = 1;
static const char *monitor_sysfile_err = CAVALRY_MONITOR_ERR_PATH;
static const char *monitor_sysfile_info = CAVALRY_MONITOR_INFO_PATH;
static const char *cavalry_core_file = NULL;
static const char *vpstatus_file = NULL;
static struct option long_options[] = {
{"err_sysfs", HAS_ARG, 0, 'e'},
{"info_sysfs", HAS_ARG, 0, 'f'},
{"cavalry_core_dump", HAS_ARG, 0, 'c'},
{"vpstatus_dump", HAS_ARG, 0, 'v'},
{"help", NO_ARG, 0, 'h'},
{0, 0, 0, 0}
};
static const char *short_options = "s:c:v:h";
struct hint_s {
const char *arg;
const char *str;
};
static const struct hint_s hint[] = {
{"", "\t\tThe err sysfs of cavalry monitor, default path: " CAVALRY_MONITOR_ERR_PATH},
{"", "\t\tThe info sysfs of cavalry monitor, default path: " CAVALRY_MONITOR_INFO_PATH},
{"", "\tThe filename to dump cavalry_core"},
{"", "\tThe filename to dump vpstatus"},
{"", "\t\tHelp info"},
};
static void usage(void)
{
char *itself = "cavalry_monitor";
unsigned long i = 0;
printf("%s usage:\n", itself);
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
printf("\nExample:\n\n");
printf(" # %s -c /tmp/cavalry_core_dump.bin -v /tmp/vpstatus_dump.bin\n\n", itself);
}
static int init_param(int argc, char **argv)
{
int ch = 0;
int option_index = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'e':
monitor_sysfile_err = optarg;
break;
case 'f':
monitor_sysfile_info = optarg;
break;
case 'c':
cavalry_core_file = optarg;
break;
case 'v':
vpstatus_file = optarg;
break;
case 'h':
usage();
return -1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static void dump_vp_context(void)
{
#define MAX_SIZE (512)
char cmd[MAX_SIZE];
char buf[MAX_SIZE];
int ret = 0;
if (cavalry_core_file) {
memset(cmd, 0, sizeof(cmd));
memset(buf, 0, sizeof(buf));
snprintf(cmd, sizeof(cmd), "cavalry_core_dump -d %s\n", cavalry_core_file);
ret = system(cmd);
if (WIFSIGNALED(ret) &&
(WTERMSIG(ret) == SIGINT || WTERMSIG(ret) == SIGQUIT)) {
printf("Execute cmd: %s failed\n", cmd);
}
} else {
printf("No cavalry_core dump\n");
}
if (vpstatus_file) {
memset(cmd, 0, sizeof(cmd));
memset(buf, 0, sizeof(buf));
snprintf(cmd, sizeof(cmd), "vpstatus -d %s\n", vpstatus_file);
ret = system(cmd);
if (WIFSIGNALED(ret) &&
(WTERMSIG(ret) == SIGINT || WTERMSIG(ret) == SIGQUIT)) {
printf("Execute cmd: %s failed\n", cmd);
}
} else {
printf("No vpstatus dump\n");
}
}
static int process_msg_err(struct cavalry_monitor_msg_err *msg)
{
struct tm t;
time_t tv_sec = 0;
uint32_t ms = 0, us = 0;
char time_string[64];
int rval = 0;
tv_sec = msg->tv_sec;
ms = msg->tv_usec / 1000;
us = msg->tv_usec - ms * 1000;
strftime(time_string, sizeof(time_string), "%Y-%m-%d %H:%M:%S",
localtime_r(&tv_sec, &t));
printf("[%s %u.%u] Receive ERR: %s, at func: %s:(%d)\n", time_string, ms, us,
msg->msg_string, msg->func_name, msg->line_num);
if (msg->err_code == MONITOR_ERR_VP_HANG) {
dump_vp_context();
rval = -1;
}
return rval;
}

{
struct tm t;
time_t tv_sec = 0;
uint32_t ms = 0, us = 0;
char time_string[64];
int rval = 0;
tv_sec = msg->tv_sec;
ms = msg->tv_usec / 1000;
us = msg->tv_usec - ms * 1000;
strftime(time_string, sizeof(time_string), "%Y-%m-%d %H:%M:%S",

printf("[%s %u.%u] Receive INFO: %s, at func: %s:(%d)\n", time_string, ms, us,
msg->msg_string, msg->func_name, msg->line_num);
return rval;
}
static void sigstop()
{
running_loop = 0;
}
int main(int argc, char **argv)
{
#define MAX_ID (3)
struct cavalry_monitor_msg_err err;
struct cavalry_monitor_msg_info info;
struct pollfd fds[MAX_ID];
int fd_sysfs_err = -1;
int fd_sysfs_info = -1;
int rsize = 0;
int rval = 0, ret = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
#if 0
if (argc < 2) {
usage();
return -1;
}
#endif
if (init_param(argc, argv) < 0) {
return -1;
}
if (!monitor_sysfile_err) {
printf("Please provide cavalry monitor err sysfile\n");
return -1;
}
if (!monitor_sysfile_info) {
printf("Please provide cavalry monitor info sysfile\n");
return -1;
}
do {
fd_sysfs_err = open(monitor_sysfile_err, O_RDONLY, 0644);
if (fd_sysfs_err < 0) {
perror(monitor_sysfile_err);
rval = -1;
break;
}
fd_sysfs_info = open(monitor_sysfile_info, O_RDONLY, 0644);
if (fd_sysfs_info < 0) {
perror(monitor_sysfile_info);
rval = -1;
break;
}
do {
rsize = read(fd_sysfs_info, (char *)&info, sizeof(info));
if (rsize < 0) {
perror("read");
rval = -1;
break;
} else if (rsize > 0) {
if (process_msg_info(&info) < 0) {
rval = -1;
break;
}
} else {
printf("No INFO MSG\n");
}
ret = lseek(fd_sysfs_info, 0, SEEK_SET);
if (ret < 0) {
perror("seek");
rval = -1;
break;
}
} while (rsize);
if (rval < 0) {
break;
}
do {
rsize = read(fd_sysfs_err, (char *)&err, sizeof(err));
if (rsize < 0) {
perror("read");
rval = -1;
break;
} else if (rsize > 0) {
if (process_msg_err(&err) < 0) {
rval = -1;
break;
}
} else {
printf("No ERR MSG\n");
}
ret = lseek(fd_sysfs_err, 0, SEEK_SET);
if (ret < 0) {
perror("seek");
rval = -1;
break;
}
} while (rsize);
if (rval < 0) {
break;
}
memset((void *)&fds, 0, sizeof(fds));
fds[0].fd = fd_sysfs_err;
fds[0].events = POLLPRI;
fds[0].revents = 0;
fds[1].fd = fd_sysfs_info;
fds[1].events = POLLPRI;
fds[1].revents = 0;
while (running_loop) {
ret = poll(fds, MAX_ID, -1);
if (ret <= 0) {
perror("poll");
rval = -1;
break;
}
if (fds[0].revents & POLLPRI) {
printf("Polled ERR\n");
ret = lseek(fd_sysfs_err, 0, SEEK_SET);
if (ret < 0) {
perror("seek");
rval = -1;
break;
}
rsize = read(fd_sysfs_err, (char *)&err, sizeof(err));
if (rsize < 0) {
perror("read");
rval = -1;
break;
} else if (rsize > 0) {
if (process_msg_err(&err) < 0) {
break;
}
} else {
printf("No ERR MSG\n");
}
}
if (fds[1].revents & POLLPRI) {
printf("Polled INFO\n");
ret = lseek(fd_sysfs_info, 0, SEEK_SET);
if (ret < 0) {
perror("seek");
rval = -1;
break;
}
rsize = read(fd_sysfs_info, (char *)&info, sizeof(info));
if (rsize < 0) {
perror("read");
rval = -1;
break;
} else if (rsize > 0) {
if (process_msg_info(&info) < 0) {
break;
}
} else {
printf("No INFO MSG\n");
}
}
}
} while (0);
if (fd_sysfs_info >= 0) {
close(fd_sysfs_info);
fd_sysfs_info = -1;
}
if (fd_sysfs_err >= 0) {
close(fd_sysfs_err);
fd_sysfs_err = -1;
}
return rval;
}
