157 u16 version;
158 union {
159 struct {
160 /* 0:DSI, 1:CSI */
161 unsigned int mipi_mode : 1;
162 /*
163 * 2'd0 : Select port A as input
164 * 2'd1 : Select port B as input
165 * 2'd2 : Select port A and port B as input
166 * 2'd3 : Autocheck the input mipi port
167 */
168 unsigned int port_select : 2;
169 /* 0:disable, 1:enable */
170 unsigned int cec_enable : 1;
171 /*
172 * 2'd0 = 4 lane
173 * 2'd1 = 1 lane
174 * 2'd2 = 2 lane
175 * 2'd3 = 3 lane
176 */
177 unsigned int lane_num : 2;
178 /*
179 * 2'd0 : HDCP disable
180 * 2'd1 : HDCP1.4 enable
181 * 2'd2 : HDCP2.2 enable
182 * 2'd3 : HDCP enable, autocheck the HDCP version
183 */
184 unsigned int hdcp_enable : 2;
185 } reg;
186 u8 val;
187 } mode;
188};
