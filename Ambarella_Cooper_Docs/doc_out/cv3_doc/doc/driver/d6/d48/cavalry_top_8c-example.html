static void usage(void)
{
char *itself = "cavalry_top";
unsigned long i = 0;
printf("%s usage:\n", itself);
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
printf("\nExample:\n\n");
printf(" # %s \n", itself);
printf(" # %s -u 100 -c 10000\n\n", itself);
}
static int init_param(int argc, char **argv)
{
int ch = 0;
int option_index = 0;
int value = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'u':
value = atoi(optarg);
if (value <= 0) {
printf("sample peroid should be > 0 .\n");
return -1;
}
sample_usecs = value;
break;
case 'c':
value = atoi(optarg);
if (value <= 0) {
printf("sample count should be > 0 .\n");
return -1;
}
sample_cnt = value;
break;
case 'b':
is_batch_mode = 1;
break;
case 'n':
value = atoi(optarg);
if (value <= 0) {
printf("num of iterations should be > 0 .\n");
return -1;
}
num_iterations = value;
break;
case 'h':
usage();
return -1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static const char *name_of_hw_type(uint32_t hw_type)
{
const char *str;
switch (hw_type) {
case CAVALRY_HW_TYPE_NVP:
str = "NVP";
break;
case CAVALRY_HW_TYPE_GVP:
str = "GVP";
break;
case CAVALRY_HW_TYPE_FEX:
str = "FEX";
break;
case CAVALRY_HW_TYPE_FMA:
str = "FMA";
break;
default:
str = "Unknown_HW";
break;
}
return str;
}
static void show_top_head(void)
{
char str_head[2048] = " PID NID";
char str_id[16];
const char *str_nvp = "NVP";
const char *str_gvp = "GVP";
const char *str_fex = "FEX";
const char *str_fma = "FMA";
const char *str_command = "COMMAND";
int sub_id = 0;
for (sub_id = 0; sub_id < CAVALRY_NVP_CNT; sub_id++) {
snprintf(str_id, sizeof(str_id), "%3s%4s%d",
SPACE_DEFAULT, str_nvp, sub_id);
strcat(str_head, str_id);
}
for (sub_id = 0; sub_id < CAVALRY_GVP_CNT; sub_id++) {
snprintf(str_id, sizeof(str_id), "%3s%4s%d",
SPACE_DEFAULT, str_gvp, sub_id);
strcat(str_head, str_id);
}
for (sub_id = 0; sub_id < CAVALRY_FEX_CNT; sub_id++) {
snprintf(str_id, sizeof(str_id), "%3s%4s%d",
SPACE_DEFAULT, str_fex, sub_id);
strcat(str_head, str_id);
}
for (sub_id = 0; sub_id < CAVALRY_FMA_CNT; sub_id++) {
snprintf(str_id, sizeof(str_id), "%3s%4s%d",
SPACE_DEFAULT, str_fma, sub_id);
strcat(str_head, str_id);
}
snprintf(str_id, sizeof(str_id), "%s%s", SPACE_DEFAULT, str_command);
strcat(str_head, str_id);
printf("-------------\n");
printf("%s\n", str_head);
}
static void sigstop()
{
exit(0);

int main(int argc, char **argv)
{
struct cavalry_stats_all all_hw;
struct stats_info *stats_info;
struct hw_stats_summary sumry[CV_HW_TOTAL_NUM];
uint32_t i = 0, j = 0, mix_id = 0, len = 0;
uint32_t pid_index = 0, nid_index = 0;
FILE *fp;
char proc_path[MAX_BUF_SIZE] = {0};
char buf[MAX_BUF_SIZE] = {0};
char data;
int rval = 0;
uint32_t arm_cluster_id = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (init_param(argc, argv) < 0) {
return -1;
}
fd_cav = open(CAVALRY_DEV_NODE, O_RDWR, 0);
if (fd_cav < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
init_vproc_op_name();
memset(&all_hw, 0, sizeof(all_hw));
stats_info = (struct stats_info *)calloc(MAX_THREAD, sizeof(struct stats_info));
if (!is_batch_mode) {
// Clear screen
printf("\e[1;1H\e[2J");
setvbuf(stdout, NULL, _IOFBF, 0);
}
if (ioctl(fd_cav, CAVALRY_GET_ARM_CLUSTER_ID, &arm_cluster_id) < 0) {
perror("CAVALRY_GET_ARM_CLUSTER_ID");
return -1;
}
while (1) {
memset(&sumry, 0, sizeof(sumry));
for (i = 0; i < sample_cnt; i++) {
if (ioctl(fd_cav, CAVALRY_GET_STATS, &all_hw) < 0) {
perror("CAVALRY_GET_STATS");
return -1;
}
for (mix_id = 0; mix_id < CV_HW_TOTAL_NUM; mix_id++) {
sumry[mix_id].hw_type = all_hw.stats_get[mix_id].hw_type;
sumry[mix_id].hw_id = all_hw.stats_get[mix_id].hw_id;
if (all_hw.stats_get[mix_id].is_hw_idle) {
sumry[mix_id].hw_idle_cnt++;
} else {
sumry[mix_id].hw_busy_cnt++;
if (all_hw.stats_get[mix_id].submitter_hw_id == arm_cluster_id) {
pid_index = (all_hw.stats_get[mix_id].current_pid % MAX_THREAD);
if (all_hw.stats_get[mix_id].current_nid >= 0 &&
all_hw.stats_get[mix_id].current_nid < CAVALRY_VPROC_NID) {
nid_index = all_hw.stats_get[mix_id].current_nid % MAX_NN_PER_THREAD;
} else if (all_hw.stats_get[mix_id].current_nid >= CAVALRY_VPROC_NID &&
all_hw.stats_get[mix_id].current_nid < CAVALRY_VPROC_NID_END) {
nid_index = MAX_NN_PER_THREAD + all_hw.stats_get[mix_id].current_nid % MAX_NN_PER_THREAD;
} else {
continue;
}
stats_info[pid_index].pid = all_hw.stats_get[mix_id].current_pid;
stats_info[pid_index].nn_info[nid_index].nid = all_hw.stats_get[mix_id].current_nid;
stats_info[pid_index].nn_info[nid_index].has_busy = 1;
stats_info[pid_index].nn_info[nid_index].busy_cnt[mix_id]++;
}
}
}
usleep(sample_usecs);
}
for (mix_id = 0; mix_id < CV_HW_TOTAL_NUM; mix_id++) {
sumry[mix_id].hw_total_cnt = sumry[mix_id].hw_idle_cnt + sumry[mix_id].hw_busy_cnt;
}
if (!is_batch_mode) {
// Clear screen
printf("\033" "[H" "\033""[J");
fflush(stdout);
}
show_top_head();
for (i = 0; i < MAX_THREAD; i++) {
if (stats_info[i].pid != 0) {
snprintf(proc_path, MAX_BUF_SIZE, "/proc/%d/cmdline", stats_info[i].pid);
if ((fp = fopen(proc_path, "rb")) == NULL) {
// Thread exit, we don't show it.
continue;
}
for (len = 0; len < MAX_BUF_SIZE; len++) {
if (fscanf(fp, "%c", &data) == EOF) {
break;
}
if (data == '\0') {
data = ' ';
}
buf[len] = data;
}
buf[MAX_BUF_SIZE - 1] = '\0';
if (fp) {
fclose(fp);
fp = NULL;
}
if (!len) {
continue;
}
for (j = 0; j < 2 * MAX_NN_PER_THREAD; j++) {
if ((stats_info[i].nn_info[j].nid >= 0) && (stats_info[i].nn_info[j].has_busy)) {
/* one row */
if (stats_info[i].nn_info[j].nid >= CAVALRY_VPROC_NID) {
printf("%5d%s%5s%s", stats_info[i].pid, SPACE_DEFAULT,
CVT_TO_OP_NAME(stats_info[i].nn_info[j].nid), SPACE_DEFAULT);
} else {
printf("%5d%s%5d%s", stats_info[i].pid, SPACE_DEFAULT,
stats_info[i].nn_info[j].nid, SPACE_DEFAULT);
}
for (mix_id = 0; mix_id < CV_HW_TOTAL_NUM; mix_id++) {
printf("%05.1f%s",
stats_info[i].nn_info[j].busy_cnt[mix_id] * 100.0 / sumry[mix_id].hw_total_cnt,
SPACE_DEFAULT);
}
printf("%s \n", buf);
}
}
memset(buf, 0, MAX_BUF_SIZE);
}
}
/* AISP: pid is 0 */
for (j = 0; j < 2 * MAX_NN_PER_THREAD; j++) {
if ((stats_info[0].nn_info[j].nid >= 0) && (stats_info[0].nn_info[j].has_busy)) {
printf("%5s%s%5d%s", "IDSP", SPACE_DEFAULT, stats_info[0].nn_info[j].nid, SPACE_DEFAULT);
for (mix_id = 0; mix_id < CV_HW_TOTAL_NUM; mix_id++) {
printf("%05.1f%s",
stats_info[0].nn_info[j].busy_cnt[mix_id] * 100.0 / sumry[mix_id].hw_total_cnt,
SPACE_DEFAULT);
}
printf("--\n");
}
}
/* Summary all hw */
printf("-------------\n");
for (mix_id = 0; mix_id < CV_HW_TOTAL_NUM; mix_id++) {
printf("%02u: %s%u IDLE %05.1f%%\n", mix_id,
name_of_hw_type(sumry[mix_id].hw_type), sumry[mix_id].hw_id,
sumry[mix_id].hw_idle_cnt * 100.0 / sumry[mix_id].hw_total_cnt);
}
fflush(stdout);
memset((void *)stats_info, 0, MAX_THREAD * sizeof(struct stats_info));
if (is_batch_mode && num_iterations != INF_ITERA_NUM && --num_iterations == 0) {
break;
}
}
if (stats_info) {
free(stats_info);
stats_info = NULL;
}
return rval;
}
