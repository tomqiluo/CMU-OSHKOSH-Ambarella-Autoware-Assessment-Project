60/*
61 * Take amba_addr_t as a glue layer between dsp_addr and phys_addr.
62 * 1.When iav enc modules try to download addresses of configurations, IK etc to dsp,
63 * amba_addr_t become the glue layer between iav enc modules and dsp cmd download layer
64 * 2.Used by buffer_cap queue buffer which aims at stitching producer(irq message[dsp_addr])
65 * and consumer(buffer query[phys_addr]).
66 *
67 * +-----------------------+ +-----------------------------+
68 * | iav enc modules | | dequeue buffer |
69 * | (phys_addr) | | (comsumer phys_addr) |
70 * +-----------------------+ +-----------------------------+
71 * | ^
72 * v |
73 * +-----------------------+ +-----------------------------+
74 * | address_glue_layer | | address_glue_layer |
75 * | (amba_addr_t) | | (queue buffer amba_addr_t) |
76 * +-----------------------+ +-----------------------------+
77 * | ^
78 * v |
79 * +-----------------------+ +-----------------------------+
80 * |dsp cmd download layer | | dsp irq msg procedure |
81 * | (dsp_addr) | | (dsp_addr) |
82 * +-----------------------+ +-----------------------------+
83 * Figure 1 Figure 2
84 */
85typedef union {
86 phys_addr_t phys_addr;
87 u32 dsp_addr;
88} amba_addr_t;

91/* When the total DRAM size >= 8GB, DSP memory 0 ~ 4GB is remapped to Physical space 4GB ~ 8GB.
92 * Else, the DSP memory addr is equal to Physical memory addr.
93 */
94
95/*
96 * +----------------------------------------------------------------+
97 * | |
98 * | +--------------------------------------------------+ |
99 * | | | |
100 * | +---------------------->IDSP_RAM_START=4GB | |
101 * | | +---->4GB+IDSP_PRIVATE_SIZE_MB | |
102 * V V | | |
103 * +--------------------------------+--------+-----------------+---------------------------+ | |
104 * | | | | | | |
105 * +--------------------------------+--------+-----------------+---------------------------+ | |
106 * ^ ^ | |
107 * | | | |
108 * +---------------------------------------------+ | | |
109 * | +--------------------------------------+ | |
110 * | | | |
111 * | Mapped by ATT | | |
112 * | | | |
113 * +----------------------- +------------+--------+ | |
114 * | | | | DSP Memory layout(0~-4GB) | |
115 * +----------------------- +------------+--------+ | |
116 * |<-IDSP_PRIVATE_SIZE_MB->| | | | |
117 * | | | | | |
118 * | | | | | |
119 * | | | | | |
120 * | | | +-------->IDSP_SHARED_END(4GB)--------------------+ |
121 * | | | Mapped by ATT |
122 * | | +----------------->IDSP_SHARED_START(4GB-IDSP_SHARE_SIZE_MB)----+
123 * | |
124 * | +------------------------------> IDSP_PRIVATE_END(0MB + IDSP_PRIVATE_SIZE_MB)
125 * |
126 * +------------------------------------------------------->IDSP_PRIVATE_START(0MB)
127 *
128 *
129 * Due to the 32bit bus width, DSP and GDMA of archv6 can only access 0~4GB memory size. Here we use ATT to mmap
130 * physical memory range[IDSP_RAM_START(4GB), IDSP_RAM_START(4GB) + IDSP_PRIVATE_SIZE_MB] to IDSP private memory
131 * [IDSP_PRIVATE_START(0) ~ IDSP_PRIVATE_END(0+IDSP_PRIVATE_SIZE_MB)].On the contrary, We use ATT to mmap physical memory
132 * range[IDSP_SHARED_START, IDSP_SHARED_END] directly to the same private dsp memory area[IDSP_SHARED_START, IDSP_SHARED_END]
133 * So when DSP report a private IDSP_PRIVATE AREA address which is always smaller than IDSP_SHARED_START, converting DSP
134 * private address to physical memory become a must step.
135 */
136/*
137 * Physical memory
138 * +-----------------+-------+---------------+
139 * | | | |
140 * +-----------------+-------+---------------+
141 * | | | |
142 * | | | |
143 * | | | |
144 * | | | +---------------------------------------------------------------->DRAM_END
145 * | | | | ^
146 * | | | +----------------->IDSP_PRIVATE_END |
147 * | | | ^ |
148 * | | | | |
149 * | | | IDSP_PRIVATE_SIZE_MB |
150 * | | | | |
151 * | | | V |
152 * | | +-------------------------------> IDSP_PRIVATE_START DRAM_SIZE <= 4GB
153 * | | | |
154 * | | +------------------------------->IDSP_SHARED_END |
155 * | | ^ |
156 * | | | |
157 * | | IDSP_SHARED_SIZE_MB |
158 * | | | |
159 * | | v |
160 * | +--------------------------------------->IDSP_SHARED_START V
161 * +--------------------------------------------------------------------------------------------------------->DRAM_START
162 *
163 */
164#define DSP_TO_PHYS amba_dsp_to_phys

195#define wake_event_after_signal(event, ms_time_out) \
196({ \
197 int rval = -1, loop; \
198 iav_debug("Wake up by the signal.\n"); \
199 loop = ms_time_out / HALF_SECOND_IN_MS; \
200 do { \
201 if (!loop) { \
202 break; \
203 } \
204 msleep(HALF_SECOND_IN_MS); \
205 if (event) { \
206 rval = 1; \
207 break; \
208 } \
209 } while (--loop); \
210 if (rval < 0) { \
211 iav_error("Failed to switch condition!\n"); \
212 } else { \
213 iav_debug("Successfully switch condition!\n"); \
214 } \
215 rval; \
216})

250 break; \

252 _dsp_work_state = DSP_WORK_STATE_PREVIEW; \
253 break; \
254 case IAV_STATE_ENCODING: \
255 _dsp_work_state = DSP_WORK_STATE_ENCODING; \
256 break; \
257 case IAV_STATE_DECODING: \
