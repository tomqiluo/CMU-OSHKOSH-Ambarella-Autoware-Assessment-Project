57/* MMB attributes */
58typedef enum {
59 AMBCMA_MMB_USR_CANNOT_FREE = BIT(0), /* cannot be freed by user space */
60 AMBCMA_MMB_USR_CANNOT_MAP = BIT(1), /* cannot be mapped by user space */
61 AMBCMA_MMB_CANNOT_IN_PREALLOC = BIT(2), /* cannot be merged in IAV_PART_PREALLOC */
62 AMBCMA_MMB_CACHE_ENABLED = BIT(3), /* enable cache */
63 AMBCMA_MMB_BIND_DMA_BUF_OBJ = BIT(4), /* bind dma-buf:fd */
64 AMBCMA_MMB_FREE_BY_DMA_BUF = BIT(5), /* be freed automatically when dma-buf:fd is closed */
65 AMBCMA_MMB_ACCESS_BY_ADV = BIT(6), /* accessible by ARM, DSP and VP */
66 AMBCMA_MMB_ACCESS_BY_AD = BIT(7), /* accessible by ARM and DSP */
67 AMBCMA_MMB_DIRECT_IO = BIT(8), /* accessible by ARM and DSP, access as direct io region */
68} AMBCMA_MMB_ATTRIBUTE;
