#define FILENAME_LENGTH (256)
#define BUFFERSIZE (1 << 20)
#define MAX_SLOT_NUM (32)

int fd_slot;

};

uint32_t log_level_flag : 1;

char output_text_name[FILENAME_LENGTH];

};

{"live", HAS_ARG, 0, 'l'},

static void usage(void)
{
char *itself = "cavalry_log";
unsigned long i = 0;
printf("%s usage:\n", itself);
for (i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val))
printf("-%c ", long_options[i].val);
else
printf(" ");
printf("--%s", long_options[i].name);
if (hint[i].arg[0] != 0)
printf(" [%s]", hint[i].arg);
printf("\t%s\n", hint[i].str);
}
printf("\nExample:\n\n");
printf("1. Capture log and show in live mode:\n");
printf(" # %s -u /lib/firmware/cavalry.bin -l\n\n", itself);
printf("2. Capture log to binary file:\n");
printf(" # %s -b /tmp/log.bin\n\n", itself);
printf("3. Parse log from binary file to txt file:\n");
printf(" # %s -u /lib/firmware/cavalry.bin -i /tmp/log.bin -o /tmp/log.txt\n", itself);
printf("4. Capture log if have hotlink, show in live mode:\n");
printf(" # %s -u /lib/firmware/cavalry.bin -e /lib/firmware/hotlink/hl.bin -l\n", itself);
}
static int init_param(int argc, char **argv, struct log_ctx *pctx)
{
int ch = 0;
int option_index = 0;
int value = 0, idx = 0;
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'b':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
pctx->capture_bin_flag = 1;
snprintf(pctx->capture_bin_name, sizeof(pctx->capture_bin_name), "%s", optarg);
break;
case 'i':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
pctx->input_bin_flag = 1;
snprintf(pctx->input_bin_name, sizeof(pctx->input_bin_name), "%s", optarg);
break;
case 'o':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
pctx->output_text_flag = 1;
snprintf(pctx->output_text_name, sizeof(pctx->output_text_name), "%s", optarg);
break;
case 'u':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
pctx->ucode_flag = 1;
snprintf(pctx->ucode_name, sizeof(pctx->ucode_name), "%s", optarg);
break;
case 'e':
value = strlen(optarg);
if (value >= FILENAME_LENGTH) {
printf("Filename [%s] is too long [%d] (>%d).\n", optarg,
value, FILENAME_LENGTH);
return -1;
}
idx = pctx->slot_num;
if (idx >= MAX_SLOT_NUM) {
printf("Too much hotlink file: %u >= %u\n", idx, MAX_SLOT_NUM);
return -1;
}
snprintf(pctx->slot[idx].file_name, sizeof(pctx->slot[idx].file_name), "%s", optarg);
pctx->slot_num++;
break;
case 'l':
if (!pctx->ucode_flag) {
printf("Ucode path must be set in live mode.\n");
return -1;
}
pctx->live_mode = 1;
break;
case 'd':
value = atoi(optarg);
if ((value < CAVALRY_LOG_LEVEL_FIRST) ||
(value > CAVALRY_LOG_LEVEL_LAST)) {
printf("Log level [%d] should be within [%u-%u].\n",
value, CAVALRY_LOG_LEVEL_FIRST, CAVALRY_LOG_LEVEL_LAST);;
return -1;
}
pctx->log_level = value;
pctx->log_level_flag = 1;
break;
default:
printf("unknown option found: %c\n", ch);
return -1;
}
}
return 0;
}
static long get_file_size(const char *path)
{
struct stat statbuff;
if (stat(path, &statbuff) < 0) {
return 0;
} else {
return statbuff.st_size;
}
}
static int get_hotlink_slot_id(const char *file_name, uint8_t *slot_id)
{
FILE *in = NULL;
uint32_t id = 0;
int rval = 0;
do {
if ((in = fopen(file_name, "rb")) == NULL) {
perror(file_name);
rval = -1;
break;
}
if (fseek(in, CAVALRY_SLOT_ID_OFFSET, SEEK_SET) < 0) {
perror("SEEK_SET");
rval = -1;
break;
}
if (fread(&id, 1, sizeof(uint32_t), in) != sizeof(uint32_t)) {
perror("fread input file");
rval = -1;
break;
}
} while (0);
if (in) {
fclose(in);
in = NULL;
}
if (!rval) {
if (id >= MAX_SLOT_NUM) {
printf("Invalid slot_id: %u >= %u\n", id, MAX_SLOT_NUM);
rval = -1;
} else {
printf("Hotlink file: %s, slot_id: %u\n", file_name, id);
*slot_id = (uint8_t)id;
}
}
return rval;
}
static int check_hotlink_slot(struct log_ctx *pctx)
{
struct cavalry_hotlink_slot_cfg slot_cfg = {0};
uint32_t slot_map = 0, found = 0;
uint32_t id = 0, j = 0;
int rval = 0;
if (ioctl(pctx->fd_cav, CAVALRY_QUERY_HOTLINK_SLOT_BUF_CFG, &slot_cfg) < 0) {
perror("CAVALRY_QUERY_HOTLINK_SLOT_BUF_CFG");
return -1;
}
if (ioctl(pctx->fd_cav, CAVALRY_GET_HOTLINK_SLOT_CFG, &slot_map) < 0) {
perror("CAVALRY_GET_HOTLINK_SLOT_CFG");
return -1;
}
for (id = 0; id < slot_cfg.slot_num; id++) {
if (slot_map & (1 << id)) {
found = 0;
for (j = 0; j < pctx->slot_num; j++) {
if (pctx->slot[j].slot_id == id) {
found = 1;
break;
}
}
if (!found) {
printf("Hotlink slot [%u] is actived, please input slot ucode\n", id);
rval = -1;
break;
}
}
}
return rval;
}
static int map_ucode(struct log_ctx *pctx)
{
void *virt = NULL;
uint32_t fsize = 0;
int rval = 0;
pctx->fd_ucode = open(pctx->ucode_name, O_RDONLY, 0);
if (pctx->fd_ucode < 0) {
perror("open");
printf("open %s err\n", pctx->ucode_name);
return -1;
}
fsize = get_file_size(pctx->ucode_name);
virt = mmap(NULL, fsize, PROT_READ, MAP_SHARED, pctx->fd_ucode, 0);
if (virt == MAP_FAILED) {
perror("mmap\n");
rval = -1;
} else {
pctx->ucode_addr = virt;
pctx->ucode_size = fsize;
}
return rval;
}
static int map_hotlink(struct log_ctx *pctx)
{
void *virt = NULL;
uint32_t fsize = 0;
uint32_t i = 0;
int fd_slot = 0;
int rval = 0;
for (i = 0; i < pctx->slot_num; i++) {
if (get_hotlink_slot_id(pctx->slot[i].file_name, &pctx->slot[i].slot_id) < 0) {
printf("get hotlink slot id err\n");
rval = -1;
break;
}
}
if (check_hotlink_slot(pctx) < 0) {
printf("check hotlink err\n");
return -1;
}
for (i = 0; i < pctx->slot_num; i++) {
fd_slot = open(pctx->slot[i].file_name, O_RDONLY, 0);
if (fd_slot < 0) {
perror("open");
printf("open %s err\n", pctx->slot[i].file_name);
rval = -1;
break;
}
pctx->slot[i].fd_slot = fd_slot;
fsize = get_file_size(pctx->slot[i].file_name);
virt = mmap(NULL, fsize, PROT_READ, MAP_SHARED, fd_slot, 0);
if (virt == MAP_FAILED) {
perror("mmap\n");
rval = -1;
break;
} else {
pctx->slot[i].virt = virt;
pctx->slot[i].size = fsize;
}
}
return rval;
}
static void ucode_ver_to_str(struct version_info_s *ver, char *ver_str, uint32_t size)
{
const char *name = NULL;
switch (ver->chip) {
case CHIP_TYPE_CV22:
name = "CV22";
break;
case CHIP_TYPE_CV2:
name = "CV2";
break;
case CHIP_TYPE_CV25:
name = "CV25";
break;
case CHIP_TYPE_CV28:
name = "CV28";
break;
case CHIP_TYPE_CV5:
name = "CV5";
break;
case CHIP_TYPE_CV52:
name = "CV52";
break;
case CHIP_TYPE_CV3:
name = "CV3";
break;
case CHIP_TYPE_CV3AD685:
name = "CV3AD685";
break;
case CHIP_TYPE_CV72:
name = "CV72";
break;
default:
name = "Unknown";
break;
}
snprintf(ver_str, size, "%s-Ver.%u-%08u-%08x-hl_fw.%08u\n",
name, ver->ucode_version, ver->build_date, ver->git_hash, ver->hotlink_fw_version);
}
static void get_ucode_ver(void *addr, struct version_info_s *ver)
{
char *base = (char *)(addr + VERSION_INFO_OFFSET);
if (ver == NULL) {
printf("Param is NULL\n");
} else {
memcpy(ver, base, sizeof(struct version_info_s));
}
}
static int get_curr_ucode_ver(int fd_cav, struct version_info_s *ver)
{
struct cavalry_querybuf querybuf;
void *virt = NULL;
int rval = 0;
memset(&querybuf, 0, sizeof(querybuf));
querybuf.buf = CAVALRY_MEM_UCODE;
if (ioctl(fd_cav, CAVALRY_QUERY_BUF, &querybuf) < 0) {
perror("CAVALRY_QUERY_BUF");
return -1;
}
virt = mmap(NULL, querybuf.length, PROT_READ, MAP_SHARED, fd_cav,
querybuf.offset);
if (virt == MAP_FAILED) {
perror("mmap\n");
rval = -1;
} else {
get_ucode_ver(virt, ver);
if (munmap(virt, querybuf.length) < 0) {
perror("munmap mem ucode");
rval = -1;
}
virt = NULL;
}
return rval;
}
static int capture(struct log_ctx *pctx)
{
struct version_info_s ver;
char ver_str[FILENAME_LENGTH] = {0};
struct cavalry_log_entry *entry = NULL;
uint8_t *buffer = NULL;
FILE *fp = NULL;
ssize_t size = 0;
unsigned long i = 0;
uint32_t j = 0;
int found = 0, rval = 0;
if (!pctx->live_mode) {
fp = fopen(pctx->capture_bin_name, "wb");
if (fp == NULL) {
perror("fopen");
printf("fopen %s err\n", pctx->capture_bin_name);
return -1;
}
}
do {
if (get_curr_ucode_ver(pctx->fd_cav, &ver) < 0) {
printf("get curr ucode ver err\n");
rval = -1;
break;
}
ucode_ver_to_str(&ver, ver_str, sizeof(ver_str));
if (!pctx->live_mode) {
/* write ucode version head in binary */
printf("Cavalry Ucode Version = %s", ver_str);
if (fwrite(&ver, sizeof(ver), 1, fp) != 1) {
perror("fwrite");
rval = -1;
break;
}
} else {
/* check ucode version */
if (memcmp(&pctx->ucode_ver, &ver, sizeof(ver))) {
printf("ucode version mismatch, curr: %s, input: %s\n",
ver_str, pctx->ucode_ver_str);
rval = -1;
break;
}
}
buffer = malloc(BUFFERSIZE);
if (buffer == NULL) {
perror("malloc");
rval = -1;
break;
}
while (run_loop) {
size = read(pctx->fd_cav, buffer, BUFFERSIZE);
if (size < 0) {
perror("read log in cavalry");
rval = -1;
break;
}
entry = (struct cavalry_log_entry *)buffer;
if (!pctx->live_mode) {
if (fwrite(buffer, 1, size, fp) != (unsigned long)size) {
perror("fwrite");
rval = -1;
break;
}
} else {
for (i = 0; i < size / sizeof(struct cavalry_log_entry); i++) {
if (entry->slot_id == HOTLINK_SLOT_CAVALRY_FRAMEWORK) {
printf("[T %u C %u t %u :%u] ", entry->orc_type,
entry->core_id, entry->thread_id, entry->seq_num);
if (entry->format_offset > pctx->ucode_size) {
printf("offset is lager than size: 0x%x > 0x%x\n",
entry->format_offset, pctx->ucode_size);
rval = -1;
break;
}
printf((char *)(entry->format_offset + pctx->ucode_addr),
entry->arg1, entry->arg2, entry->arg3, entry->arg4,
entry->arg5);
} else {
/* hotlink */
printf("[T %u C %u t %u S %u:%u] ", entry->orc_type,
entry->core_id, entry->thread_id, entry->slot_id, entry->seq_num);
found = 0;
for (j = 0; j < pctx->slot_num; j++) {
if (entry->slot_id == pctx->slot[j].slot_id) {
found = 1;
break;
}
}
if (!found) {
printf("Not found slot id: %u ucode file\n", entry->slot_id);
rval = -1;
break;
} else {
if (entry->format_offset > pctx->slot[j].size) {
printf("offset is lager than size: 0x%x > 0x%lx\n",
entry->format_offset, pctx->slot[j].size);
rval = -1;
break;
}
printf((char *)(entry->format_offset + pctx->slot[j].virt),
entry->arg1, entry->arg2, entry->arg3, entry->arg4,
entry->arg5);
}
}
entry++;
}
}
}
} while (0);
if (buffer != NULL) {
free(buffer);
buffer = NULL;
}
if (fp != NULL) {
fclose(fp);
fp = NULL;
}
return rval;
}
static int parse(struct log_ctx *pctx)
{
struct version_info_s ver;
char ver_str[FILENAME_LENGTH] = {0};
struct cavalry_log_entry entry;
FILE *fp_in = NULL;
FILE *fp_out = NULL;
long file_length = 0;
uint32_t i = 0, j = 0;
int found = 0, rval = 0;
do {
fp_in = fopen(pctx->input_bin_name, "rb");
if (fp_in == NULL) {
perror("fopen");
printf("fopen %s err\n", pctx->input_bin_name);
rval = -1;
break;
}
file_length = get_file_size(pctx->input_bin_name);
fp_out = fopen(pctx->output_text_name, "wb");
if (fp_out == NULL) {
perror("fopen");
printf("fopen %s err\n", pctx->output_text_name);
rval = -1;
break;
}
if (fread(&ver, sizeof(ver), 1, fp_in) != 1) {
perror("fread ucode ver");
rval = -1;
break;
}
ucode_ver_to_str(&ver, ver_str, FILENAME_LENGTH);
if (memcmp(&pctx->ucode_ver, &ver, sizeof(ver))) {
printf("ucode version mismatch, curr: %s, input: %s\n",
ver_str, pctx->ucode_ver_str);
rval = -1;
break;
}
if (fwrite(pctx->ucode_ver_str, strlen(pctx->ucode_ver_str), 1, fp_out) != 1) {
perror("fwrite ucode ver");
rval = -1;
break;
}
for (i = 0; i < file_length / sizeof(struct cavalry_log_entry); i++) {
if (fread(&entry, sizeof(struct cavalry_log_entry), 1, fp_in) != 1) {
perror("fread");
rval = -1;
break;
}
if (entry.slot_id == HOTLINK_SLOT_CAVALRY_FRAMEWORK) {
fprintf(fp_out, "[T %u C %u t %u :%u] ", entry.orc_type,
entry.core_id, entry.thread_id, entry.seq_num);
fprintf(fp_out, (char *)(entry.format_offset + pctx->ucode_addr),
entry.arg1, entry.arg2, entry.arg3, entry.arg4, entry.arg5);
} else {
fprintf(fp_out, "[T %u C %u t %u S %u:%u] ", entry.orc_type,
entry.core_id, entry.thread_id, entry.slot_id, entry.seq_num);
found = 0;
for (j = 0; j < pctx->slot_num; j++) {
if (entry.slot_id == pctx->slot[j].slot_id) {
found = 1;
break;
}
}
if (!found) {
printf("Not found slot id: %u ucode file\n", entry.slot_id);
rval = -1;
break;
} else {
fprintf(fp_out, (char *)(entry.format_offset + pctx->slot[j].virt),
entry.arg1, entry.arg2, entry.arg3, entry.arg4, entry.arg5);
}
}
}
if (rval < 0) {
printf("parser err\n");
break;
} else {
printf("save log %s ok, line num: %u\n", pctx->output_text_name, i);
}
} while (0);
if (fp_out != NULL) {
fclose(fp_out);
fp_out = NULL;
}
if (fp_in != NULL) {
fclose(fp_in);
fp_in = NULL;
}
return rval;
}
static void release(struct log_ctx *pctx)
{
uint32_t i = 0;
for (i = 0; i < pctx->slot_num; i++) {
if (pctx->slot[i].virt) {
if (munmap(pctx->slot[i].virt, pctx->slot[i].size) < 0) {
perror("munmap slot");
}
pctx->slot[i].virt = NULL;
}
if (pctx->slot[i].fd_slot > 0) {
close(pctx->slot[i].fd_slot);
pctx->slot[i].fd_slot = -1;
}
}
if (pctx->ucode_addr) {
if (munmap(pctx->ucode_addr, pctx->ucode_size) < 0) {
perror("munmap ucode");
}
pctx->ucode_addr = NULL;
}
if (pctx->fd_ucode > 0) {
close(pctx->fd_ucode);
pctx->fd_ucode = -1;
}
if (pctx->fd_cav > 0) {
close(pctx->fd_cav);
pctx->fd_cav = -1;
}
}
static void sigstop()
{
run_loop = 0;
printf("sigstop\n");
}
int main(int argc, char **argv)
{
struct log_ctx log;
struct log_ctx *pctx = &log;
struct cavalry_set_log_level set_log_level;
int rval = 0;
signal(SIGINT, sigstop);
signal(SIGQUIT, sigstop);
signal(SIGTERM, sigstop);
if (argc < 2) {
usage();
return -1;
}
memset(pctx, 0, sizeof(struct log_ctx));
pctx->log_level = CAVALRY_LOG_LEVEL_NORMAL;
if (init_param(argc, argv, pctx) < 0) {
return -1;
}
pctx->fd_cav = open(CAVALRY_DEV_NODE, O_RDWR, 0);
if (pctx->fd_cav < 0) {
perror(CAVALRY_DEV_NODE);
return -1;
}
do {
if (pctx->ucode_flag) {
if (map_ucode(pctx) < 0) {
printf("map ucode err\n");
rval = -1;
break;
}
if (map_hotlink(pctx) < 0) {
printf("map hotlink err\n");
rval = -1;
break;
}
get_ucode_ver(pctx->ucode_addr, &pctx->ucode_ver);
ucode_ver_to_str(&pctx->ucode_ver,
pctx->ucode_ver_str, FILENAME_LENGTH);
printf("Cavalry Ucode Version = %s", pctx->ucode_ver_str);
}
if (pctx->log_level_flag) {
memset(&set_log_level, 0, sizeof(set_log_level));
set_log_level.log_level = pctx->log_level;
if (ioctl(pctx->fd_cav, CAVALRY_SET_LOG_LEVEL, &set_log_level) < 0) {
perror("CAVALRY_SET_LOG_LEVEL");
rval = -1;
break;
}
}
if (pctx->capture_bin_flag || pctx->live_mode) {
if (ioctl(pctx->fd_cav, CAVALRY_START_LOG, NULL) < 0) {
perror("CAVALRY_START_LOG");
rval = -1;
break;
}
// Capture
rval = capture(pctx);
if (ioctl(pctx->fd_cav, CAVALRY_STOP_LOG, NULL) < 0) {
perror("CAVALRY_STOP_LOG");
rval = -1;
break;
}
} else if (pctx->input_bin_flag && pctx->output_text_flag && pctx->ucode_flag) {
// Parse
rval = parse(pctx);
}
} while (0);
release(pctx);
return rval;
}
