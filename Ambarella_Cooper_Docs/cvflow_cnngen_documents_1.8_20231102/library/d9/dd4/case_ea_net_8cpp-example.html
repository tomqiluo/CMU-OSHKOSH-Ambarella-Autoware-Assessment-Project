int p01a_ea_net_01()
{
int rval = 0;
int features = 0;
ea_net_t *net = NULL;
char in_path[2 * MAX_PATH_STRLEN + 1] = {0};
ea_tensor_t *image_bin = NULL;
ea_tensor_t *output_bin = NULL;
size_t shape[EA_DIM];
do {
EA_R_ASSERT(strlen(test_api_input_dir()) != 0);
EA_R_ASSERT(strlen(test_api_arch_dir()) != 0);
features = EA_ENV_ENABLE_CAVALRY | EA_ENV_ENABLE_NNCTRL;
EA_R_OK(ea_env_open(features));
if (ea_env_arch() == EA_ENV_ARCH_CV72) {
printf("ignored this case for CV72\n");
ea_env_close();
break;
}
net = ea_net_new(NULL);
EA_R_ASSERT(net != NULL);
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_arch_dir(), "onnx_yolox_tiny_cavalry.bin");
ea_net_config_input(net, "images");
ea_net_config_output(net, "output");
EA_R_OK(ea_net_load(net, EA_NET_LOAD_FILE, in_path, 1));
EA_R_ASSERT(ea_tensor_shape(ea_net_input(net, "images"))[EA_N] == 1);
EA_R_ASSERT(ea_tensor_shape(ea_net_input(net, "images"))[EA_C] == 3);
EA_R_ASSERT(ea_tensor_shape(ea_net_input(net, "images"))[EA_H] == 416);
EA_R_ASSERT(ea_tensor_shape(ea_net_input(net, "images"))[EA_W] == 416);
EA_R_ASSERT(ea_tensor_shape(ea_net_output(net, "output"))[EA_N] == 1);
EA_R_ASSERT(ea_tensor_shape(ea_net_output(net, "output"))[EA_C] == 1);
EA_R_ASSERT(ea_tensor_shape(ea_net_output(net, "output"))[EA_H] == 3549);
EA_R_ASSERT(ea_tensor_shape(ea_net_output(net, "output"))[EA_W] == 85);
EA_R_ASSERT(ea_tensor_dtype(ea_net_input(net, "images")) == EA_U8);
EA_R_ASSERT(ea_tensor_data_format(ea_net_input(net, "images"))->sign == 0);
EA_R_ASSERT(ea_tensor_data_format(ea_net_input(net, "images"))->datasize == 0);
EA_R_ASSERT(ea_tensor_data_format(ea_net_input(net, "images"))->exp_offset == 0);
EA_R_ASSERT(ea_tensor_data_format(ea_net_input(net, "images"))->exp_bits == 0);
EA_R_ASSERT(ea_tensor_dtype(ea_net_output(net, "output")) == EA_F32);
EA_R_ASSERT(ea_tensor_data_format(ea_net_output(net, "output"))->sign == 1);
EA_R_ASSERT(ea_tensor_data_format(ea_net_output(net, "output"))->datasize == 2);
EA_R_ASSERT(ea_tensor_data_format(ea_net_output(net, "output"))->exp_offset == 0);
EA_R_ASSERT(ea_tensor_data_format(ea_net_output(net, "output"))->exp_bits == 7);
shape[EA_N] = 1;
shape[EA_C] = 3;
shape[EA_H] = 416;
shape[EA_W] = 416;
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_input_dir(), "dog.bin");
image_bin = ea_tensor_new_from_file(EA_U8, shape, in_path);
EA_R_ASSERT(image_bin != NULL);
EA_R_ASSERT(ea_tensor_size(image_bin) == ea_tensor_size(ea_net_input(net, "images")));
memcpy(ea_tensor_data(ea_net_input(net, "images")), ea_tensor_data(image_bin), ea_tensor_size(image_bin));
EA_R_OK(ea_tensor_sync_cache(ea_net_input(net, "images"), EA_CPU, EA_VP));
EA_R_OK(ea_net_forward(net, 1));
EA_R_OK(ea_tensor_sync_cache(ea_net_output(net, "output"), EA_VP, EA_CPU));
shape[EA_N] = 1;
shape[EA_C] = 1;
shape[EA_H] = 3549;
shape[EA_W] = 85;
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_arch_dir(), "golden_sample_onnx_yolox_tiny_output.bin");
output_bin = ea_tensor_new_from_file(EA_F32, shape, in_path);
EA_R_ASSERT(output_bin != NULL);
EA_R_OK(tensor_data_compare_equal(output_bin, ea_net_output(net, "output")));
ea_tensor_free(output_bin);
ea_tensor_free(image_bin);
ea_net_free(net);
ea_env_close();
} while (0);
if (rval < 0) {
if (net) {
ea_net_free(net);
}
if (image_bin) {
ea_tensor_free(image_bin);
}
if (output_bin) {
ea_tensor_free(output_bin);
}
ea_env_close();
}
return rval;
}

net = ea_net_new(NULL);
EA_R_ASSERT(net != NULL);
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_arch_dir(), "onnx_yolox_tiny_cavalry.bin");
ea_net_config_input(net, "images");
ea_net_config_output(net, "output");
EA_R_OK(ea_net_load(net, EA_NET_LOAD_FILE, in_path, 1));
EA_MEASURE_TIME_START();
EA_R_OK(ea_net_forward(net, 1));
EA_MEASURE_TIME_END("net forward");
EA_R_OK(vp_time_is_in_span(&pairs, EA_MEASURE_TIME_US()));

if (net) {
ea_net_free(net);
}
ea_env_close();

return rval;
