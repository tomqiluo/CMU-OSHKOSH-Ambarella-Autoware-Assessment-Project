int p05a_ea_env_03()
{
int rval = 0;
int stream_id = 0;
unsigned long cv2x_overlay_buffer_size = 0x011EFC00; // IAV return length is less then 0x01200000
unsigned long cv5x_overlay_buffer_size = 0x011E7000; // IAV return length is less then 0x01200000
unsigned long cv7x_overlay_buffer_size = 0x011DD000; // IAV return length is less then 0x01200000
unsigned long overlay_buffer_offset;
ea_osd_t *osd = NULL;
do {
EA_R_OK(ea_env_open(EA_ENV_ENABLE_OSD_STREAM));
switch (ea_env_arch()) {
case EA_ENV_ARCH_CV22:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv2x_overlay_buffer_size);
overlay_buffer_offset = cv2x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
case EA_ENV_ARCH_CV2:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv2x_overlay_buffer_size);
overlay_buffer_offset = cv2x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
case EA_ENV_ARCH_CV25:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv2x_overlay_buffer_size);
overlay_buffer_offset = cv2x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
case EA_ENV_ARCH_CV28:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv2x_overlay_buffer_size);
overlay_buffer_offset = cv2x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
case EA_ENV_ARCH_CV5:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv5x_overlay_buffer_size);
overlay_buffer_offset = cv5x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
case EA_ENV_ARCH_CV52:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv5x_overlay_buffer_size);
overlay_buffer_offset = cv5x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
case EA_ENV_ARCH_CV72:
EA_R_ASSERT(ea_env_overlay_buffer_size() == cv7x_overlay_buffer_size);
overlay_buffer_offset = cv7x_overlay_buffer_size - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN;
break;
default :
EA_LOG_ERROR("ARCH is invalid: %d\n", ea_env_arch());
rval = -1;
break;
}
EA_R_BREAK();
osd = ea_osd_new(EA_OSD_STREAM, stream_id, NULL);
EA_R_ASSERT(osd != NULL);
ea_osd_free(osd);
ea_env_close();
EA_R_OK(ea_env_open(EA_ENV_ENABLE_OSD_STREAM));
EA_R_OK(ea_env_set_overlay_buffer_offset(overlay_buffer_offset - EA_ENV_STREAM_OVERLAY_BUFFER_OFFSET_ALIGN));
osd = ea_osd_new(EA_OSD_STREAM, stream_id, NULL);
EA_R_ASSERT(osd == NULL);
ea_env_close();
} while (0);
if (rval < 0) {
ea_env_close();
}
return rval;
}
static void *memory_leak_test_thread_func(void *arg)
{
int rval = 0;
int features = 0;
int *quit_flag = (int *)arg;
ea_tensor_t *tensor = NULL;
size_t shape[EA_DIM];
ea_img_resource_t *canvas_resource = NULL;
ea_img_resource_t *pyramid_resource = NULL;
int canvas_id = 0;
ea_img_resource_data_t canvas_data[3];
int channel_id = 0;
ea_img_resource_data_t pyramid_data[3];
ea_net_t *net = NULL;
char in_path[2 * MAX_PATH_STRLEN + 1] = {0};
do {
features = EA_ENV_ENABLE_IAV |
EA_ENV_ENABLE_CAVALRY |
EA_ENV_ENABLE_VPROC |
EA_ENV_ENABLE_NNCTRL |
EA_ENV_ENABLE_OSD_STREAM;
EA_R_OK(ea_env_open(features));
shape[EA_N] = 1;
shape[EA_C] = 3;
shape[EA_H] = (rand() % 1080) + 1;
shape[EA_W] = (rand() % 1920) + 1;
tensor = ea_tensor_new(EA_U8, shape, 0);
RVAL_ASSERT(tensor != NULL);
canvas_resource = ea_img_resource_new(EA_CANVAS, (void *)(unsigned long)canvas_id);
EA_R_ASSERT(canvas_resource != NULL);
EA_R_OK(ea_img_resource_hold_data(canvas_resource, &canvas_data[0]));
EA_R_OK(ea_img_resource_hold_data(canvas_resource, &canvas_data[1]));
EA_R_OK(ea_img_resource_hold_data(canvas_resource, &canvas_data[2]));
EA_R_OK(ea_img_resource_drop_data(canvas_resource, &canvas_data[0]));
EA_R_OK(ea_img_resource_drop_data(canvas_resource, &canvas_data[1]));
EA_R_OK(ea_img_resource_drop_data(canvas_resource, &canvas_data[2]));
pyramid_resource = ea_img_resource_new(EA_PYRAMID, (void *)(unsigned long)channel_id);
EA_R_ASSERT(pyramid_resource != NULL);
EA_R_OK(ea_img_resource_hold_data(pyramid_resource, &pyramid_data[0]));
EA_R_OK(ea_img_resource_hold_data(pyramid_resource, &pyramid_data[1]));
EA_R_OK(ea_img_resource_hold_data(pyramid_resource, &pyramid_data[2]));
EA_R_OK(ea_img_resource_drop_data(pyramid_resource, &pyramid_data[0]));
EA_R_OK(ea_img_resource_drop_data(pyramid_resource, &pyramid_data[1]));
EA_R_OK(ea_img_resource_drop_data(pyramid_resource, &pyramid_data[2]));
net = ea_net_new(NULL);
EA_R_ASSERT(net != NULL);
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_arch_dir(), "onnx_fgfd_cavalry.bin");
ea_net_config_input(net, "input");
ea_net_config_output(net, "scores");
ea_net_config_output(net, "boxes");
EA_R_OK(ea_net_load(net, EA_NET_LOAD_FILE, in_path, 1));
EA_R_OK(ea_net_forward(net, 1));
ea_tensor_free(tensor);
tensor = NULL;
ea_img_resource_free(canvas_resource);
canvas_resource = NULL;
ea_img_resource_free(pyramid_resource);
pyramid_resource = NULL;
ea_net_free(net);
net = NULL;
ea_env_close();
} while (*quit_flag == 0);
if (rval < 0) {
if (tensor) {
ea_tensor_free(tensor);
}
if (canvas_resource) {
ea_img_resource_free(canvas_resource);
}
if (pyramid_resource) {
ea_img_resource_free(pyramid_resource);
}
if (net) {
ea_net_free(net);
}
ea_env_close();
}
return (void *)(unsigned long)rval;
}
int p05m_ea_env_04()
{
int rval = 0;
int quit_flag = 0;
pthread_t test_thread;
do {
RVAL_OK(pthread_create(&test_thread, NULL, memory_leak_test_thread_func, &quit_flag));
EA_R_OK(manual_confirm("Please open another shell to use the following command to monitor the memory usage of test_api_eazyai_core process.\n"
"board # top\n"
"Is the memory usage of test_api_eazyai_core waving without continuous increasing?"));
quit_flag = 1;
pthread_join(test_thread, NULL);
} while (0);
if (rval < 0) {
quit_flag = 1;
pthread_join(test_thread, NULL);
}
return rval;
}
