int file_binary_compare_equal(const char *file_path_1, const char *file_path_2)
{
int rval = 0;
FILE *f1 = NULL;
FILE *f2 = NULL;
size_t f1_size;
size_t f2_size;
uint8_t *f1_data = NULL;
uint8_t *f2_data = NULL;
size_t bytes_read;
do {
f1 = fopen(file_path_1, "rb");
EA_R_ASSERT(f1 != NULL);
f2 = fopen(file_path_2, "rb");
EA_R_ASSERT(f2 != NULL);
fseek(f1, 0L, SEEK_END);
f1_size = ftell(f1);
fseek(f1, 0L, SEEK_SET);
fseek(f2, 0L, SEEK_END);
f2_size = ftell(f2);
fseek(f2, 0L, SEEK_SET);
EA_R_ASSERT(f1_size == f2_size);
f1_data = (uint8_t *)malloc(f1_size);
EA_R_ASSERT(f1_data != NULL);
f2_data = (uint8_t *)malloc(f2_size);
EA_R_ASSERT(f2_data != NULL);
bytes_read = fread(f1_data, 1, f1_size, f1);
EA_R_ASSERT(bytes_read == f1_size);
bytes_read = fread(f2_data, 1, f2_size, f2);
EA_R_ASSERT(bytes_read == f2_size);
EA_R_ASSERT(memcmp(f1_data, f2_data, f1_size) == 0);
fclose(f1);
fclose(f2);
} while (0);
if (rval < 0) {
if (f1) {
fclose(f1);
}
if (f2) {
fclose(f2);
}
if (f1_data) {
free(f1_data);
}
if (f2_data) {
free(f2_data);
}
}
return rval;
}
int tensor_data_compare_equal(ea_tensor_t *tensor_1, ea_tensor_t *tensor_2)
{
int rval = 0;
int element_size = 0;
uint8_t *data_1 = NULL;
uint8_t *data_2 = NULL;
size_t i;
do {
EA_R_ASSERT(ea_tensor_shape(tensor_1)[EA_N] == ea_tensor_shape(tensor_2)[EA_N]);
EA_R_ASSERT(ea_tensor_shape(tensor_1)[EA_C] == ea_tensor_shape(tensor_2)[EA_C]);
EA_R_ASSERT(ea_tensor_shape(tensor_1)[EA_H] == ea_tensor_shape(tensor_2)[EA_H]);
EA_R_ASSERT(ea_tensor_shape(tensor_1)[EA_W] == ea_tensor_shape(tensor_2)[EA_W]);
EA_R_ASSERT(ea_tensor_data_format(tensor_1)->sign == ea_tensor_data_format(tensor_2)->sign);
EA_R_ASSERT(ea_tensor_data_format(tensor_1)->datasize == ea_tensor_data_format(tensor_2)->datasize);
EA_R_ASSERT(ea_tensor_data_format(tensor_1)->exp_offset == ea_tensor_data_format(tensor_2)->exp_offset);
EA_R_ASSERT(ea_tensor_data_format(tensor_1)->exp_bits == ea_tensor_data_format(tensor_2)->exp_bits);
switch (ea_tensor_data_format(tensor_1)->datasize) {
case 0:
element_size = 1;
break;
case 1:
element_size = 2;
break;
case 2:
element_size = 4;
break;
default :
break;
}
EA_R_ASSERT(element_size != 0);
data_1 = (uint8_t *)ea_tensor_data_for_read(tensor_1, EA_CPU);
data_2 = (uint8_t *)ea_tensor_data_for_read(tensor_2, EA_CPU);
for (i = 0; i < ea_tensor_shape(tensor_1)[EA_N] * ea_tensor_shape(tensor_1)[EA_C] * ea_tensor_shape(tensor_1)[EA_H]; i++) {
EA_R_ASSERT(memcmp(data_1, data_2, ea_tensor_shape(tensor_1)[EA_W] * element_size) == 0);
data_1 += ea_tensor_pitch(tensor_1);
data_2 += ea_tensor_pitch(tensor_2);
}
EA_R_BREAK();
} while (0);
return rval;
}
static float float16_in_uint16_to_float32(uint16_t f16_in_u16)
{
typedef union {
float f32;
uint32_t u32;
} f32_u32_union_t;
f32_u32_union_t magic = {.u32 = (254UL - 15UL) << 23};
f32_u32_union_t was_infnan = {.u32 = (127UL + 16UL) << 23};
f32_u32_union_t out = {.u32 = (f16_in_u16 & 0x7FFFU) << 13};
out.f32 *= magic.f32;
if (out.f32 >= was_infnan.f32)
{
out.u32 |= 255UL << 23;
}
out.u32 |= (f16_in_u16 & 0x8000U) << 16;
return out.f32;
}
static int float16_in_uint16_compare_float32_equal(uint16_t f16, float f32)
{
int rval = 0;
float to_f32 = float16_in_uint16_to_float32(f16);
if (abs(to_f32 - f32) > 1e-6) {
rval = -1;
}
return rval;
}
int tensor_data_f16_f32_compare_equal(ea_tensor_t *f16_tensor, ea_tensor_t *f32_tensor)
{
int rval = 0;
const size_t *shape = NULL;
uint16_t *f16_data = NULL;
float *f32_data = NULL;
size_t i, k;
do {
EA_R_ASSERT(ea_tensor_shape(f16_tensor)[EA_N] == ea_tensor_shape(f32_tensor)[EA_N]);
EA_R_ASSERT(ea_tensor_shape(f16_tensor)[EA_C] == ea_tensor_shape(f32_tensor)[EA_C]);
EA_R_ASSERT(ea_tensor_shape(f16_tensor)[EA_H] == ea_tensor_shape(f32_tensor)[EA_H]);
EA_R_ASSERT(ea_tensor_shape(f16_tensor)[EA_W] == ea_tensor_shape(f32_tensor)[EA_W]);
EA_R_ASSERT(ea_tensor_data_format(f16_tensor)->sign == 1);
EA_R_ASSERT(ea_tensor_data_format(f16_tensor)->datasize == 1);
EA_R_ASSERT(ea_tensor_data_format(f16_tensor)->exp_offset == 0);
EA_R_ASSERT(ea_tensor_data_format(f16_tensor)->exp_bits == 4);
EA_R_ASSERT(ea_tensor_data_format(f32_tensor)->sign == 1);
EA_R_ASSERT(ea_tensor_data_format(f32_tensor)->datasize == 2);
EA_R_ASSERT(ea_tensor_data_format(f32_tensor)->exp_offset == 0);
EA_R_ASSERT(ea_tensor_data_format(f32_tensor)->exp_bits == 7);
f16_data = (uint16_t *)ea_tensor_data_for_read(f16_tensor, EA_CPU);
f32_data = (float *)ea_tensor_data_for_read(f32_tensor, EA_CPU);
shape = ea_tensor_shape(f16_tensor);
for (i = 0; i < shape[EA_N] * shape[EA_C] * shape[EA_H]; i++) {
for (k = 0; k < shape[EA_W]; k++) {
EA_R_OK(float16_in_uint16_compare_float32_equal(f16_data[k], f32_data[k]));
}
EA_R_BREAK();
f16_data = (uint16_t *)((uint8_t *)f16_data + ea_tensor_pitch(f16_tensor));
f32_data = (float *)((uint8_t *)f32_data + ea_tensor_pitch(f32_tensor));
}
EA_R_BREAK();
} while (0);
return rval;
}
int manual_confirm(const char * question_format, ...)
{
int rval = 0;
char yes_or_no = 'n';
do {
va_list args;
va_start(args, question_format);
vprintf(question_format, args);
va_end(args);
printf(" (y/n)");
do {
EA_R_OK(scanf("%c", &yes_or_no));
} while (yes_or_no == '\n');
EA_R_BREAK();
EA_R_ASSERT(yes_or_no == 'y');
do {

return rval;
}
static void calc_expected_time_span(const int expected_time_us, int *span_lower, int *span_upper)
{
if (span_lower != NULL && span_upper != NULL) {
if (expected_time_us < 10 * 1000) {
*span_lower = expected_time_us - 2 * 1000;
*span_upper = expected_time_us + 2 * 1000;
} else if (expected_time_us < 50 * 1000) {
*span_lower = expected_time_us - 5 * 1000;
*span_upper = expected_time_us + 5 * 1000;
} else if (expected_time_us < 100 * 1000) {
*span_lower = expected_time_us - 10 * 1000;
*span_upper = expected_time_us + 10 * 1000;
} else {
*span_lower = expected_time_us - 20 * 1000;
*span_upper = expected_time_us + 20 * 1000;
}
}
}

static void usage(void)
{
printf("Options:\n");
for (size_t i = 0; i < sizeof(long_options) / sizeof(long_options[0]) - 1; i++) {
if (isalpha(long_options[i].val)) {

static int parse_param(int argc, char **argv, test_params_t *params)
{
int rval = 0;
int ch;
int option_index = 0;
int value;
do {
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'i':
value = strlen(optarg);
if (value == 0) {
printf("input_dir is empty\n");
rval = -1;
break;
}
if (value > MAX_PATH_STRLEN) {
printf("input_dir should be no more than %d characters, %s\n", MAX_PATH_STRLEN, optarg);
rval = -1;
break;
}
snprintf(params->input_dir, sizeof(params->input_dir), "%s", optarg);
if (optarg[value - 1] != '/') {
strncat(params->input_dir, "/", MAX_PATH_STRLEN - strlen(params->input_dir));
}
break;
case 'a':
value = strlen(optarg);
if (value == 0) {
printf("arch_dir is empty\n");
rval = -1;
break;
}
if (value > MAX_PATH_STRLEN) {
printf("arch_dir should be no more than %d characters, %s\n", MAX_PATH_STRLEN, optarg);
rval = -1;
break;
}
snprintf(params->arch_dir, sizeof(params->arch_dir), "%s", optarg);
if (optarg[value - 1] != '/') {
strncat(params->arch_dir, "/", MAX_PATH_STRLEN - strlen(params->arch_dir));
}
break;
case 'o':
value = strlen(optarg);
if (value == 0) {
printf("output_dir is empty\n");
rval = -1;
break;
}
if (value > MAX_PATH_STRLEN) {
printf("output_dir should be no more than %d characters, %s\n", MAX_PATH_STRLEN, optarg);
rval = -1;
break;
}
snprintf(params->output_dir, sizeof(params->output_dir), "%s", optarg);
if (optarg[value - 1] != '/') {
strncat(params->output_dir, "/", MAX_PATH_STRLEN - strlen(params->output_dir));
}
break;
case 'l':
params->log_level = atoi(optarg);
if (params->log_level < EA_LOG_LEVEL_NONE || params->log_level > EA_LOG_LEVEL_VERBOSE) {
printf("log level is wrong, %d\n", params->log_level);
return -1;
}
break;
case 'b':
params->break_failed = atoi(optarg);
break;
case 's':
strncpy(params->case_prefix, optarg, MAX_PATH_STRLEN);
break;
case 'h':
params->help = 1;
break;
default:
printf("unknown option found: %c\n", ch);
rval = -1;
break;
}
}
} while (0);
return rval;
}
static int init_param(int argc, char **argv, test_params_t *params)
{
int rval = 0;
memset(params, 0, sizeof(test_params_t));
params->log_level = EA_LOG_LEVEL_NONE;
params->break_failed = 0;
do {
if (argc < 2) {
usage();
exit(0);
} else {
rval = parse_param(argc, argv, params);
if (rval < 0) {
break;
}
if (test_params.help) {
usage();
printf("%s\n", help_info);
exit(0);
}
}
EA_LOG_SET_LOCAL(params->log_level);
EA_LOG_NOTICE("test parameters:\n");
EA_LOG_NOTICE("\tinput_dir path: %s\n", params->input_dir);
EA_LOG_NOTICE("\tarch_dir path: %s\n", params->arch_dir);
EA_LOG_NOTICE("\toutput_dir path: %s\n", params->output_dir);
EA_LOG_NOTICE("\tlog level: %d (0=None, 1=Error, 2=Notice, 3=Debug, 4=Verbose)\n", params->log_level);
EA_LOG_NOTICE("\tbreak when failed: %d\n", params->break_failed);
EA_LOG_NOTICE("\tThe prefix of cases to test: %s\n", params->case_prefix);
if (strlen(params->input_dir) == 0) {
printf("The parameter --input_dir is necessary.\n");
rval = -1;
break;
}
if (strlen(params->arch_dir) == 0) {
printf("The parameter --arch_dir is necessary.\n");
rval = -1;
break;
}
if (strlen(params->output_dir) == 0) {
printf("The parameter --output_dir is necessary.\n");
rval = -1;
break;
}
if (strlen(params->case_prefix) == 0) {
printf("The parameter --case_prefix is necessary.\n");
rval = -1;
break;
}
} while (0);
return rval;
}

static long get_proc_ambarella_clock(const char *name)
{
int rval = 0;
FILE *clock_file = NULL;
long clock_hz = -1;
char *name_start = NULL;
char *colon = NULL;
char *hz_start = NULL;
char buf[MAX_PATH_STRLEN + 1];
do {
clock_file = fopen("/proc/ambarella/clock", "r");
EA_R_ASSERT(clock_file != NULL);
memset(buf, '\0', MAX_PATH_STRLEN + 1);
while(fgets(buf, MAX_PATH_STRLEN + 1, clock_file)) {
name_start = strstr(buf, name);
if (name_start) {
colon = strstr(name_start, ":");
EA_R_ASSERT(colon != NULL);
hz_start = strstr(colon, "Hz");
EA_R_ASSERT(hz_start != NULL);
clock_hz = strtol(colon + 1, NULL, 10);
EA_R_ASSERT(clock_hz >= 0 && clock_hz != LONG_MAX && clock_hz != LONG_MIN);
break;
}
}
EA_R_BREAK();
EA_R_ASSERT(name_start != NULL);
fclose(clock_file);
} while (0);
if (rval < 0) {
if (clock_file) {
