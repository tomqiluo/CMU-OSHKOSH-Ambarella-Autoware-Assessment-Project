int p05a_ea_preprocess_cvt_color_resize_cpu_01()
{
int rval = 0;
int features;
ea_img_resource_t *img_resource = NULL;
ea_img_resource_data_t data;
ea_tensor_t *nv12_tensor = NULL;
ea_tensor_t *rgb_tensor = NULL;
ea_tensor_t *bgr_tensor = NULL;
size_t shape[EA_DIM] = {1, 3, 0, 0};
int channel_id = 0;
char out_path_1[2 * MAX_PATH_STRLEN + 1] = {0};
char out_path_2[2 * MAX_PATH_STRLEN + 1] = {0};
do {
EA_R_ASSERT(strlen(test_api_output_dir()) != 0);
features = EA_ENV_ENABLE_IAV | EA_ENV_ENABLE_CAVALRY;
EA_R_OK(ea_env_open(features));
img_resource = ea_img_resource_new(EA_PYRAMID, (void *)(unsigned long)channel_id);
EA_R_OK(img_resource != NULL);
memset(&data, 0, sizeof(ea_img_resource_data_t));
EA_R_OK(ea_img_resource_hold_data(img_resource, &data));
nv12_tensor = data.tensor_group[1]; // layer 1 is 720x480
shape[EA_H] = ea_tensor_shape(nv12_tensor)[EA_H];
shape[EA_W] = ea_tensor_shape(nv12_tensor)[EA_W];
rgb_tensor = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(rgb_tensor != NULL);
bgr_tensor = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(bgr_tensor != NULL);
rval = ea_cvt_color_resize(nv12_tensor, rgb_tensor, EA_COLOR_YUV2RGB_NV12, EA_CPU);
EA_R_ASSERT(rval < 0);
EA_R_OK(ea_utils_env_callbacks_register());
EA_R_OK(ea_cvt_color_resize(nv12_tensor, rgb_tensor, EA_COLOR_YUV2RGB_NV12, EA_CPU));
EA_R_OK(ea_cvt_color_resize(rgb_tensor, bgr_tensor, EA_COLOR_RGB2BGR, EA_CPU));
snprintf(out_path_1, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_output_dir(), "p01a_ea_cvt_color_resize_cpu_01_out_1.png");
EA_R_OK(ea_tensor_to_image(nv12_tensor, EA_TENSOR_COLOR_MODE_YUV_NV12, out_path_1));
snprintf(out_path_2, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_output_dir(), "p01a_ea_cvt_color_resize_cpu_01_out_2.png");
EA_R_OK(ea_tensor_to_image(bgr_tensor, EA_TENSOR_COLOR_MODE_BGR, out_path_2));
EA_R_OK(file_binary_compare_equal(out_path_1, out_path_2));
// EA_R_OK(manual_confirm("Does %s have the content of canvas 1?", out_path));
EA_R_OK(ea_img_resource_drop_data(img_resource, &data));
ea_tensor_free(rgb_tensor);
ea_tensor_free(bgr_tensor);
ea_img_resource_free(img_resource);
ea_env_close();
} while (0);
if (rval < 0) {
if (img_resource) {
ea_img_resource_free(img_resource);
}
if (rgb_tensor) {
ea_tensor_free(rgb_tensor);
}
if (bgr_tensor) {
ea_tensor_free(bgr_tensor);
}
ea_env_close();
}
return rval;
}
int p00a_ea_preprocess_crop_resize_cpu_02()
{
int rval = 0;
ea_tensor_t *png_tensor = NULL;
ea_tensor_t *out_tensor = NULL;
size_t shape[EA_DIM];
char in_path[2 * MAX_PATH_STRLEN + 1] = {0};
char out_path[2 * MAX_PATH_STRLEN + 1] = {0};
ea_roi_t roi;
do {
EA_R_ASSERT(strlen(test_api_input_dir()) != 0);
EA_R_ASSERT(strlen(test_api_output_dir()) != 0);
EA_R_OK(ea_env_open(EA_ENV_ENABLE_CAVALRY));
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_input_dir(), "dog.png");
EA_R_OK(ea_utils_env_callbacks_register());
png_tensor = ea_tensor_new_from_image(in_path);
EA_R_ASSERT(png_tensor != NULL);
shape[EA_N] = ea_tensor_shape(png_tensor)[EA_N];
shape[EA_C] = ea_tensor_shape(png_tensor)[EA_C];
shape[EA_H] = ea_tensor_shape(png_tensor)[EA_H] / 2;
shape[EA_W] = ea_tensor_shape(png_tensor)[EA_W] / 2;
out_tensor = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(out_tensor != NULL);
roi.x = 0;
roi.y = 0;
roi.w = ea_tensor_shape(png_tensor)[EA_W];
roi.h = ea_tensor_shape(png_tensor)[EA_H];
EA_R_OK(ea_crop_resize(&png_tensor, 1, &out_tensor, 1, &roi, EA_TENSOR_COLOR_MODE_BGR, EA_CPU));
snprintf(out_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_output_dir(), "p00a_ea_crop_resize_cpu_02_out.png");
EA_R_OK(ea_tensor_to_image(out_tensor, EA_TENSOR_COLOR_MODE_BGR, out_path));
snprintf(in_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_input_dir(), "golden_sample_p00a_ea_crop_resize_cpu_02_out.png");
EA_R_OK(file_binary_compare_equal(in_path, out_path));
ea_tensor_free(png_tensor);
ea_tensor_free(out_tensor);
ea_env_close();
} while (0);
if (rval < 0) {
if (png_tensor) {
ea_tensor_free(png_tensor);
}
if (out_tensor) {
ea_tensor_free(out_tensor);
}
ea_env_close();
}
return rval;
}

shape[EA_H] * ea_tensor_pitch(nv12_tensor) * 2 / 3);
memcpy((uint8_t *)ea_tensor_data(nv12_tensor) + shape[EA_H] * ea_tensor_pitch(nv12_tensor) * 2 / 3,

shape[EA_H] = 480 * 3 / 2;
shape[EA_W] = 720;

shape[EA_H] = 480;
shape[EA_W] = 720;
out_gray = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(out_gray != NULL);
shape[EA_C] = 1;
shape[EA_H] = 720;
shape[EA_W] = 480;
out_gray_rot = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(out_gray_rot != NULL);
shape[EA_C] = 3;
shape[EA_H] = 480;
shape[EA_W] = 720;
out_rgb = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(out_rgb != NULL);
shape[EA_C] = 3;
shape[EA_H] = 720;
shape[EA_W] = 480;
out_rgb_rot = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(out_rgb_rot != NULL);
