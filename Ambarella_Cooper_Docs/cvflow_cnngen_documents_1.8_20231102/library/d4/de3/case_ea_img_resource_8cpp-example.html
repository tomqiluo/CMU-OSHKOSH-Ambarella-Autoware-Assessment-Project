int p02a_ea_img_resource_01()
{
int rval = 0;
int canvas_id = 0;
ea_img_resource_t *img_resource = NULL;
ea_img_resource_data_t data[3];
EA_MEASURE_TIME_DECLARE();
do {
EA_R_OK(ea_env_open(EA_ENV_ENABLE_IAV));
img_resource = ea_img_resource_new(EA_CANVAS, (void *)(unsigned long)canvas_id);
EA_R_ASSERT(img_resource != NULL);
EA_MEASURE_TIME_START();
EA_R_OK(ea_img_resource_hold_data(img_resource, &data[0]));
EA_R_OK(ea_img_resource_hold_data(img_resource, &data[1]));
EA_R_OK(ea_img_resource_hold_data(img_resource, &data[2]));
EA_MEASURE_TIME_END("blocking query");
EA_R_ASSERT(EA_MEASURE_TIME_US() > 33 * 1000);
EA_R_ASSERT(ea_tensor_shape(EA_CANVAS_YUV_TENSOR(data[0].tensor_group))[EA_H] == 1080);
EA_R_ASSERT(ea_tensor_shape(EA_CANVAS_YUV_TENSOR(data[0].tensor_group))[EA_W] == 1920);
EA_R_ASSERT(data[0].dsp_pts != data[1].dsp_pts);
EA_R_ASSERT(data[1].dsp_pts != data[2].dsp_pts);
EA_R_ASSERT(data[0].mono_pts != data[1].mono_pts);
EA_R_ASSERT(data[1].mono_pts != data[2].mono_pts);
EA_R_OK(ea_img_resource_drop_data(img_resource, &data[0]));
EA_R_OK(ea_img_resource_drop_data(img_resource, &data[1]));
EA_R_OK(ea_img_resource_drop_data(img_resource, &data[2]));
EA_MEASURE_TIME_START();
EA_R_OK(ea_img_resource_hold_data_nowait(img_resource, &data[0]));
EA_R_OK(ea_img_resource_hold_data_nowait(img_resource, &data[1]));
EA_R_OK(ea_img_resource_hold_data_nowait(img_resource, &data[2]));
EA_MEASURE_TIME_END("non blocking query");
EA_R_ASSERT(EA_MEASURE_TIME_US() < 5 * 1000);
EA_R_ASSERT(data[0].dsp_pts == data[1].dsp_pts || data[1].dsp_pts == data[2].dsp_pts);
EA_R_ASSERT(data[0].mono_pts == data[1].mono_pts || data[1].mono_pts == data[2].mono_pts);
EA_R_OK(ea_img_resource_drop_data(img_resource, &data[0]));
EA_R_OK(ea_img_resource_drop_data(img_resource, &data[1]));
EA_R_OK(ea_img_resource_drop_data(img_resource, &data[2]));
ea_img_resource_free(img_resource);
ea_env_close();
} while (0);
if (rval < 0) {
if (img_resource) {
ea_img_resource_free(img_resource);
}
ea_env_close();
}
return rval;
}
int p02m_ea_img_resource_02()
{
int rval = 0;
ea_img_resource_t *img_resource = NULL;
ea_img_resource_data_t data;
ea_tensor_t *canvas_0_tensor = NULL;
ea_tensor_t *canvas_1_tensor = NULL;
char out_path[2 * MAX_PATH_STRLEN + 1] = {0};
uint32_t dsp_pts[CANVAS_GROUP_PTS_SAMPLE_NUM];
uint64_t mono_pts[CANVAS_GROUP_PTS_SAMPLE_NUM];
uint64_t dsp_pts_interval_sum = 0;
uint64_t mono_pts_interval_sum = 0;
uint64_t dsp_pts_interval_avg = 0;
uint64_t mono_pts_interval_avg = 0;
int i;
EA_MEASURE_TIME_DECLARE();
do {
EA_R_OK(ea_env_open(EA_ENV_ENABLE_IAV));
EA_R_OK(ea_utils_env_callbacks_register());
img_resource = ea_img_resource_new(EA_CANVAS_GROUP,
(void*)(unsigned long)ea_img_resource_canvas_id_map(2, 0, 1));
EA_R_ASSERT(img_resource != NULL);
EA_R_OK(ea_img_resource_hold_data(img_resource, &data));
canvas_0_tensor = EA_CANVAS_GROUP_YUV_TENSOR(data.tensor_group, 0);
canvas_1_tensor = EA_CANVAS_GROUP_YUV_TENSOR(data.tensor_group, 1);
EA_R_ASSERT(canvas_0_tensor != NULL);
EA_R_ASSERT(canvas_1_tensor != NULL);
EA_R_ASSERT(ea_tensor_shape(canvas_0_tensor)[EA_H] == 1080);
EA_R_ASSERT(ea_tensor_shape(canvas_0_tensor)[EA_W] == 1920);
EA_R_ASSERT(ea_tensor_shape(canvas_1_tensor)[EA_H] == 480);
EA_R_ASSERT(ea_tensor_shape(canvas_1_tensor)[EA_W] == 720);
snprintf(out_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_output_dir(), "p02m_ea_img_resource_02_out_1.jpg");
EA_R_OK(ea_tensor_to_image(canvas_0_tensor, EA_TENSOR_COLOR_MODE_YUV_NV12, out_path));
EA_R_OK(manual_confirm("Does %s have the content of canvas 0?", out_path));
snprintf(out_path, 2 * MAX_PATH_STRLEN + 1, "%s%s", test_api_output_dir(), "p02m_ea_img_resource_02_out_2.jpg");
EA_R_OK(ea_tensor_to_image(canvas_1_tensor, EA_TENSOR_COLOR_MODE_YUV_NV12, out_path));
EA_R_OK(manual_confirm("Does %s have the content of canvas 1?", out_path));
EA_R_OK(ea_img_resource_drop_data(img_resource, &data));
// check canvas group PTS interval is stable
EA_R_OK(ea_img_resource_hold_data(img_resource, &data));
dsp_pts[0] = data.dsp_pts;
mono_pts[0] = data.mono_pts;
EA_R_OK(ea_img_resource_drop_data(img_resource, &data));
for (i = 1; i < CANVAS_GROUP_PTS_SAMPLE_NUM; i++) {
EA_MEASURE_TIME_START();
EA_R_OK(ea_img_resource_hold_data(img_resource, &data));
EA_R_ASSERT(EA_MEASURE_TIME_US() < (33 + 5) * 1000);
dsp_pts[i] = data.dsp_pts;
dsp_pts_interval_sum += dsp_pts[i] - dsp_pts[i - 1];
mono_pts[i] = data.mono_pts;
mono_pts_interval_sum += mono_pts[i] - mono_pts[i - 1];
EA_R_OK(ea_img_resource_drop_data(img_resource, &data));
}
EA_R_BREAK();
dsp_pts_interval_avg = dsp_pts_interval_sum / (CANVAS_GROUP_PTS_SAMPLE_NUM - 1);
mono_pts_interval_avg = mono_pts_interval_sum / (CANVAS_GROUP_PTS_SAMPLE_NUM - 1);
if (ea_env_dsp_version() == EA_ENV_DSP_V5) {
EA_R_ASSERT(dsp_pts_interval_avg / 90 < 33 + 5); // 90KHz clock source for dsp_pts
EA_R_ASSERT(dsp_pts_interval_avg / 90 > 33 - 5);
}
EA_R_ASSERT(mono_pts_interval_avg / 90 < 33 + 5); // 90KHz clock source for mono_pts
EA_R_ASSERT(mono_pts_interval_avg / 90 > 33 - 5);
for (i = 1; i < CANVAS_GROUP_PTS_SAMPLE_NUM; i++) {
if (ea_env_dsp_version() == EA_ENV_DSP_V5) {
EA_R_ASSERT(ABS_DIFF(dsp_pts[i] - dsp_pts[i - 1], dsp_pts_interval_avg) / 90 < 5); // 90KHz clock source for dsp_pts
EA_R_ASSERT(ABS_DIFF(dsp_pts[i] - dsp_pts[i - 1], dsp_pts_interval_avg) / 90 < 5);
}
EA_R_ASSERT(ABS_DIFF(mono_pts[i] - mono_pts[i - 1], mono_pts_interval_avg) / 90 < 5); // 90KHz clock source for mono_pts
EA_R_ASSERT(ABS_DIFF(mono_pts[i] - mono_pts[i - 1], mono_pts_interval_avg) / 90 < 5);
}
EA_R_BREAK();
ea_img_resource_free(img_resource);
ea_env_close();
} while (0);
if (rval < 0) {
if (img_resource) {
ea_img_resource_free(img_resource);
}
ea_env_close();
}
return rval;
}
