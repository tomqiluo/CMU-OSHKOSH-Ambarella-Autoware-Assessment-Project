static void sort(nn_task_name_t* array, int size)
{
int j, i;
char tmp[MAX_STR_LEN - 1];
for (j = 0; j < (size - 1); j++) {
for(i = 0; i < (size - j - 1); i++) {
if(array[i].label[0] > array[i+1].label[0]) {
strncpy(tmp, array[i].label, MAX_STR_LEN - 1);
tmp[MAX_STR_LEN - 2] = '\0';
strncpy(array[i].label, array[i+1].label, MAX_STR_LEN);
array[i].label[MAX_STR_LEN - 1] = '\0';
strncpy(array[i+1].label, tmp, MAX_STR_LEN - 1);
array[i+1].label[MAX_STR_LEN - 1] = '\0';
}
}
}
}

#define PRINT_ROW 8
#define PRINT_COLUMN 6
#define SEPARATOR_LEN 3
#define ITEM_LEN 20
static void fill_print_buf(char *buf, struct option *opt, int n)
{
char *tmp = buf;
char str[ITEM_LEN] = {0};
int i;
for (i = 0; i < n; i++) {
if (isalpha(opt[i].val)) {
snprintf(str, 5, " -%c ", opt[i].val);
memcpy(tmp, str, strlen(str));
}
snprintf(str, strlen(opt[i].name) + 3, "--%s", opt[i].name);
memcpy(tmp + 4, str, strlen(str));
tmp = tmp + (ITEM_LEN + SEPARATOR_LEN) * PRINT_COLUMN;
}
}
static void print_equal_symbol()
{
int i;
for (i = 0; i < (ITEM_LEN + SEPARATOR_LEN) * PRINT_COLUMN; i++) {
printf("=");
}
printf("\n");
}
static void print_vertical_symbol(char *buf)
{
char *tmp = buf;
int i;
for (i = 0; i < PRINT_ROW; i++) {
*tmp = '|';
tmp = tmp + (ITEM_LEN + SEPARATOR_LEN) * PRINT_COLUMN;
}
}
static void generate_table(pipeline_info_t *info, char *buf, char *mode, int num)
{
int i = 0;
char format_str[MAX_STR_LEN] = {0};
char *tmp;
int skip = 0;
if (buf == NULL) {
printf("generate pipeline info error\n");
} else {
memset(buf, ' ', PRINT_ROW * ((ITEM_LEN + SEPARATOR_LEN) * PRINT_COLUMN));
printf("\ntest_eazyai %s pipeline:\n", mode);
snprintf(format_str, MAX_STR_LEN, "%%-%ds %%-%ds==>%%-%ds==>%%-%ds==>%%-%ds==>%%-%ds\n",
ITEM_LEN, ITEM_LEN, ITEM_LEN, ITEM_LEN, ITEM_LEN, ITEM_LEN);
print_equal_symbol();
printf(format_str, "Mode", "Input", "Preprocess", "Inference", "Postprocess", "Show_Results");
print_equal_symbol();
tmp = buf + 1;
memcpy(tmp, mode, strlen(mode));
for (i = 0; i < num; i++) {
if (skip == 1) {
tmp = buf + (ITEM_LEN + SEPARATOR_LEN) * (i + 2);
} else {
tmp = buf + (ITEM_LEN + SEPARATOR_LEN) * (i + 1);
}
fill_print_buf(tmp, info[i].option, info[i].num);
if (skip == 0 && strcmp(mode, "file") == 0) {
skip = 1;
}
}
skip = 0;
for (i = 1; i < PRINT_COLUMN; i++) {
if (skip == 1 && ((i + 1) < PRINT_COLUMN)) {
print_vertical_symbol(buf + (i + 1) * (ITEM_LEN + SEPARATOR_LEN));
} else {
if (skip == 0 && strcmp(mode, "file") == 0) {
skip = 1;
}
print_vertical_symbol(buf + i * (ITEM_LEN + SEPARATOR_LEN));
}
}
for (i = 0; i < PRINT_ROW; i++) {
*(buf + (i + 1) * ((ITEM_LEN + SEPARATOR_LEN) * PRINT_COLUMN - 1)) = '\0';
printf("%s\n", buf + i * (ITEM_LEN + SEPARATOR_LEN) * PRINT_COLUMN);
}
print_equal_symbol();
}
}

static int parse_param(int argc, char **argv, test_eazyai_params_t *params)
{
int rval = EA_SUCCESS;
int ch;
int option_index = 0;
int value;
do {
opterr = 0;
while ((ch = getopt_long(argc, argv, short_options, long_options, &option_index)) != -1) {
switch (ch) {
case 'm':
params->mode = atoi(optarg);
if (params->mode < RUN_MODE_LIVE || params->mode >= RUN_MODE_NUM) {
EA_LOG_ERROR("mode is wrong, %d\n", params->mode);
rval = EA_FAIL;
break;
}
break;
case 'v':
params->log_level = atoi(optarg);
if (params->log_level < EA_LOG_LEVEL_NONE || params->log_level >= EA_LOG_LEVEL_INVALID) {
EA_LOG_ERROR("log_level is wrong, %d\n", params->log_level);
rval = EA_FAIL;
break;
}
break;
case OPTION_OUTPUT_DIR:
value = strlen(optarg);
if (value == 0) {
EA_LOG_ERROR("output_dir is empty\n");
rval = EA_FAIL;
break;
}
if (value > MAX_STR_LEN) {
EA_LOG_ERROR("output_dir should be no more than %d characters, %s\n", MAX_STR_LEN, optarg);
rval = EA_FAIL;
break;
}
strncpy(params->output_dir, optarg, MAX_STR_LEN);
if (optarg[value - 1] != '/') {
strncat(params->output_dir, "/", MAX_STR_LEN - strlen(params->output_dir));
}
break;
case OPTION_EXTRA_INPUT:
value = strlen(optarg);
if (value == 0) {
EA_LOG_ERROR("path for extra input is empty\n");
rval = EA_FAIL;
break;
}
if (value > MAX_STR_LEN) {
EA_LOG_ERROR("path for extra input should be no more than %d characters, %s\n", MAX_STR_LEN, optarg);
rval = EA_FAIL;
break;
}
strncpy(params->extra_input, optarg, MAX_STR_LEN);
break;
case OPTION_SUPPORT_LIST:
params->support = 1;
break;
case 'd':
value = atoi(optarg);
if (value < DRAW_BBOX_TEXTBOX || value > DRAW_256_COLORS_IMAGE) {
EA_LOG_ERROR("draw mode is wrong, %d\n", params->draw_mode);
rval = EA_FAIL;
break;
}
params->draw_mode = value;
break;
case 's':
params->stream_id = atoi(optarg);
if (params->stream_id < 0) {
EA_LOG_ERROR("stream ID is wrong, %d\n", params->stream_id);
rval = EA_FAIL;
break;
}
params->out_type_map |= NN_OUT_DST_STREAM;
break;
case 'c':

static int check_params(test_eazyai_params_t *params)
{
int rval = EA_SUCCESS;
do {
if ((((params->out_type_map & (NN_OUT_DST_VOUT | NN_OUT_DST_STREAM)) ==
(NN_OUT_DST_VOUT | NN_OUT_DST_STREAM)) ||
(((params->out_type_map & (NN_OUT_DST_VOUT | NN_OUT_DST_STREAM)) == 0) &&
(params->stream_id >= 0 && params->vout_id >= 0))) &&
(params->mode == RUN_MODE_LIVE)) {

rval = EA_FAIL;
break;
}
} while (0);
return rval;
}

static int get_y_uv_id_for_yuv_input(ea_net_t *net, int *y_id, int *uv_id)
{
int rval = EA_SUCCESS;
int y_size, uv_size;
ea_tensor_t *y_tensor = NULL;
ea_tensor_t *uv_tensor = NULL;
do {
RVAL_ASSERT(net != NULL);
y_tensor = ea_net_input_by_index(net, 0);
uv_tensor = ea_net_input_by_index(net, 1);
y_size = ea_tensor_size(y_tensor);
uv_size = ea_tensor_size(uv_tensor);
RVAL_ASSERT((y_size == 2 * uv_size) || (2 * y_size == uv_size));
*y_id = y_size > uv_size ? 0 : 1;
*uv_id = y_size > uv_size ? 1: 0;
} while (0);
return rval;
}
static ea_tensor_t *create_yuv_tensor(ea_net_t *net, int y_id)
{
int rval = EA_SUCCESS;
ea_tensor_t *ret = NULL;
do {
RVAL_ASSERT(net != NULL);
ret =
ea_tensor_new_from_iav_anon_mem(ea_tensor_shape(ea_net_input_by_index(net, y_id)));
RVAL_ASSERT(ret != NULL)
} while (0);
if (rval < 0) {
if (ret != NULL) {
ea_tensor_free(ret);
ret = NULL;
}
}
return ret;
}
static int cvflow_init(test_eazyai_ctx_t *ctx, test_eazyai_params_t *params,
nn_cvflow_params_t *nn_cvflow_params, nn_prep_params_t *preproc_params, test_eazyai_nn_in_params_t *in_params)

nn_cvflow_params->nvp_id = params->nvp_id;

break;

memset(&info, 0, sizeof(info));

}
nn_in_file_set_file_info(ctx->in_ctx, &info, i);
}
} else {
EA_LOG_ERROR("%s cannot support %d mode.\n", __func__, params->mode);
rval = EA_FAIL;
}
RVAL_BREAK();
ctx->vp_output = (vp_output_t *)malloc(sizeof(vp_output_t));
RVAL_ASSERT(ctx->vp_output);
memset(ctx->vp_output, 0, sizeof(vp_output_t));
ctx->vp_output->out_num = ctx->nn_cvflow.out_num;

ctx->vp_output->out[i].out = ea_net_output_by_index(ctx->nn_cvflow.net, i);

} while (0);
return rval;
}

static int check_crop(const size_t *shape, ea_roi_t *roi, int *crop_flag)
{
int rval = EA_SUCCESS;
do {
if (roi->x == 0 && roi->y == 0) {
if (((size_t)roi->h == shape[EA_H] &&
(size_t)roi->w == shape[EA_W]) ||
(roi->h == 0 && roi->w == 0)) {
*crop_flag = 0;
} else {
RVAL_ASSERT(roi->h > 0 && roi->w > 0);
if ((size_t)roi->w > shape[EA_W]) {
roi->w = shape[EA_W];
}
if ((size_t)roi->h > shape[EA_H]) {
roi->h = shape[EA_H];
}
*crop_flag = 1;
}
} else {
RVAL_ASSERT(roi->h > 0 && roi->w > 0 && roi->x >= 0 && roi->y >= 0);
if ((size_t)(roi->x + roi->w) > shape[EA_W]) {
roi->w = shape[EA_W] - roi->x;
}
if ((size_t)(roi->y + roi->h) > shape[EA_H]) {
roi->h = shape[EA_H] - roi->y;
}
*crop_flag = 1;
}
} while (0);
return rval;
}
