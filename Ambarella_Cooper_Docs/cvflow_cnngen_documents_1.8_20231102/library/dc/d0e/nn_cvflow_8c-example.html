EA_LOG_DECLARE_LOCAL(EA_LOG_LEVEL_NOTICE);
int nn_cvflow_init(nn_cvflow_t *nn, const nn_cvflow_params_t *params)
{
int rval = EA_SUCCESS;
int i = 0;
ea_net_params_t net_param;
do {
RVAL_ASSERT(nn != NULL);
RVAL_ASSERT(params != NULL);
RVAL_ASSERT(params->model_path != NULL);
EA_LOG_SET_LOCAL(params->log_level);
memset(nn, 0, sizeof(nn_cvflow_t));
memset(&net_param, 0, sizeof(net_param));
net_param.ades_cmd_file = params->ades_cmd_file;
net_param.acinf_gpu_id = params->acinf_gpu_id;
net_param.disable_output_fp32_from_fp16_on_non_fp32_arch = params->disable_output_fp16_2_fp32;
net_param.nvp_affinity = params->nvp_id;
if (params->log_level == EA_LOG_LEVEL_VERBOSE) {
net_param.verbose_print = 1;
}
nn->net = ea_net_new(&net_param);
RVAL_ASSERT(nn->net != NULL);
RVAL_OK(ea_net_load(nn->net, EA_NET_LOAD_FILE, (void *)params->model_path, 1));
nn->in_num = ea_net_input_num(nn->net);
RVAL_ASSERT(nn->in_num <= NN_MAX_PORT_NUM);
for (i = 0; i < nn->in_num; i++) {
nn->in_info[i].name = ea_net_input_name(nn->net, i);
nn->in_info[i].tensor = ea_net_input_by_index(nn->net, i);
}
EA_LOG_NOTICE("network input:\n");
for (i = 0; i < nn->in_num; i++) {
EA_LOG_NOTICE("%s\n", nn->in_info[i].name);
}
nn->out_num = ea_net_output_num(nn->net);
RVAL_ASSERT(nn->out_num <= NN_MAX_PORT_NUM);
for (i = 0; i < nn->out_num; i++) {
nn->out_info[i].name = ea_net_output_name(nn->net, i);
nn->out_info[i].tensor = ea_net_output_by_index(nn->net, i);
}
EA_LOG_NOTICE("network output:\n");
for (i = 0; i < nn->out_num; i++) {
EA_LOG_NOTICE("%s\n", nn->out_info[i].name);
}
} while (0);
if (rval < 0) {
if (nn) {
if (nn->net) {
ea_net_free(nn->net);
nn->net = NULL;
}
}
}
EA_LOG_NOTICE("nn_cvflow_init\n");
return rval;
}
void nn_cvflow_deinit(nn_cvflow_t *nn)
{
if (nn) {
if (nn->net) {
ea_net_free(nn->net);
nn->net = NULL;
}
}
EA_LOG_NOTICE("nn_cvflow_deinit\n");
}
ea_tensor_t *nn_cvflow_input(nn_cvflow_t *nn, int index)
{
return (nn->in_info[index].tensor);
}
int nn_cvflow_inference(nn_cvflow_t *nn)
{
int rval = EA_SUCCESS;
do {
RVAL_OK(ea_net_forward(nn->net, 1));
} while (0);
return rval;
}
