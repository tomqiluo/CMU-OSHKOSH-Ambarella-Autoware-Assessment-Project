int p00a_ea_tensor_new_from_vect_desc_01()
{
int rval = 0;
struct cv_mem vp_mem;
vect_desc_t vp_desc;
int en_cache = 1;
ea_tensor_t *tensor = NULL;
vect_desc_t vp_desc_from_tensor;
do {
memset(&vp_mem, 0, sizeof(vp_mem));
memset(&vp_desc, 0, sizeof(vp_desc));
// vp_desc.shape.p = 1;
vp_desc.shape.w = 100;
vp_desc.shape.h = 200;
vp_desc.shape.d = 3;
vp_desc.pitch = ALIGN_PITCH(vp_desc.shape.w);
vp_desc.color_space = CS_VECT;
if (vp_desc.shape.p == 0) {
vp_mem.size = vp_desc.shape.d * vp_desc.shape.h * vp_desc.pitch;
}
else {
vp_mem.size = vp_desc.shape.p * vp_desc.shape.d * vp_desc.shape.h * vp_desc.pitch;
}
EA_R_OK(ea_env_open(EA_ENV_ENABLE_CAVALRY));
EA_R_OK(cavalry_mem_alloc(&vp_mem.size, &vp_mem.phys, &vp_mem.virt, en_cache));
vp_desc.data_addr = vp_mem.phys;
memset(vp_mem.virt, 100, vp_mem.size);
tensor = ea_tensor_new_from_vect_desc(EA_MEM_CAVALRY, &vp_desc, vp_mem.virt);
EA_R_ASSERT(tensor != NULL);
EA_R_OK(ea_tensor_sync_cache(tensor, EA_CPU, EA_VP));
EA_R_ASSERT(ea_tensor_dtype(tensor) == EA_U8);
EA_R_ASSERT(ea_tensor_data_format(tensor)->sign == vp_desc.data_format.sign);
EA_R_ASSERT(ea_tensor_data_format(tensor)->datasize == vp_desc.data_format.datasize);
EA_R_ASSERT(ea_tensor_data_format(tensor)->exp_offset == vp_desc.data_format.exp_offset);
EA_R_ASSERT(ea_tensor_data_format(tensor)->exp_bits == vp_desc.data_format.exp_bits);
EA_R_ASSERT(ea_tensor_mem_type(tensor) == EA_MEM_CAVALRY);
if (vp_desc.shape.p == 0) {
EA_R_ASSERT(ea_tensor_shape(tensor)[EA_N] == 1);
}
else {
EA_R_ASSERT(ea_tensor_shape(tensor)[EA_N] == vp_desc.shape.p);
}
EA_R_ASSERT(ea_tensor_shape(tensor)[EA_C] == vp_desc.shape.d);
EA_R_ASSERT(ea_tensor_shape(tensor)[EA_H] == vp_desc.shape.h);
EA_R_ASSERT(ea_tensor_shape(tensor)[EA_W] == vp_desc.shape.w);
EA_R_ASSERT(ea_tensor_pitch(tensor) == vp_desc.pitch);
EA_R_ASSERT(ea_tensor_size(tensor) == vp_mem.size);
EA_R_ASSERT(ea_tensor_data(tensor) == vp_mem.virt);
EA_R_ASSERT(ea_tensor_phys(tensor) == vp_mem.phys);
EA_R_ASSERT(*(uint8_t *)ea_tensor_at(tensor, 0, 0, 0, 0) == 100);
EA_R_OK(ea_tensor_vect_desc(tensor, &vp_desc_from_tensor));
EA_R_ASSERT(vp_desc_from_tensor.color_space == CS_VECT);
EA_R_ASSERT(vp_desc_from_tensor.data_addr == vp_desc.data_addr);
EA_R_ASSERT(vp_desc_from_tensor.pitch == vp_desc.pitch);
if (vp_desc.shape.p == 0) {
EA_R_ASSERT(vp_desc_from_tensor.shape.p == 1);
EA_R_ASSERT(vp_desc_from_tensor.shape.d == vp_desc.shape.d);
EA_R_ASSERT(vp_desc_from_tensor.shape.h == vp_desc.shape.h);
EA_R_ASSERT(vp_desc_from_tensor.shape.w == vp_desc.shape.w);
}
else {
EA_R_ASSERT(memcmp(&vp_desc_from_tensor.shape, &vp_desc.shape, sizeof(vect_shape_t)) == 0);
}
EA_R_ASSERT(memcmp(&vp_desc_from_tensor.data_format, &vp_desc.data_format, sizeof(data_format_t)) == 0);
EA_R_ASSERT(memcmp(&vp_desc_from_tensor.roi, &vp_desc.roi, sizeof(roi_area_t)) == 0);
ea_tensor_free(tensor);
cavalry_mem_free(vp_mem.size , vp_mem.phys, vp_mem.virt);
ea_env_close();
} while (0);
if (rval < 0) {
if (tensor) {
ea_tensor_free(tensor);
}
if (vp_mem.size) {
cavalry_mem_free(vp_mem.size , vp_mem.phys, vp_mem.virt);
}
ea_env_close();
}
return rval;
}

int p01a_ea_tensor_sync_cache_invalidate_05()
{
int rval = 0;
ea_tensor_t *from = NULL;
ea_tensor_t *to = NULL;
size_t shape[EA_DIM] = {1, 1, 1, 256};
int has_one = 0;
size_t i;
do {
EA_R_OK(ea_env_open(EA_ENV_ENABLE_CAVALRY | EA_ENV_ENABLE_VPROC));
from = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(from != NULL);
to = ea_tensor_new(EA_U8, shape, 0);
EA_R_ASSERT(to != NULL);
memset(ea_tensor_data(from), 0, ea_tensor_size(from));
EA_R_OK(ea_tensor_sync_cache(from, EA_CPU, EA_VP));
for (i = 0; i < ea_tensor_size(from); i++) {
EA_R_ASSERT(((uint8_t *)ea_tensor_data(from))[i] == 0);
}
EA_R_BREAK();
memset(ea_tensor_data(to), 1, ea_tensor_size(to));
EA_R_OK(ea_tensor_sync_cache(to, EA_CPU, EA_VP));
for (i = 0; i < ea_tensor_size(to); i++) {
EA_R_ASSERT(((uint8_t *)ea_tensor_data(to))[i] == 1);
}
EA_R_BREAK();
EA_R_OK(ea_tensor_memcpy_by_vp(to, from, ea_tensor_size(to)));
has_one = 0;
for (i = 0; i < ea_tensor_size(to); i++) {
if (((uint8_t *)ea_tensor_data(to))[i] == 1) {
has_one = 1;
break;
}
}
EA_R_ASSERT(has_one == 1);
EA_R_OK(ea_tensor_sync_cache(to, EA_VP, EA_CPU)); // invalidate
has_one = 0;
for (i = 0; i < ea_tensor_size(to); i++) {
if (((uint8_t *)ea_tensor_data(to))[i] == 1) {
has_one = 1;
break;
}
}
EA_R_ASSERT(has_one == 0);
ea_tensor_free(from);
ea_tensor_free(to);
ea_env_close();
} while (0);
if (rval < 0) {
if (from) {
ea_tensor_free(from);
}
if (to) {
ea_tensor_free(to);
}
ea_env_close();
}
return rval;
}
