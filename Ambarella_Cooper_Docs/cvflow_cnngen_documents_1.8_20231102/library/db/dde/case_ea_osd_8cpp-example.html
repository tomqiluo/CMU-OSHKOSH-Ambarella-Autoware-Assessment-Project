int p03m_ea_osd_vout_01()
{
int rval = 0;
ea_env_arch_t arch;
const ea_env_properties_t *env_prop = NULL;
ea_osd_t *osd = NULL;
int fb_id = -1;
const ea_osd_properties_t *osd_prop = NULL;
int local_area_id = 0;
const ea_osd_area_buffer_t *osd_area = NULL;
int i, k;
uint8_t *pattern_buf = NULL;
int h, pattern_index;
int success_flag = 0;
do {
EA_R_OK(ea_env_open(EA_ENV_ENABLE_OSD_VOUT));
env_prop = ea_env_properties();
EA_R_ASSERT(env_prop != NULL);
EA_R_ASSERT(env_prop->vout.frame_buffer_color_table_size == 256);
arch = ea_env_arch();
switch (arch) {
case EA_ENV_ARCH_CV22:
printf("ARCH CV22\n");
fb_id = 1;
break;
case EA_ENV_ARCH_CV2:
printf("ARCH CV2\n");
fb_id = 1;
break;
case EA_ENV_ARCH_CV25:
printf("ARCH CV25\n");
fb_id = 1;
break;
case EA_ENV_ARCH_CV28:
printf("ARCH CV28\n");
fb_id = 0;
break;
case EA_ENV_ARCH_CV5:
printf("ARCH CV5\n");
fb_id = 2;
break;
case EA_ENV_ARCH_CV52:
printf("ARCH CV52\n");
fb_id = 2;
break;
case EA_ENV_ARCH_CV72:
printf("ARCH CV72\n");
fb_id = 2;
break;
default :
printf("ARCH not test\n");
break;
}
EA_R_ASSERT(arch != EA_ENV_ARCH_INVALID);
if (fb_id == -1) { // this platform doesn't support frame buffer, pass this case
success_flag = 1;
rval = -1;
break;
}
osd = ea_osd_new(EA_OSD_VOUT, fb_id, NULL);
EA_R_ASSERT(osd != NULL);
osd_prop = ea_osd_properties(osd);
EA_R_ASSERT(osd_prop != NULL);
EA_R_ASSERT(osd_prop->type == EA_OSD_VOUT);
EA_R_OK(ea_osd_area_set_default_16_color_table(osd, local_area_id));
pattern_buf = (uint8_t *) malloc(osd_prop->resolution_width * env_prop->vout.frame_buffer_color_table_size * 32);
EA_R_ASSERT(pattern_buf != NULL);
for (i = 0; i < env_prop->vout.frame_buffer_color_table_size; i++) {
memset(pattern_buf + 32 * i * osd_prop->resolution_width, i, 32 * osd_prop->resolution_width);
}
EA_R_OK(manual_confirm("Ready to check display on VOUT screen?"));
printf("Displaying on VOUT screen for 30 seconds ...\n");
printf("Please view the moving of the color pattern.\n");
for (k = 0; k < 3; k++) {
for (i = 0; i < env_prop->vout.frame_buffer_color_table_size; i++) {
osd_area = ea_osd_area_current_buffer(osd, local_area_id, NULL);
EA_R_ASSERT(osd_area != NULL);
EA_R_ASSERT(osd_area->type == EA_OSD_AREA_BUFFER_TYPE_CLUT_8BPP);
EA_R_ASSERT((int)osd_area->width == osd_prop->resolution_width);
EA_R_ASSERT((int)osd_area->height == osd_prop->resolution_height);
// cv::Mat area_mat = cv::Mat(osd_area->height, osd_area->width, CV_8UC1, osd_area->buf, osd_area->pitch);
pattern_index = i * 32;
for (h = 0; h < (int)osd_area->height; h++) {
memcpy(osd_area->buf + h * osd_area->pitch,
pattern_buf + pattern_index * osd_prop->resolution_width,
osd_prop->resolution_width);
pattern_index++;
if (pattern_index == env_prop->vout.frame_buffer_color_table_size * 32) {
pattern_index = 0;
}
}
EA_R_OK(ea_osd_refresh(osd, NULL));
usleep(33 * 1000);
}
EA_R_BREAK();
}
EA_R_BREAK();
EA_R_OK(manual_confirm("Is the color pattern on the screen moving smoothly without glitch?"));
free(pattern_buf);
ea_osd_free(osd);
ea_env_close();
} while (0);
if (rval < 0) {
if (pattern_buf) {
free(pattern_buf);
}
if (osd) {
ea_osd_free(osd);
}
ea_env_close();
}
if (success_flag) {
rval = 0;
}
return rval;
}

osd_a = ea_osd_new(EA_OSD_STREAM, 0, NULL);

osd_a_area = ea_osd_area_current_buffer(osd_a, local_area_id, NULL);
osd_b_area = ea_osd_area_current_buffer(osd_b, local_area_id, NULL);
EA_R_ASSERT(osd_a_area->type == EA_OSD_AREA_BUFFER_TYPE_CLUT_8BPP);
memset(osd_a_area->buf, EA_OSD_16_COLORS_WHITE * 16 + 15, osd_a_area->height * osd_a_area->pitch);
EA_R_ASSERT(osd_b_area->type == EA_OSD_AREA_BUFFER_TYPE_CLUT_8BPP);
memset(osd_b_area->buf, EA_OSD_16_COLORS_GREEN * 16 + 15, osd_b_area->height * osd_b_area->pitch);
EA_R_OK(ea_osd_refresh(osd_a, NULL));
EA_R_OK(ea_osd_refresh(osd_b, NULL));
EA_R_OK(manual_confirm("Stream A turns white and stream B turns green?"));
ea_osd_free(osd_a);
ea_osd_free(osd_b);
ea_env_close();
} while (0);
if (rval < 0) {
if (osd_a) {
ea_osd_free(osd_a);
}
if (osd_b) {
ea_osd_free(osd_b);
}
ea_env_close();
}
return rval;
}
