nn_in_context_type_t *nn_in_live_init(nn_in_live_params_t *params)
{
int rval = EA_SUCCESS;
nn_in_context_type_t *ctx = NULL;
do {
RVAL_ASSERT(params != NULL);
ctx = (nn_in_context_type_t *)malloc(sizeof(*ctx));
RVAL_ASSERT(ctx != NULL);
memset(ctx, 0, sizeof(*ctx));
memcpy(&ctx->live_params, params, sizeof(*params));
if((params->in_type_map & NN_IN_SRC_PYRAMID_BUF) == NN_IN_SRC_PYRAMID_BUF) {
ctx->img_resource[0] = ea_img_resource_new(EA_PYRAMID,
(void *)(unsigned long)params->pyramid[0]);
} else {
ctx->img_resource[0] = ea_img_resource_new(EA_CANVAS,
(void *)(unsigned long)params->canvas_id);
}
RVAL_ASSERT(ctx->img_resource[0] != NULL);
} while (0);
if (rval < 0) {
if (ctx != NULL) {
free(ctx);
ctx = NULL;
}
}
return ctx;
}
int nn_in_live_query(nn_in_context_type_t *ctx, ea_img_resource_data_t *img)
{
int rval = EA_SUCCESS;
do {
RVAL_ASSERT(img != NULL);
ea_img_resource_drop_data(ctx->img_resource[0], img);
RVAL_OK(ea_img_resource_hold_data(ctx->img_resource[0], img));
} while (0);
return rval;
}
void nn_in_live_deinit(nn_in_context_type_t *ctx)
{
if (ctx != NULL) {
if (ctx->img_resource[0] != NULL) {
ea_img_resource_free(ctx->img_resource[0]);
ctx->img_resource[0] = NULL;
}
}
}
nn_in_context_type_t *nn_in_file_init(nn_in_file_params_t *params)
{
int i;
int rval = EA_SUCCESS;
nn_in_context_type_t *ctx = NULL;
struct stat s_buf;
do {
RVAL_ASSERT(params != NULL);
ctx = (nn_in_context_type_t *)malloc(sizeof(*ctx));
RVAL_ASSERT(ctx != NULL);
memset(ctx, 0, sizeof(*ctx));
memcpy(&ctx->file_params, params, sizeof(*params));
for (i = 0; i < params->multi_in_num; i++) {
if (params->file_type[i] == FILE_RAW) {
stat(params->in_dir[i], &s_buf);
if (S_ISDIR(s_buf.st_mode)) {
ctx->img_resource[i] =
ea_img_resource_new(EA_RAW_FILE_FOLDER, (void *)params->in_dir[i]);
} else {
ctx->img_resource[i] =
ea_img_resource_new(EA_RAW_FILE, (void *)params->in_dir[i]);
}
} else {
ctx->img_resource[i] =
ea_img_resource_new(EA_JPEG_FOLDER, (void *)params->in_dir[i]);
}
RVAL_ASSERT(ctx->img_resource[i] != NULL);
}
RVAL_BREAK();
} while (0);
if (rval < 0) {
if (ctx != NULL) {
free(ctx);
ctx = NULL;
}
}
return ctx;
}
static int file_in_get_resource_index_by_name(nn_in_context_type_t *ctx, const char *name)
{
int rval = -1;
int i;
for (i = 0; i < ctx->file_params.multi_in_num; i++) {
if (strcmp(ctx->file_params.port_name[i], name) == 0) {
rval = i;
}
}
return rval;
}
int nn_in_file_query(nn_in_context_type_t *ctx, ea_img_resource_data_t *img, const char *name)
{
int rval = EA_SUCCESS;
int index;
do {
RVAL_ASSERT(name != NULL);
RVAL_ASSERT(img != NULL);
index = file_in_get_resource_index_by_name(ctx, name);
RVAL_ASSERT(index >= 0);
ea_img_resource_drop_data(ctx->img_resource[index], img);
RVAL_OK(ea_img_resource_hold_data(ctx->img_resource[index], img));
} while (0);
return rval;
}
int nn_in_file_set_file_info(nn_in_context_type_t *ctx, const ea_img_resource_file_info_t *info, int index)
{
int rval = EA_SUCCESS;
do {
RVAL_ASSERT(ctx != NULL);
RVAL_ASSERT(index >= 0 && index < ctx->file_params.multi_in_num);
rval = ea_img_resource_set_file_info(ctx->img_resource[index], info);
} while (0);
return rval;
}
void nn_in_file_deinit(nn_in_context_type_t *ctx)
{
int i;
if (ctx != NULL) {
for (i = 0; i < ctx->file_params.multi_in_num; i++) {
if (ctx->img_resource[i] != NULL) {
ea_img_resource_free(ctx->img_resource[i]);
ctx->img_resource[i] = NULL;
}
}
}
